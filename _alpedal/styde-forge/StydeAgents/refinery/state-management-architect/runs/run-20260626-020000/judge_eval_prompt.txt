## 1. XState v5 — State Machines & Actors

XState v5 brings a streamlined actor model with first-class TypeScript support. Use it when your UI state has clearly defined discrete states, complex transitions, or parallel concerns (e.g., a multi-step wizard, an audio player, or a checkout flow).

### 1.1 Finite State Machine Setup

```typescript
// machines/checkoutMachine.ts
import { setup, assign } from 'xstate';

interface CheckoutContext {
  cart: { productId: string; name: string; price: number; quantity: number }[];
  shippingAddress: {
    line1: string;
    city: string;
    zip: string;
  } | null;
  paymentToken: string | null;
  error: string | null;
}

type CheckoutEvent =
  | { type: 'SUBMIT_SHIPPING'; address: CheckoutContext['shippingAddress'] }
  | { type: 'SUBMIT_PAYMENT'; token: string }
  | { type: 'CONFIRM' }
  | { type: 'RETRY' }
  | { type: 'CANCEL' }
  | { type: 'BACK' };

export const checkoutMachine = setup({
  types: {
    context: {} as CheckoutContext,
    events: {} as CheckoutEvent,
  },
  actions: {
    clearError: assign({ error: null }),
    setShipping: assign(({ event }) =>
      event.type === 'SUBMIT_SHIPPING'
        ? { shippingAddress: event.address }
        : {}
    ),
    setPayment: assign(({ event }) =>
      event.type === 'SUBMIT_PAYMENT'
        ? { paymentToken: event.token }
        : {}
    ),
    setError: assign(({ event }) => ({
      error: `Step failed: ${(event as any).message ?? 'unknown'}`,
    })),
  },
  guards: {
    hasCart: ({ context }) => context.cart.length > 0,
    hasAddress: ({ context }) => context.shippingAddress !== null,
    hasPayment: ({ context }) => context.paymentToken !== null,
  },
}).createMachine({
  id: 'checkout',
  initial: 'cart',
  context: {
    cart: [],
    shippingAddress: null,
    paymentToken: null,
    error: null,
  },
  states: {
    cart: {
      on: {
        SUBMIT_SHIPPING: {
          target: 'shipping',
          guard: 'hasCart',
        },
      },
    },
    shipping: {
      entry: 'clearError',
      on: {
        SUBMIT_SHIPPING: {
          target: 'payment',
          guard: 'hasAddress',
          actions: 'setShipping',
        },
        BACK: 'cart',
      },
    },
    payment: {
      entry: 'clearError',
      on: {
        SUBMIT_PAYMENT: {
          target: 'review',
          guard: 'hasPayment',
          actions: 'setPayment',
        },
        BACK: 'shipping',
      },
    },
    review: {
      on: {
        CONFIRM: 'confirmed',
        BACK: 'payment',
        CANCEL: 'cart',
      },
    },
    confirmed: {
      type: 'final',
    },
    error: {
      entry: 'setError',
      on: {
        RETRY: 'shipping',
        CANCEL: 'cart',
      },
    },
  },
});
```

```typescript
// machines/authFlowMachine.ts — guards, delayed transitions, and entry actions
import { setup, assign, fromPromise } from 'xstate';

interface AuthContext {
  user: { id: string; email: string } | null;
  attempts: number;
  cooldownUntil: number | null;
}

type AuthEvent =
  | { type: 'LOGIN'; email: string; password: string }
  | { type: 'LOGOUT' }
  | { type: 'REFRESH' };

// Actor: async login service
const loginService = fromPromise(
  async ({ input }: { input: { email: string; password: string } }) => {
    const res = await fetch('/api/auth/login', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(input),
    });
    if (!res.ok) throw new Error(`Login failed: ${res.status}`);
    return res.json() as Promise<{ id: string; email: string }>;
  }
);

export const authFlowMachine = setup({
  types: {
    context: {} as AuthContext,
    events: {} as AuthEvent,
  },
  actors: { loginService },
  actions: {
    setUser: assign(({ event }) =>
      event.type === 'LOGIN' && 'output' in event
        ? { user: (event as any).output }
        : { user: null }
    ),
    incrementAttempts: assign({
      attempts: ({ context }) => context.attempts + 1,
    }),
    setCooldown: assign({
      cooldownUntil: () => Date.now() + 30_000, // 30-second cooldown
    }),
  },
  guards: {
    maxAttemptsReached: ({ context }) => context.attempts >= 5,
    cooldownElapsed: ({ context }) =>
      context.cooldownUntil === null || Date.now() > context.cooldownUntil,
    underMaxAttempts: ({ context }) => context.attempts < 5,
  },
}).createMachine({
  id: 'auth',
  initial: 'unauthenticated',
  context: { user: null, attempts: 0, cooldownUntil: null },
  states: {
    unauthenticated: {
      initial: 'idle',
      states: {
        idle: {
          on: { LOGIN: 'authenticating' },
        },
        authenticating: {
          invoke: {
            src: 'loginService',
            input: ({ event }) =>
              event.type === 'LOGIN'
                ? { email: event.email, password: event.password }
                : { email: '', password: '' },
            onDone: {
              target: '#auth.authenticated',
              actions: 'setUser',
            },
            onError: [
              {
                target: 'rateLimited',
                guard: 'maxAttemptsReached',
              },
              {
                target: 'idle',
                actions: 'incrementAttempts',
              },
            ],
          },
        },
        rateLimited: {
          entry: 'setCooldown',
          after: {
            30000: {
              target: 'idle',
              guard: 'cooldownElapsed',
            },
          },
        },
      },
    },
    authenticated: {
      id: 'auth.authenticated',
      on: {
        LOGOUT: {
          target: 'unauthenticated',
          actions: assign({ user: null }),
        },
      },
    },
  },
});
```

```typescript
type VolumeLevel = 'normal' | 'muted';

interface PlayerContext {
  currentTrack: { id: string; title: string; duration: number } | null;
  elapsed: number;
  volume: number; // 0–1

  | { type: 'PLAY' }
  | { type: 'PAUSE' }
  | { type: 'STOP' }
  | { type: 'TOGGLE_MUTE' }

  | { type: 'NEXT' }
  | { type: 'PREVIOUS' }
  | { type: 'TICK'; elapsed: number };

    context: {} as PlayerContext,
    events: {} as PlayerEvent,

      queueIndex: ({ context }) =>
```

```typescript
// machines/notificationPoller.ts — long-running actor with lifecycle
import { setup, fromPromise, fromCallback } from 'xstate';

interface PollerContext {
  lastFetched: number | null;
  notifications: { id: string; message: string; read: boolean }[];
}

export const notificationPoller = setup({
  types: { context: {} as PollerContext },
  actors: {
    fetchNotifications: fromPromise(async () => {
      const res = await fetch('/api/notifications?since=' + Date.now());
      return res.json();
    }),
    intervalCallback: fromCallback<{ type: 'TICK' }>(({ sendBack }) => {
      const id = setInterval(() => sendBack({ type: 'TICK' }), 30_000); // poll every 30s
      return () => clearInterval(id);
    }),
  },
}).createMachine({
  id: 'notifications',
  initial: 'idle',
  context: { lastFetched: null, notifications: [] },
  states: {
    idle: {
      always: 'polling', // transition immediately to polling
    },
    polling: {
      invoke: [
        {
          src: 'fetchNotifications',
          onDone: {
            actions: assign({
              notifications: ({ event }) => event.output,
              lastFetched: () => Date.now(),
            }),
          },
        },
        {
          src: 'intervalCallback',
        },
      ],
      on: {
        TICK: {
          target: 'polling',
          reenter: true, // re-invoke fetchNotifications on each tick
        },
      },
    },
  },
```

## 2. Valtio — Proxy-Based Reactive State

Valtio wraps plain JavaScript objects/arrays in reactive proxies. Mutate state directly (no setter functions, no immutability ceremony) and Valtio tracks which components consume which properties — triggering only the renders that need to change.

### 2.1 Proxy Stores & Mutable Updates

```typescript
// stores/appState.ts
import { proxy } from 'valtio';

interface User {
  id: string;
  email: string;
  displayName: string;
  preferences: {
    theme: 'light' | 'dark' | 'system';
    notifications: boolean;
  };
}

interface Project {
  id: string;
  name: string;
  tasks: { id: string; title: string; done: boolean }[];
}

interface AppState {
  user: User | null;
  projects: Project[];
  ui: {
    sidebarOpen: boolean;
    activeModal: string | null;
    toasts: { id: string; message: string; type: 'info' | 'error' }[];
  };
}

export const appState = proxy<AppState>({
  user: null,
  projects: [],
  ui: {
    sidebarOpen: true,
    activeModal: null,
    toasts: [],
  },
});

// ---------- Mutations (plain functions, no set() ceremony) ----------

export function loginUser(user: User) {
  appState.user = user;
}

export function logoutUser() {
  appState.user = null;
  appState.projects = [];
}

export function addProject(project: Project) {
  appState.projects.push(project);
}

export function toggleTask(projectId: string, taskId: string) {
  const project = appState.projects.find((p) => p.id === projectId);
  if (!project) return;
  const task = project.tasks.find((t) => t.id === taskId);
  if (task) task.done = !task.done; // mutable update — just flip the boolean
}

export function renameProject(projectId: string, newName: string) {
  const project = appState.projects.find((p) => p.id === projectId);
  if (project) project.name = newName;
}

export function addToast(message: string, type: 'info' | 'error' = 'info') {
  const id = crypto.randomUUID();
  appState.ui.toasts.push({ id, message, type });
  setTimeout(() => {
    appState.ui.toasts = appState.ui.toasts.filter((t) => t.id !== id);
  }, 4000);
}
```

### 2.2 Derived Proxies & Snapshots

```typescript
// stores/derivedState.ts
import { derive, snapshot } from 'valtio';
import { appState } from './appState';

// Derived proxy — automatically recomputed; components reading it
// only re-render when the specific derived value changes
export const dashboardSummary = derive({
  taskCount: (get) => {
    let total = 0;
    let done = 0;
    for (const p of get(appState).projects) {
      total += p.tasks.length;
      done += p.tasks.filter((t) => get(t).done).length; // fine-grained: track each task
    }
    return { total, done, pending: total - done };
  },
  activeProjectName: (get) => {
    const userPrefs = get(appState).user?.preferences;
    if (!userPrefs) return null;
    // Access a specific project name — only triggers if that project name changes
    return get(appState).projects[0]?.name ?? null;
  },
});

// Snapshot: get an immutable plain-object copy at a point in time
// Useful for serialization, logging, or passing to non-reactive code
export function saveSnapshot(): AppState {
  return snapshot(appState);
}
```

```typescript
// stores/subscriptions.ts
import { subscribe, watch } from 'valtio';
import { appState } from './appState';

// subscribe: callback fires whenever *any* property accessed inside the callback changes
const unsub = subscribe(appState, () => {

  localStorage.setItem('appState', JSON.stringify(snapshot(appState)));

// watch: low-level callback receiving the specific operation path
// Useful for fine-grained side effects, analytics, or undo/redo stacks
watch((get) => {
  // Track only the user's theme preference
  const theme = get(appState).user?.preferences.theme;
  if (theme) {
    document.documentElement.setAttribute('data-theme', theme);
  }
});

// Undo/redo with history snapshots
const history: AppState[] = [];
const MAX_HISTORY = 50;

subscribe(appState, () => {
  history.push(snapshot(appState));
  if (history.length > MAX_HISTORY) history.shift();
});

export function undo() {
  const prev = history.pop();
  if (prev) Object.assign(appState, prev);
}
```

```typescript
export function optimisticAddProject(project: Omit<Project, 'id'>) {

  const optimistic: Project = { id: tempId, ...project };

  remoteProjects$.push(optimistic);

  // Server will return the real ID; Legend-State auto-reconciles
  fetch('/api/projects', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(project),
  })
    .then((res) => res.json())
    .then((created: Project) => {
      const idx = remoteProjects$.findIndex((p) => p.id.get() === tempId);
      if (idx >= 0) remoteProjects$[idx].set(created);
    })
    .catch(() => {
      remoteProjects$.set((prev) => prev.filter((p) => p.id !== tempId));
    });
}
```

## 5. Offline-First State with RxDB

RxDB is a reactive, offline-first database that runs in the browser (IndexedDB, OPFS, SQLite via OPFS). ### 5.4 Conflict Resolution Strategies

```typescript
// db/conflictResolution.ts
import { getDatabase } from './setup';
import type { RxConflictHandler, RxDocumentData } from 'rxdb';

// Strategy 1: Last-Write-Wins (default for simple data)
const lastWriteWinsHandler: RxConflictHandler<any> = {
  isEqual(a: RxDocumentData<any>, b: RxDocumentData<any>): boolean {
    return a.updatedAt === b.updatedAt;
  },
  resolve(a: RxDocumentData<any>, b: RxDocumentData<any>): RxDocumentData<any> {
    // The document with the most recent updatedAt wins
    return new Date(a.updatedAt) > new Date(b.updatedAt) ? a : b;
  },
};

// Strategy 2: Merge conflicts for task fields (field-level resolution)
const taskMergeHandler: RxConflictHandler<any> = {
  isEqual(a, b) {
    // Consider equal if all non-meta fields match
    const { _rev: _a, updatedAt: _ua, ...fieldsA } = a;
    const { _rev: _b, updatedAt: _ub, ...fieldsB } = b;
    return JSON.stringify(fieldsA) === JSON.stringify(fieldsB);
  },
  resolve(a: any, b: any): any {
    // Take the most recent update per field, except description which we merge
    const winner = new Date(a.updatedAt) > new Date(b.updatedAt) ? a : b;
    const loser = winner === a ? b : a;

    return {
      ...winner,
      // Merge description: append loser's description if different
      description:
        a.description !== b.description
          ? `${winner.description}\n\n--- Conflict Resolution ---\n${loser.description}`
          : winner.description,
      // Take the higher priority if changed in both
      priority:
        a.priority !== b.priority
          ? ['high', 'medium', 'low'].indexOf(a.priority) <
            ['high', 'medium', 'low'].indexOf(b.priority)
            ? a.priority
            : b.priority
          : winner.priority,
      updatedAt: new Date().toISOString(),
      _rev: winner._rev,
    };
  },
};

// Strategy 3: Custom conflict handler with user intervention flag
const userResolutionHandler: RxConflictHandler<any> = {
  isEqual(a, b) {
    return a._rev === b._rev;
  },
  resolve(_a, _b) {
    // Flag for user resolution — don't auto-resolve
    return {
      ..._a,
      _conflict: true,
      _conflictWith: _b,
      conflictDetectedAt: new Date().toISOString(),
    };
  },
};

// Register conflict handler per collection
export async function setupConflictHandlers() {
  const db = await getDatabase();

  db.tasks.addConflictHandler(taskMergeHandler);
  db.projects.addConflictHandler(lastWriteWinsHandler);
  db.comments.addConflictHandler(lastWriteWinsHandler);
}
```

```tsx
// Optimistic mutation with rollback
export function useTaskMutations() {
  const db = useRxDB<TaskDatabase>();

  const addTask = async (task: Omit<TaskDoc, 'id' | 'createdAt' | 'updatedAt' | '_rev' | '_deleted'>) => {
    const id = crypto.randomUUID();
    const now = new Date().toISOString();
    const doc = {
      id,
      ...task,
      createdAt: now,
      updatedAt: now,
      _deleted: false,
    };

    // Optimistic insert — appears immediately in UI
    const inserted = await db.tasks.insert(doc as any);
    return inserted;
  };

  const toggleTaskStatus = async (taskId: string) => {
    const doc = await db.tasks.findOne({ selector: { id: taskId } }).exec();
    if (!doc) return;
    const nextStatus =
      doc.status === 'todo'
        ? 'in_progress'
        : doc.status === 'in_progress'
          ? 'done'
          : 'todo';
    // patch() is atomic and reactive — subscribed components update instantly
    await doc.patch({ status: nextStatus, updatedAt: new Date().toISOString() });
  };

  return { addTask, toggleTaskStatus, deleteTask: softDeleteTask };
}
```

## 6. Unified Architecture Patterns

### When to Use Each Tool

| State Concern | Primary Tool | Rationale |
|---|---|---|
| **Multi-step flows with discrete states** (wizards, checkout, auth) | XState v5 | Exhaustive state coverage, visualizable, prevents impossible states |
| **Mutable global app state** (user, UI panel visibility, theme) | Valtio | Minimal boilerplate, familiar imperative mutations, property-level reactivity |
| **Real-time, high-frequency data** (dashboards, collaborative editing, live feeds) | Legend-State | Observable primitives with fine-grained DOM updates — no VDOM diff |
| **Server cache & async data** (REST/GraphQL API data) | TanStack Query | Stale-while-revalidate, cache dedup, mutations, retry, pagination |
| **URL-derived state** (filters, pagination, sort order, active tab) | TanStack Router | URL as single source of truth, shareable/bookmarkable state, validated search params |
| **Offline-first persistent data** (task manager, notes, field-service apps) | RxDB | Built-in replication, conflict resolution, reactive queries over local DB |

> **Cycle 2 Summary:** This guide extends the Cycle 1 foundation (Zustand, TanStack Query, Jotai) with five advanced state management patterns. XState v5 handles discrete state machines and long-running actors. Valtio provides proxy-based mutable state with zero-setter overhead. Legend-State delivers fine-grained reactivity for high-frequency real-time data. TanStack Router elevates URL search params and loader data to first-class typed state. RxDB enables offline-first architectures with built-in replication and conflict resolution. These tools compose naturally — choose the right primitive for each state concern rather than forcing one library to handle everything.
