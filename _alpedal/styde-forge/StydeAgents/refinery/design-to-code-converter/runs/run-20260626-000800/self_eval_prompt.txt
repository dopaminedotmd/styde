```typescript
// extractTokens.ts — Figma plugin to extract design tokens from styles
// Uses Figma Plugin API. Run inside Figma via Plugins → Development → Import plugin from manifest.

interface DesignToken {
  name: string;
  type: 'color' | 'typography' | 'effect' | 'grid';
  value: any;
  description?: string;
}

interface TokenCollection {
  colors: Record<string, string>;
  typography: {
    fontFamilies: Record<string, string>;
    fontSizes: Record<string, { value: number; unit: 'px' | 'rem' }>;
    fontWeights: Record<string, number>;
    lineHeights: Record<string, number>;
    letterSpacings: Record<string, { value: number; unit: 'px' | 'em' }>;
  };
  spacing: Record<string, string>;
  borderRadius: Record<string, string>;
  shadows: Record<string, string>;
  breakpoints: Record<string, string>;
}

// ─── Color extraction ───
function extractColorTokens(): Record<string, string> {
  const colors: Record<string, string> = {};

  // Extract from Figma local paint styles
  figma.getLocalPaintStyles().forEach((style) => {
    const paint = style.paints[0];
    if (paint && paint.type === 'SOLID') {
      const { r, g, b } = paint.color;
      const opacity = paint.opacity ?? 1;
      const hex = rgbaToHex(r, g, b, opacity);
      const tokenName = sanitizeTokenName(`color-${style.name}`);
      colors[tokenName] = hex;
    }
  });

  // Extract from named color variables (Figma Variables API)
  figma.variables.getLocalVariables('COLOR').forEach((variable) => {
    const tokenName = sanitizeTokenName(`color-${variable.name}`);
    const value = variable.valuesByMode[Object.keys(variable.valuesByMode)[0]];
    if (value && typeof value === 'object' && 'r' in value) {
      colors[tokenName] = rgbaToHex(value.r, value.g, value.b, 1);
    }
  });

  return colors;
}

// ─── Typography extraction ───
function extractTypographyTokens(): TokenCollection['typography'] {
  const typography: TokenCollection['typography'] = {
    fontFamilies: {},
    fontSizes: {},
    fontWeights: {},
    lineHeights: {},
    letterSpacings: {},
  };

  figma.getLocalTextStyles().forEach((style) => {
    const tokenName = sanitizeTokenName(`text-${style.name}`);

    typography.fontFamilies[tokenName] = style.fontName.family;
    typography.fontSizes[tokenName] = { value: style.fontSize, unit: 'px' };
    typography.fontWeights[tokenName] = style.fontName.style === 'Regular' ? 400
      : style.fontName.style === 'Medium' ? 500
      : style.fontName.style === 'Semi Bold' ? 600
      : style.fontName.style === 'Bold' ? 700 : 400;
    typography.lineHeights[tokenName] = style.lineHeight.unit === 'PIXELS'
      ? style.lineHeight.value
      : style.lineHeight.value / 100;
    typography.letterSpacings[tokenName] = style.letterSpacing.unit === 'PIXELS'
      ? { value: style.letterSpacing.value, unit: 'px' }
      : { value: style.letterSpacing.value / 100, unit: 'em' };
  });

  return typography;
}

// ─── Effect (shadow) extraction ───
function extractShadowTokens(): Record<string, string> {
  const shadows: Record<string, string> = {};

  figma.getLocalEffectStyles().forEach((style) => {
    const tokenName = sanitizeTokenName(`shadow-${style.name}`);
    const cssShadow = style.effects
      .filter((e) => e.type === 'DROP_SHADOW' || e.type === 'INNER_SHADOW')
      .map((effect) => {
        if (effect.type === 'DROP_SHADOW' || effect.type === 'INNER_SHADOW') {
          const inner = effect.type === 'INNER_SHADOW' ? 'inset ' : '';
          return `${inner}${effect.offset.x}px ${effect.offset.y}px ${effect.radius}px ${effect.spread ?? 0}px ${rgbaToHex(effect.color.r, effect.color.g, effect.color.b, effect.color.a)}`;
        }
        return '';
      })
      .filter(Boolean)
      .join(', ');
    shadows[tokenName] = cssShadow;
  });

  return shadows;
}

// ─── Utility ───
function sanitizeTokenName(name: string): string {
  return name
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, '-')
    .replace(/^-|-$/g, '')
    .replace(/--+/g, '-');
}

function rgbaToHex(r: number, g: number, b: number, a: number = 1): string {
  const toHex = (n: number) =>
    Math.round(n * 255)
      .toString(16)
      .padStart(2, '0');
  const hex = `#${toHex(r)}${toHex(g)}${toHex(b)}`;
  return a < 1 ? `${hex}${toHex(a)}` : hex;
}

// ─── Main export ───

      'spacing-3': '12px',
      'spacing-4': '16px',
      'spacing-5': '20px',

      'spacing-8': '32px',

      'spacing-12': '48px',
      'spacing-16': '64px',
      'spacing-20': '80px',
      'spacing-24': '96px',
    },

      'radius-none': '0',
```

```typescript
// tokensToCSS.ts — Convert extracted token JSON to CSS custom properties
// Run as: npx ts-node tokensToCSS.ts tokens.json tokens.css

import * as fs from 'fs';

interface TokenCollection {
  colors: Record<string, string>;
  typography: {
    fontFamilies: Record<string, string>;
    fontSizes: Record<string, { value: number; unit: string }>;
    fontWeights: Record<string, number>;
    lineHeights: Record<string, number>;
    letterSpacings: Record<string, { value: number; unit: string }>;
  };
  spacing: Record<string, string>;
  borderRadius: Record<string, string>;
  shadows: Record<string, string>;
  breakpoints: Record<string, string>;
}

function generateCSS(tokens: TokenCollection): string {
  const lines: string[] = [
    '/* ================================================================',
    '   DESIGN TOKENS — Auto-generated from Figma',
    '   DO NOT EDIT MANUALLY. Run tokensToCSS.ts to regenerate.',
    '   ================================================================ */',
    '',
    ':root {',
  ];

  // ─── Colors ───
  lines.push('  /* ── Colors ── */');
  for (const [name, value] of Object.entries(tokens.colors)) {
    lines.push(`  --${name}: ${value};`);
  }

  // ─── Typography ───
  lines.push('', '  /* ── Typography ── */');
  for (const [name, value] of Object.entries(tokens.typography.fontFamilies)) {
    lines.push(`  --font-family-${name.replace('text-', '')}: '${value}', sans-serif;`);
  }
  for (const [name, value] of Object.entries(tokens.typography.fontSizes)) {
    const pxValue = value.unit === 'px' ? value.value : value.value * 16;
    const remValue = (pxValue / 16).toFixed(4);
    lines.push(`  --font-size-${name.replace('text-', '')}: ${remValue}rem; /* ${pxValue}px */`);
  }
  for (const [name, value] of Object.entries(tokens.typography.fontWeights)) {
    lines.push(`  --font-weight-${name.replace('text-', '')}: ${value};`);
  }
  for (const [name, value] of Object.entries(tokens.typography.lineHeights)) {
    lines.push(`  --line-height-${name.replace('text-', '')}: ${value};`);
  }
  for (const [name, value] of Object.entries(tokens.typography.letterSpacings)) {
    lines.push(`  --letter-spacing-${name.replace('text-', '')}: ${value.value}${value.unit};`);
  }

  // ─── Spacing ───
  lines.push('', '  /* ── Spacing ── */');
  for (const [name, value] of Object.entries(tokens.spacing)) {
    lines.push(`  --${name}: ${value};`);
  }

  // ─── Border Radius ───
  lines.push('', '  /* ── Border Radius ── */');
  for (const [name, value] of Object.entries(tokens.borderRadius)) {
    lines.push(`  --${name}: ${value};`);
  }

  // ─── Shadows ───
  lines.push('', '  /* ── Shadows ── */');
  for (const [name, value] of Object.entries(tokens.shadows)) {
    lines.push(`  --${name}: ${value};`);
  }

  // ─── Breakpoints ───
  lines.push('', '  /* ── Breakpoints ── */');
  for (const [name, value] of Object.entries(tokens.breakpoints)) {
    lines.push(`  --${name}: ${value};`);
  }

  lines.push('}');

  // ─── Dark theme overrides ───
  lines.push('', '/* ── Dark Theme ── */', '[data-theme="dark"] {');
  // In practice, map light tokens to dark equivalents here
  // e.g.: `--color-surface-primary: var(--color-gray-900);`
  lines.push('  /* Dark overrides would go here */');
  lines.push('}', '');

  return lines.join('\n');
}
```

## 2. Figma Auto-Layout → CSS Grid / Flexbox Mapping

### 2.1 Mapping Philosophy

Figma's Auto-Layout is a 1D layout system. The mapping strategy:

| Figma Auto-Layout Property     | CSS Equivalent                        |
|-------------------------------|---------------------------------------|
| `layoutMode: HORIZONTAL`      | `display: flex; flex-direction: row`  |
| `layoutMode: VERTICAL`        | `display: flex; flex-direction: column` |
| `primaryAxisAlignItems`       | `justify-content`                     |
| `counterAxisAlignItems`       | `align-items`                         |
| `itemSpacing`                 | `gap`                                 |
| `paddingLeft/Right/Top/Bottom` | `padding`                            |
| `layoutWrap: WRAP`            | `flex-wrap: wrap`                     |
| `strokes` (borders)           | `border`                              |
| `cornerRadius`                | `border-radius`                       |
| `primaryAxisSizingMode: AUTO` | `width: fit-content` / `height: fit-content` |
| `counterAxisSizingMode: AUTO` | `width: fit-content` / `height: fit-content` |
| `primaryAxisSizingMode: FIXED` | `width` / `height` (fixed px)       |
| `counterAxisSizingMode: FILL` | `flex: 1` or `align-self: stretch`    |

**When to use Grid instead of Flexbox:**
- The layout has an explicit 2D structure (rows AND columns defined by designers)
- A Figma Group with nested Auto-Layout frames that form a true grid
- The design uses a repeat-based pattern (e.g., card grids)

### 2.2 Flexbox Code Generator

```typescript
// figmaAutoLayoutToFlexbox.ts
// Parses Figma node JSON and emits CSS flexbox declarations

interface FigmaAutoLayout {
  layoutMode: 'HORIZONTAL' | 'VERTICAL' | 'NONE';
  primaryAxisAlignItems: 'MIN' | 'CENTER' | 'MAX' | 'SPACE_BETWEEN';
  counterAxisAlignItems: 'MIN' | 'CENTER' | 'MAX' | 'BASELINE';
  itemSpacing: number;
  paddingLeft: number;
  paddingRight: number;
  paddingTop: number;
  paddingBottom: number;
  primaryAxisSizingMode: 'FIXED' | 'AUTO';
  counterAxisSizingMode: 'FIXED' | 'AUTO';
  layoutWrap: 'NO_WRAP' | 'WRAP';
  itemReverseZIndex: boolean;
}

const JUSTIFY_MAP: Record<string, string> = {
  MIN: 'flex-start',
  CENTER: 'center',
  MAX: 'flex-end',
  SPACE_BETWEEN: 'space-between',
};

const ALIGN_MAP: Record<string, string> = {
  MIN: 'flex-start',
  CENTER: 'center',
  MAX: 'flex-end',
  BASELINE: 'baseline',
};

function autoLayoutToFlexbox(layout: FigmaAutoLayout): string {
  if (layout.layoutMode === 'NONE') {
    return '/* No auto-layout applied */';
  }

  const direction = layout.layoutMode === 'HORIZONTAL' ? 'row' : 'column';
  const justifyContent = JUSTIFY_MAP[layout.primaryAxisAlignItems] ?? 'flex-start';
  const alignItems = ALIGN_MAP[layout.counterAxisAlignItems] ?? 'stretch';
  const gap = layout.itemSpacing > 0 ? `${layout.itemSpacing}px` : '0';

  const rules: string[] = [
    `display: flex;`,
    `flex-direction: ${direction};`,
    `justify-content: ${justifyContent};`,
    `align-items: ${alignItems};`,
    `gap: var(--spacing-${layout.itemSpacing / 4});`, // Map to spacing token
    `padding: ${layout.paddingTop}px ${layout.paddingRight}px ${layout.paddingBottom}px ${layout.paddingLeft}px;`,
  ];

  if (layout.layoutWrap === 'WRAP') {
    rules.push('flex-wrap: wrap;');
  }

  return rules.join('\n  ');
}

// {
//   layoutMode: 'VERTICAL',
//   primaryAxisAlignItems: 'CENTER',
//   counterAxisAlignItems: 'CENTER',
//   itemSpacing: 16,
//   paddingLeft: 24, paddingRight: 24,
//   paddingTop: 16, paddingBottom: 16,
//   ...
// }
// →
// display: flex;
// flex-direction: column;
// justify-content: center;
```

### 2.3 CSS Grid Code Generator (for 2D layouts)

```typescript
// figmaGroupToGrid.ts
// Detects grid-like patterns in Figma frames and emits CSS Grid

interface GridConfig {
  columns: number;
  rows: number;
  columnGap: number;
  rowGap: number;
  columnWidths: string[];  // '1fr', '300px', etc.
  rowHeights: string[];
}

function detectGridFromNodes(
  nodes: Array<{ x: number; y: number; width: number; height: number; name: string }>
): GridConfig | null {
  if (nodes.length < 2) return null;

  // Group by Y-coordinate to detect rows
  const rowsMap = new Map<number, typeof nodes>();
  const tolerance = 4; // pixel tolerance for alignment

  nodes.forEach((node) => {
    let found = false;
    for (const [y, group] of rowsMap) {
      if (Math.abs(node.y - y) <= tolerance) {
        group.push(node);
        found = true;
        break;
      }
    }
    if (!found) rowsMap.set(node.y, [node]);
  });

  const rows = Array.from(rowsMap.entries()).sort((a, b) => a[0] - b[0]);
  if (rows.length < 2) return null;

  const columns = Math.max(...rows.map(([, group]) => group.length));
  const columnGap = detectGap(rows[0][1].map((n) => n.x + n.width));
  const rowGap = detectGap(rows.map(([y]) => y));

  return {
    columns,
    rows: rows.length,
    columnGap,
    rowGap,
    columnWidths: Array(columns).fill('1fr'),
    rowHeights: Array(rows.length).fill('auto'),
  };
}

function detectGap(positions: number[]): number {
  const sorted = [...positions].sort((a, b) => a - b);
  const gaps = [];
  for (let i = 1; i < sorted.length; i++) {
    gaps.push(sorted[i] - sorted[i - 1]);
  }
  if (gaps.length === 0) return 0;
  // Return the mode (most common) gap
  return Math.round(gaps.reduce((a, b) => a + b, 0) / gaps.length);
}

function gridConfigToCSS(config: GridConfig): string {
  return [
    `display: grid;`,
    `grid-template-columns: repeat(${config.columns}, 1fr);`,
    `grid-template-rows: auto;`,
    `gap: ${config.rowGap}px ${config.columnGap}px;`,
  ].join('\n  ');
}
```

```typescript
// responsiveMapper.ts
// Adds responsive overrides based on Figma variants/prototype constraints

interface ResponsiveOverride {
  breakpoint: string;   // match against --bp-* token
  layoutMode?: 'HORIZONTAL' | 'VERTICAL';
  columns?: number;
  gap?: number;
  padding?: { top: number; right: number; bottom: number; left: number };
}

function generateResponsiveCSS(
  overrides: ResponsiveOverride[],
  baseSelector: string
): string {
  let css = '';

  for (const override of overrides) {
    css += `\n@media (max-width: var(--${override.breakpoint})) {\n`;
    css += `  ${baseSelector} {\n`;

    if (override.columns !== undefined) {
      css += `    grid-template-columns: repeat(${override.columns}, 1fr);\n`;
    }
    if (override.layoutMode) {
      const dir = override.layoutMode === 'HORIZONTAL' ? 'row' : 'column';
      css += `    flex-direction: ${dir};\n`;
    }
    if (override.gap !== undefined) {
      css += `    gap: var(--spacing-${Math.round(override.gap / 4)});\n`;
    }
    if (override.padding) {
      const { top, right, bottom, left } = override.padding;
      css += `    padding: ${top}px ${right}px ${bottom}px ${left}px;\n`;
    }

    css += `  }\n}\n`;
  }

  return css;
}
```

```typescript
// figmaNameToCSS.ts
// Transforms Figma layer names to CSS class names

function figmaNameToCSS(figmaLayerName: string): string {
  // Step 1: Strip category prefix
  const prefixMap: Record<string, string> = {
    'Page/': 'page',
    'Section/': 'section',
    'Hero/': 'hero',
    'Card/': 'card',
    'Form/': 'form',
    'Input/': 'input',
    'Button/': 'btn',
    'Nav/': 'nav',
    'Header/': 'header',
    'Footer/': 'footer',
    'Modal/': 'modal',
    'Table/': 'table',
    'List/': 'list',
    'Icon/': 'icon',
    'Badge/': 'badge',
    'Avatar/': 'avatar',
  };

  let name = figmaLayerName;
  let prefix = '';

  for (const [key, value] of Object.entries(prefixMap)) {
    if (name.startsWith(key)) {
      name = name.slice(key.length);
      prefix = value;
      break;
    }
  }

  // Step 2: PascalCase → kebab-case
  // "PrimaryButton" → "primary-button"
  name = name
    .replace(/([a-z])([A-Z])/g, '$1-$2')
    .replace(/([A-Z])([A-Z][a-z])/g, '$1-$2')
    .toLowerCase();

  // Step 3: Handle nested elements (separated by /)
  // "Section/Header/Title" → "section__header__title"
  const parts = name.split(/\/|\\/);
  name = parts.join('__');

  // Step 4: Handle variants (separated by =)
  // "Button=Primary" → "btn--primary"
  name = name.replace(/=/g, '--');

  // Step 5: Clean up
  name = name
    .replace(/[^a-z0-9\-_]/g, '-')
    .replace(/-{2,}/g, '-')
    .replace(/^-|-$/g, '');

  return prefix ? `${prefix}${name ? '-' + name : ''}` : name;
}

// Examples:
// figmaNameToCSS('Button/Primary')          → 'btn--primary'
// figmaNameToCSS('Card/Header/Title')       → 'card__header__title'
// figmaNameToCSS('Section/HeroBanner')      → 'section__hero-banner'
// figmaNameToCSS('Input/TextField=Error')   → 'input__text-field--error'
// figmaNameToCSS('Modal/Header')            → 'modal__header'
```

```typescript
// exportAssets.ts — Figma plugin to bulk-export assets
// Uses Figma Plugin API exportAsync()

interface ExportResult {
  nodeName: string;
  format: string;
  scale: number;
  filePath: string;
  buffer: Uint8Array;
}

async function exportNode(
  node: SceneNode,
  format: 'SVG' | 'PNG' | 'JPG' | 'WEBP',
  scale: number = 1
): Promise<Uint8Array> {
  const exportSettings: ExportSettings = format === 'SVG'
    ? {
        format: 'SVG',
        svgOutlineText: true,
        svgIdAttribute: false,
      } as ExportSettingsSVG
    : {
        format,
        constraint: { type: 'SCALE', value: scale },
      } as ExportSettingsImage;

  return await (node as ExportMixin).exportAsync(exportSettings);
}

async function exportAllAssets(configs: AssetExportConfig[]): Promise<ExportResult[]> {
  const results: ExportResult[] = [];

  for (const config of configs) {
    // Find all nodes matching the pattern
    const matchingNodes = figma.currentPage.findAll((n) => {
      const pattern = config.nodePattern.replace('*', '.*');
      return new RegExp(`^${pattern}$`).test(n.name) && 'exportAsync' in n;
    });

    for (const node of matchingNodes) {
      for (const scale of config.scales) {
        const buffer = await exportNode(node, config.format, scale);
        const suffix = config.suffixMode === 'scale'
          ? `@${scale}x`
          : `_${scale}x`;
        const ext = config.format.toLowerCase();
        const fileName = config.scales.length > 1
          ? `${slugify(node.name)}${suffix}.${ext}`
          : `${slugify(node.name)}.${ext}`;

        results.push({
          nodeName: node.name,
          format: config.format,
          scale,
          filePath: `${config.outputDir}/${fileName}`,
          buffer,
        });

        console.log(`  ✅ ${node.name} → ${fileName}`);
      }
    }
  }

  return results;
}

function slugify(name: string): string {
  return name
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, '-')
    .replace(/^-|-$/g, '');
}

// Execute
(async () => {
  const results = await exportAllAssets(defaultExportConfig);
  console.log(`\n🎉 Exported ${results.length} assets`);
  // In a real plugin, send results to UI or save to disk
  figma.ui.postMessage({ type: 'export-complete', results });
})();
```

### 4.5 Sprite Sheet Generation (for icons)

```typescript
// generateSprite.ts
// Combines SVG icons into a single sprite sheet

interface SpriteEntry {
  id: string;
  svgContent: string;
}

function generateSpriteSheet(icons: SpriteEntry[]): string {
  const symbols = icons.map((icon) => {
    // Extract inner SVG content (strip <svg> wrapper)
    const inner = icon.svgContent
      .replace(/<svg[^>]*>/, '')
      .replace(/<\/svg>/, '')
      .trim();

    return `  <symbol id="${icon.id}" viewBox="0 0 24 24">\n    ${inner}\n  </symbol>`;
  });

  return `<!-- Auto-generated SVG sprite — DO NOT EDIT -->
<svg xmlns="http://www.w3.org/2000/svg" style="display:none">
${symbols.join('\n\n')}
</svg>`;
}

// Usage in HTML:
// <svg><use href="#icon-check" /></svg>
```

## 5. Design Handoff Checklist

### 5.1 Pre-Handoff (Designer Responsibility)

- [ ] **All frames use Auto-Layout** — No absolute-positioned layers except for decorative elements or explicit positioning requirements.
- [ ] **Design tokens are defined** — All colors, typography, spacing, and effects come from Figma styles or variables, not ad-hoc values.
- [ ] **Naming conventions are followed** — All layers use the `Category/ComponentName` pattern (see Section 3).
- [ ] **Components are properly structured** — Variants are set up for all states (default, hover, focus, active, disabled, loading, error).
- [ ] **Responsive variants exist** — At minimum, mobile (375px), tablet (768px), and desktop (1440px) layouts.
- [ ] **All interactive states are documented** — Hover, focus, active, disabled, loading, empty, error states for every interactive element.
- [ ] **Edge cases are covered** — Empty states, long text truncation, extreme data scenarios.
- [ ] **Assets are marked for export** — All exportable nodes have the correct export settings in Figma.
- [ ] **Design is on a consistent grid** — Layout aligns to the 4px or 8px base grid.
- [ ] **Typography hierarchy is clear** — Heading levels (h1–h6), body text sizes, and caption sizes are consistent.
- [ ] **Color contrast meets WCAG AA** — Text/background contrast ratio ≥ 4.5:1 (normal text) or ≥ 3:1 (large text).
- [ ] **Spacing is consistent** — Padding and margins use values from the spacing scale, not arbitrary numbers.
- [ ] **Prototype flows are linked** — Key user flows are connected in Figma prototype mode for reference.
- [ ] **Annotations are added** — Any non-obvious behavior (animations, conditional visibility, data dependencies) is annotated on the canvas or in a companion document.

### 5.5 Automated Handoff Validation Script

```typescript
// validateHandoff.ts
// Validates that a Figma file meets handoff requirements before the developer picks it up

interface HandoffValidationResult {
  passed: boolean;
  checks: Array<{
    name: string;
    passed: boolean;
    message: string;
  }>;
  summary: {
    total: number;
    passed: number;
    failed: number;
    warnings: number;
  };
}

async function validateHandoff(): Promise<HandoffValidationResult> {
  const checks: HandoffValidationResult['checks'] = [];

  // Check 1: All top-level frames use Auto-Layout
  const frames = figma.currentPage.findAll((n) => n.type === 'FRAME') as FrameNode[];
  const framesWithoutLayout = frames.filter((f) => !f.layoutMode || f.layoutMode === 'NONE');
  checks.push({
    name: 'Auto-Layout Coverage',
    passed: framesWithoutLayout.length === 0,
    message: framesWithoutLayout.length === 0
      ? `All ${frames.length} frames use Auto-Layout ✅`
      : `${framesWithoutLayout.length}/${frames.length} frames missing Auto-Layout ⚠️\n  ${framesWithoutLayout.map((f) => `"${f.name}"`).join(', ')}`,
  });

  // Check 2: No unnamed layers
  const allNodes = figma.currentPage.findAll();
  const unnamedNodes = allNodes.filter((n) => !n.name || n.name.trim() === '' || n.name.startsWith('Frame ') || n.name.startsWith('Rectangle '));
  checks.push({
    name: 'Layer Naming',
    passed: unnamedNodes.length === 0,
    message: unnamedNodes.length === 0
      ? 'All layers are named ✅'
      : `${unnamedNodes.length} layers have default/generic names ⚠️`,
  });

  // Check 3: All colors come from styles or variables
  const textNodes = allNodes.filter((n) => n.type === 'TEXT') as TextNode[];
  const nodesWithAdHocColors = textNodes.filter((n) => {
    const fills = (n as any).fills;
    if (!Array.isArray(fills) || fills.length === 0) return false;
    return fills.some((f: any) => f.type === 'SOLID' && !f.boundVariable);
  });
  checks.push({
    name: 'Color Styles',
    passed: nodesWithAdHocColors.length === 0,
    message: nodesWithAdHocColors.length === 0
      ? 'All text colors use styles/variables ✅'
      : `${nodesWithAdHocColors.length} text nodes use ad-hoc colors ⚠️`,
  });

  // Check 4: Export settings configured
  const exportableNodes = allNodes.filter((n) => 'exportSettings' in n && (n as any).exportSettings?.length > 0);
  const expectedExportNodes = allNodes.filter((n) =>
    n.name.startsWith('Icon/') || n.name.startsWith('Image/') || n.name.startsWith('Logo/')
  );
  const missingExports = expectedExportNodes.filter((n) => !exportableNodes.includes(n));
  checks.push({
    name: 'Export Settings',
    passed: missingExports.length === 0,
    message: missingExports.length === 0
      ? 'All exportable nodes have export settings ✅'
      : `${missingExports.length} nodes missing export settings ⚠️\n  ${missingExports.map((n) => `"${n.name}"`).slice(0, 5).join(', ')}...`,
  });

  // Check 5: Variants for interactive components
  const componentSets = figma.currentPage.findAll((n) => n.type === 'COMPONENT_SET') as ComponentSetNode[];
  const componentsWithoutStates = componentSets.filter((cs) => {
    // Check if variants include common states
    const variantNames = cs.children.map((c) => c.name.toLowerCase());
    const requiredStates = ['default', 'hover', 'focus'];
    return !requiredStates.every((s) => variantNames.some((n) => n.includes(s)));
  });
  checks.push({
    name: 'Interactive States',
    passed: componentsWithoutStates.length === 0,
    message: componentsWithoutStates.length === 0
      ? 'All component sets define interactive states ✅'
      : `${componentsWithoutStates.length} component sets may be missing states ⚠️\n  ${componentsWithoutStates.map((c) => `"${c.name}"`).join(', ')}`,
  });

  // Summarize
  const total = checks.length;
  const passed = checks.filter((c) => c.passed).length;
  const failed = total - passed;

  return {
    passed: failed === 0,
    checks,
    summary: { total, passed, failed, warnings: 0 },
  };
}

// Run validation
(async () => {
  const result = await validateHandoff();
  console.log(`\n📋 Handoff Validation: ${result.passed ? 'PASSED ✅' : 'FAILED ❌'}`);
  console.log(`   ${result.summary.passed}/${result.summary.total} checks passed\n`);
  for (const check of result.checks) {
    const icon = check.passed ? '✅' : '⚠️';
    console.log(`${icon} ${check.name}: ${check.message.split('\n')[0]}`);
  }
})();
```

| Figma Property              | CSS Property              | Notes                                        |
|----------------------------|---------------------------|----------------------------------------------|
| `fills[0].color`           | `background-color`        | Solid fill only                              |
| `strokes[0].color`         | `border-color`            | Border color                                 |
| `strokeWeight`             | `border-width`            | Uniform border width                         |
| `cornerRadius`             | `border-radius`           | Unified radius                               |
| `effects.dropShadow`       | `box-shadow`              | Offset + blur + spread + color               |
| `effects.innerShadow`      | `box-shadow: inset`       | Inner shadow                                 |
| `effects.layerBlur`        | `filter: blur()`          | Layer blur                                   |
| `effects.backgroundBlur`   | `backdrop-filter: blur()` | Background blur                              |
| `opacity`                  | `opacity`                 | Node-level opacity                           |
| `blendMode`                | `mix-blend-mode`          | Layer blending                               |
| `clipsContent`             | `overflow: hidden`        | Clipping behavior                            |
| `visible`                  | `display: none` / `visibility: hidden` | Visibility toggle              |

## Appendix B: Responsive Strategy Decision Tree

```
Is the layout 2D (rows AND columns)?
├── YES → Use CSS Grid
│   └── Override grid-template-columns at breakpoints
└── NO → Use Flexbox
    ├── Does it wrap?
    │   ├── YES → flex-wrap: wrap
    │   └── NO → flex-wrap: nowrap
    └── Should it stack on mobile?
        ├── YES → @media query → flex-direction: column
        └── NO → Keep original direction
```
