## 1. Architecture Overview

The theming system is built on four layers:

```
┌──────────────────────────────────────┐
│         Theme Toggle (UI)            │  ← Web Component
├──────────────────────────────────────┤
│   Theme Manager (JS Controller)      │  ← localStorage + matchMedia
├──────────────────────────────────────┤
│   Color Calculation Engine (JS)      │  ← OKLCH math, contrast checks
├──────────────────────────────────────┤
│   CSS Custom Properties (themes)     │  ← light / dark / high-contrast
└──────────────────────────────────────┘
```

**Design decisions:**

- **OKLCH color space** for perceptual uniformity and accessible palette generation.
- **Three themes:** `light`, `dark`, `high-contrast` (the latter meets WCAG AAA 7:1 minimum).
- **Single-source token definitions** — every semantic token is derived from a small set of primitive hues.
- **Zero-dependency** — vanilla CSS + JS, no framework required.
- **`prefers-reduced-motion`** respected for transitions.

## 3. Color Calculation Engine (JS)

This engine provides programmatic theme color manipulation using the OKLCH color space. It powers automatic dark variant generation, contrast checking, and palette derivation.

```javascript
/* ═══════════════════════════════════════════════════════════
   color-engine.js
   OKLCH-based color calculations for automatic theming.
   Zero dependencies. Works in browser and Node.js.
   ═══════════════════════════════════════════════════════════ */

const ColorEngine = (() => {
  /* ─────────────────────────────────────────────────────────
     OKLCH ↔ sRGB Conversion
     Based on CSS Color Level 4 spec algorithms.
     ───────────────────────────────────────────────────────── */

  /**
   * Parse an oklch() CSS string into {l, c, h} object.
   * Handles: oklch(0.68 0.14 260), oklch(0.68 0.14 260 / 0.5)
   */
  function parseOKLCH(str) {
    const m = str.match(
      /oklch\(\s*([\d.]+)\s+([\d.]+)\s+([\d.]+)(?:\s*\/\s*([\d.]+))?\s*\)/i
    );
    if (!m) throw new Error(`Invalid oklch string: ${str}`);
    return {
      l: parseFloat(m[1]),
      c: parseFloat(m[2]),
      h: parseFloat(m[3]),
      alpha: m[4] !== undefined ? parseFloat(m[4]) : 1,
    };
  }

  /**
   * Convert OKLCH to sRGB hex string.
   * Algorithm: OKLCH → OKLab → linear sRGB → gamma-corrected sRGB → hex.
   */
  function oklchToHex(l, c, h, alpha = 1) {
    // Step 1: OKLCH → OKLab
    const hRad = (h * Math.PI) / 180;
    const a = c * Math.cos(hRad);
    const b = c * Math.sin(hRad);

    // Step 2: OKLab → linear sRGB
    const l_ = l + 0.3963377774 * a + 0.2158037573 * b;
    const m_ = l - 0.1055613458 * a - 0.0638541728 * b;
    const s_ = l - 0.0894841775 * a - 1.2914855480 * b;

    const l3 = l_ * l_ * l_;
    const m3 = m_ * m_ * m_;
    const s3 = s_ * s_ * s_;

    const linR = +4.0767416621 * l3 - 3.3077115913 * m3 + 0.2309699292 * s3;
    const linG = -1.2684380046 * l3 + 2.6097574011 * m3 - 0.3413193965 * s3;
    const linB = -0.0041960863 * l3 - 0.7034186147 * m3 + 1.7076147010 * s3;

    // Step 3: Linear → gamma-corrected sRGB
    function linearToGamma(c) {
      const abs = Math.abs(c);
      if (abs > 0.0031308) {
        return (c > 0 ? 1 : -1) * (1.055 * Math.pow(abs, 1 / 2.4) - 0.055);
      }
      return 12.92 * c;
    }

    let r = linearToGamma(linR);
    let g = linearToGamma(linG);
    let bl = linearToGamma(linB);

    // Clamp
    r = Math.max(0, Math.min(1, r));
    g = Math.max(0, Math.min(1, g));
    bl = Math.max(0, Math.min(1, bl));

    // Step 4: → hex
    const toHex = (v) =>
      Math.round(v * 255)
        .toString(16)
        .padStart(2, "0");

    if (alpha < 1) {
      const aHex = Math.round(alpha * 255)
        .toString(16)
        .padStart(2, "0");
      return `#${toHex(r)}${toHex(g)}${toHex(bl)}${aHex}`;
    }
    return `#${toHex(r)}${toHex(g)}${toHex(bl)}`;
  }

  /**
   * Convert sRGB hex to OKLCH {l, c, h} object.
   */
  function hexToOKLCH(hex) {
    // Strip #
    hex = hex.replace("#", "");

    // Parse
    const r = parseInt(hex.substring(0, 2), 16) / 255;
    const g = parseInt(hex.substring(2, 4), 16) / 255;
    const b = parseInt(hex.substring(4, 6), 16) / 255;

    // Gamma → linear
    function gammaToLinear(c) {
      const abs = Math.abs(c);
      if (abs <= 0.04045) return c / 12.92;
      return (c > 0 ? 1 : -1) * Math.pow((abs + 0.055) / 1.055, 2.4);
    }

    const linR = gammaToLinear(r);
    const linG = gammaToLinear(g);
    const linB = gammaToLinear(b);

    // Linear sRGB → OKLab
    const l_ = 0.4122214708 * linR + 0.5363325363 * linG + 0.0514459929 * linB;
    const m_ = 0.2119034982 * linR + 0.6806995451 * linG + 0.1073969566 * linB;
    const s_ = 0.0883024619 * linR + 0.2817188376 * linG + 0.6299787005 * linB;

    const l_rt = Math.cbrt(l_);
    const m_rt = Math.cbrt(m_);
    const s_rt = Math.cbrt(s_);

    const L = 0.2104542553 * l_rt + 0.7936177850 * m_rt - 0.0040720468 * s_rt;
    const a = 1.9779984951 * l_rt - 2.4285922050 * m_rt + 0.4505937099 * s_rt;
    const b_ = 0.0259040371 * l_rt + 0.7827717662 * m_rt - 0.8086757660 * s_rt;

    const C = Math.sqrt(a * a + b_ * b_);
    let H = (Math.atan2(b_, a) * 180) / Math.PI;
    if (H < 0) H += 360;

    return { l: L, c: C, h: H };
  }

  /* ─────────────────────────────────────────────────────────
     Contrast Ratio (WCAG 2.1)
     ───────────────────────────────────────────────────────── */

  /**
   * Calculate relative luminance from sRGB values (0–1).
   * WCAG 2.1 definition.
   */
  function relativeLuminance(r, g, b) {
    const toLinear = (c) => {
      if (c <= 0.03928) return c / 12.92;
      return Math.pow((c + 0.055) / 1.055, 2.4);
    };
    return 0.2126 * toLinear(r) + 0.7152 * toLinear(g) + 0.0722 * toLinear(b);
  }

  /**
   * Compute WCAG contrast ratio between two hex colors.
   * Ratio ≥ 4.5 → AA for normal text.
   * Ratio ≥ 7.0 → AAA for normal text.
   * Ratio ≥ 3.0 → AA for large text (18px+ bold or 24px+).
   */
  function contrastRatio(hex1, hex2) {
    const parse = (hex) => {
      hex = hex.replace("#", "");
      return {
        r: parseInt(hex.substring(0, 2), 16) / 255,
        g: parseInt(hex.substring(2, 4), 16) / 255,
        b: parseInt(hex.substring(4, 6), 16) / 255,
      };
    };

    const c1 = parse(hex1);
    const c2 = parse(hex2);

    const l1 = relativeLuminance(c1.r, c1.g, c1.b);
    const l2 = relativeLuminance(c2.r, c2.g, c2.b);

    const lighter = Math.max(l1, l2);
    const darker = Math.min(l1, l2);

    return (lighter + 0.05) / (darker + 0.05);
  }

  /**
   * Check if contrast meets WCAG level.
   * @param {number} ratio - Computed contrast ratio.
   * @param {'AA'|'AAA'} level - WCAG conformance level.
   * @param {boolean} isLargeText - true for large text (lower threshold).
   */
  function meetsWCAG(ratio, level = "AA", isLargeText = false) {
    if (level === "AAA") {
      return isLargeText ? ratio >= 4.5 : ratio >= 7.0;
    }
    return isLargeText ? ratio >= 3.0 : ratio >= 4.5;
  }

  /* ─────────────────────────────────────────────────────────
     Automatic Dark Variants
     ───────────────────────────────────────────────────────── */

  /**
   * Generate a dark-theme variant of a color by adjusting
   * OKLCH lightness and chroma.
   *
   * Strategy:
   *   - Flip lightness: newL = 1.0 - originalL (perceptual flip)
   *   - Slightly boost chroma for readability on dark
   *   - Preserve hue
   *
   * @param {string} lightColor - oklch() or hex string.
   * @param {Object} options
   * @param {number} [options.chromaBoost=1.3] - Multiplier for chroma.
   * @param {number} [options.maxChroma=0.37] - Cap chroma.
   * @returns {string} oklch() string for dark variant.
   */
  function autoDarkVariant(lightColor, options = {}) {
    const { chromaBoost = 1.3, maxChroma = 0.37 } = options;
    let l, c, h;

    if (lightColor.startsWith("oklch(")) {
      const parsed = parseOKLCH(lightColor);
      l = parsed.l;
      c = parsed.c;
      h = parsed.h;
    } else {
      const oklch = hexToOKLCH(lightColor);
      l = oklch.l;
      c = oklch.c;
      h = oklch.h;
    }

    // Flip lightness perceptually
    // If original is dark (e.g., text), make it light
    // If original is light (e.g., bg), make it dark
    const newL = clamp(1.0 - l, 0.05, 0.95);

    // Boost chroma for readability on dark backgrounds
    const newC = clamp(c * chromaBoost, 0.005, maxChroma);

    return `oklch(${newL.toFixed(3)} ${newC.toFixed(3)} ${h.toFixed(1)})`;
  }

  /**
   * Derive a full dark palette from light theme tokens.
   * @param {Object} lightTokens - { name: oklchString }
   * @returns {Object} dark tokens with same keys.
   */
  function deriveDarkPalette(lightTokens) {
    const dark = {};
    for (const [key, value] of Object.entries(lightTokens)) {
      dark[key] = autoDarkVariant(value);
    }
    return dark;
  }

  /* ─────────────────────────────────────────────────────────
     Palette Generators
     ───────────────────────────────────────────────────────── */

  /**
   * Generate a tonal palette from a single hue.
   * Produces 10 evenly-spaced lightness stops.
   *
   * @param {number} hue - OKLCH hue (0–360).
   * @param {number} chroma - Chroma value.
   * @param {number} steps - Number of steps (default 10).
   * @returns {Object} { 50: oklchStr, 100: oklchStr, ..., 950: oklchStr }
   */
  function tonalPalette(hue, chroma = 0.12, steps = 10) {
    const palette = {};
    const labels = [50, 100, 200, 300, 400, 500, 600, 700, 800, 900, 950];

    for (let i = 0; i <= steps; i++) {
      // Map step to lightness: step 0 → L=0.98 (light), step 10 → L=0.12 (dark)
      const l = 0.98 - i * ((0.98 - 0.12) / steps);
      const cAdjusted = chroma * (1 - Math.abs(i - steps / 2) / (steps / 2) * 0.3);
      palette[labels[i]] = `oklch(${l.toFixed(3)} ${cAdjusted.toFixed(3)} ${hue})`;
    }

    return palette;
  }

  /**
   * Generate an accessible text color (black or white) for a given background.
   * Uses a perceptually weighted threshold.
   *
   * @param {string} bgColor - Background color (hex or oklch).
   * @returns {string} '#000000' or '#ffffff'.
   */
  function accessibleTextColor(bgColor) {
    let hex = bgColor;
    if (bgColor.startsWith("oklch(")) {
      const { l, c, h } = parseOKLCH(bgColor);
      hex = oklchToHex(l, c, h);
    }

    const { r, g, b } = (() => {
      const h = hex.replace("#", "");
      return {
        r: parseInt(h.substring(0, 2), 16) / 255,
        g: parseInt(h.substring(2, 4), 16) / 255,
        b: parseInt(h.substring(4, 6), 16) / 255,
      };
    })();

    const lum = relativeLuminance(r, g, b);
    // Threshold ~0.18 — empirically good for WCAG AA
    return lum > 0.18 ? "#000000" : "#ffffff";
  }

  /* ─────────────────────────────────────────────────────────
     Utilities
     ───────────────────────────────────────────────────────── */

  function clamp(value, min, max) {
    return Math.max(min, Math.min(max, value));
  }

  /**
   * Linear interpolation between two OKLCH colors.
   * @param {Object} color1 - {l, c, h}
   * @param {Object} color2 - {l, c, h}
   * @param {number} t - 0.0 to 1.0
   * @returns {string} interpolated oklch() string.
   */
  function interpolateOKLCH(color1, color2, t) {
    // Handle hue interpolation (shortest path around circle)
    let dh = color2.h - color1.h;
    if (Math.abs(dh) > 180) {
      dh = dh > 0 ? dh - 360 : dh + 360;
    }
    const h = ((color1.h + dh * t) % 360 + 360) % 360;

    const l = color1.l + (color2.l - color1.l) * t;
    const c = color1.c + (color2.c - color1.c) * t;

    return `oklch(${l.toFixed(3)} ${c.toFixed(3)} ${h.toFixed(1)})`;
  }

  /* ─────────────────────────────────────────────────────────
     Public API
     ───────────────────────────────────────────────────────── */
  return {
    parseOKLCH,
    oklchToHex,
    hexToOKLCH,
    contrastRatio,
    meetsWCAG,
    autoDarkVariant,
    deriveDarkPalette,
    tonalPalette,
    accessibleTextColor,
    interpolateOKLCH,
    clamp,
  };
})();

/* ── Export for Node.js / bundlers ───────────────────────── */
if (typeof module !== "undefined" && module.exports) {
  module.exports = ColorEngine;
}
```

## 4. Theme Persistence & System Detection

### 4.1 Theme Manager

```javascript
/* ═══════════════════════════════════════════════════════════
   theme-manager.js
   Handles persistence, system preference detection,
   and theme application. Zero dependencies.
   ═══════════════════════════════════════════════════════════ */

const ThemeManager = (() => {
  const STORAGE_KEY = "styde-forge:theme";
  const VALID_THEMES = ["light", "dark", "high-contrast", "auto"];

  /* ── State ──────────────────────────────────────────────── */
  let _currentTheme = "auto";
  let _systemDark = false;
  let _listeners = new Set();

  /* ── System Preference Detection ────────────────────────── */
  const darkModeQuery = window.matchMedia("(prefers-color-scheme: dark)");
  const contrastQuery = window.matchMedia("(prefers-contrast: more)");

  function _handleSystemChange(e) {
    _systemDark = darkModeQuery.matches;
    // Re-apply if in auto mode
    if (_currentTheme === "auto") {
      _applyEffectiveTheme();
    }
  }

  /* ── Persistence ────────────────────────────────────────── */

  /**
   * Read stored preference. Returns null if none.
   */
  function getStoredTheme() {
    try {
      const stored = localStorage.getItem(STORAGE_KEY);
      if (stored && VALID_THEMES.includes(stored)) {
        return stored;
      }
    } catch {
      // localStorage unavailable (private browsing, etc.)
    }
    return null;
  }

  /**
   * Write preference to localStorage.
   */
  function setStoredTheme(theme) {
    try {
      localStorage.setItem(STORAGE_KEY, theme);
    } catch {
      // Silently fail — still applies in-memory
    }
  }

  /* ── Theme Resolution ───────────────────────────────────── */

  /**
   * Resolve "auto" to an actual theme.
   * Priority: system preference → stored preference → "light"
   * If system says high-contrast, use that.
   */
  function _resolveAuto() {
    // Check system high-contrast preference first
    if (contrastQuery.matches) {
      return "high-contrast";
    }
    // Check system dark/light preference
    if (darkModeQuery.matches) {
      return "dark";
    }
    // Default to light
    return "light";
  }

  /**
   * Get the effective theme (resolves "auto").
   */
  function getEffectiveTheme() {
    if (_currentTheme === "auto") {
      return _resolveAuto();
    }
    return _currentTheme;
  }

  /* ── Theme Application ──────────────────────────────────── */

  /**
   * Apply a theme to the document.
   * Sets data-theme attribute on <html>.
   * Also sets a color-scheme meta for native browser elements.
   */
  function _applyEffectiveTheme() {
    const effective = getEffectiveTheme();
    const root = document.documentElement;

    // Set data attribute for CSS
    root.setAttribute("data-theme", effective);

    // Set color-scheme for browser chrome (scrollbars, form controls)
    if (effective === "high-contrast") {
      root.style.colorScheme = "dark";
    } else {
      root.style.colorScheme = effective;
    }

    // Update meta theme-color for mobile browser chrome
    const metaThemeColor = document.querySelector('meta[name="theme-color"]');
    if (metaThemeColor) {
      const bgColor = getComputedStyle(root)
        .getPropertyValue("--color-surface-0")
        .trim();
      metaThemeColor.setAttribute("content", bgColor || (effective === "dark" ? "#1a1a2e" : "#fafafc"));
    }

    // Notify listeners
    _notifyListeners(effective);
  }

  /* ── Public API ─────────────────────────────────────────── */

  /**
   * Initialize the theme system.
   * Call once on page load, before any rendering.
   *
   * @param {Object} options
   * @param {string} [options.defaultTheme='auto'] - Fallback theme.
   * @param {boolean} [options.persist=true] - Use localStorage.
   */
  function init(options = {}) {
    const { defaultTheme = "auto", persist = true } = options;

    _systemDark = darkModeQuery.matches;

    // Determine initial theme
    let theme = defaultTheme;
    if (persist) {
      const stored = getStoredTheme();
      if (stored) theme = stored;
    }

    // Validate
    if (!VALID_THEMES.includes(theme)) theme = "auto";

    _currentTheme = theme;

    // Apply immediately (before first paint if possible)
    _applyEffectiveTheme();

    // Listen for system changes
    darkModeQuery.addEventListener("change", _handleSystemChange);
    contrastQuery.addEventListener("change", _handleSystemChange);

    // Apply flash-prevention class
    // Remove after first paint to allow transitions
    document.documentElement.classList.add("theme-ready");
    requestAnimationFrame(() => {
      requestAnimationFrame(() => {
        document.documentElement.classList.remove("theme-ready");
      });
    });
  }

  /**
   * Set the theme explicitly.
   * @param {'light'|'dark'|'high-contrast'|'auto'} theme
   */
  function setTheme(theme) {
    if (!VALID_THEMES.includes(theme)) {
      console.warn(`ThemeManager: invalid theme "${theme}". Must be one of: ${VALID_THEMES.join(", ")}`);
      return;
    }

    _currentTheme = theme;
    setStoredTheme(theme);
    _applyEffectiveTheme();
  }

  /**
   * Toggle between light and dark.
   * If currently "auto", resolves effective theme first,
   * then toggles to the opposite.
   */
  function toggleTheme() {
    const effective = getEffectiveTheme();

    if (effective === "light") {
      setTheme("dark");
    } else if (effective === "dark") {
      setTheme("light");
    } else if (effective === "high-contrast") {
      // Toggle high-contrast → light
      setTheme("light");
    } else {
      setTheme("dark");
    }
  }

  /**
   * Cycle through all themes: light → dark → high-contrast → auto
   */
  function cycleTheme() {
    const idx = VALID_THEMES.indexOf(_currentTheme);
    const next = VALID_THEMES[(idx + 1) % VALID_THEMES.length];
    setTheme(next);
  }

  /**
   * Get the currently set theme (may be "auto").
   */
  function getTheme() {
    return _currentTheme;
  }

  /**
   * Subscribe to theme changes.
   * @param {Function} callback - Receives (effectiveTheme, explicitTheme).
   */
  function onChange(callback) {
    _listeners.add(callback);
    return () => _listeners.delete(callback);
  }

  function _notifyListeners(effective) {
    _listeners.forEach((fn) => fn(effective, _currentTheme));
  }

  /**
   * Tear down listeners (for SPA unmount).
   */
  function destroy() {
    darkModeQuery.removeEventListener("change", _handleSystemChange);
    contrastQuery.removeEventListener("change", _handleSystemChange);
    _listeners.clear();
  }

  return {
    init,
    setTheme,
    toggleTheme,
    cycleTheme,
    getTheme,
    getEffectiveTheme,
    onChange,
    destroy,
    VALID_THEMES,
  };
})();

/* ── Export ───────────────────────────────────────────────── */
if (typeof module !== "undefined" && module.exports) {
  module.exports = ThemeManager;
}
```

### 4.2 Anti-Flash Initialization

Insert this inline `<script>` **in the `<head>` before any CSS** to prevent a flash of wrong theme:

```html
<!-- Anti-flash-of-wrong-theme script — MUST be in <head>, before CSS -->
<script>
  (function() {
    try {
      var theme = localStorage.getItem('styde-forge:theme');
      if (!theme) theme = 'auto';
      if (theme === 'auto') {
        var isDark = window.matchMedia('(prefers-color-scheme: dark)').matches;
        var isHighContrast = window.matchMedia('(prefers-contrast: more)').matches;
        theme = isHighContrast ? 'high-contrast' : (isDark ? 'dark' : 'light');
      }
      document.documentElement.setAttribute('data-theme', theme);
      // Prevent transitions on first paint
      document.documentElement.classList.add('theme-initializing');
    } catch(e) {}
  })();
</script>
```

This script runs **synchronously**, before any paint, so the correct theme is applied from the very first frame.

### 5.2 JavaScript Cross-Fade Enhancer

For elements that benefit from a cross-fade (large background areas), use this optional enhancer:

```javascript
/* ═══════════════════════════════════════════════════════════
   theme-transition-enhancer.js
   Optional: Adds a cross-fade overlay for large background
   transitions. Attach to specific elements.
   ═══════════════════════════════════════════════════════════ */

const ThemeTransitionEnhancer = (() => {
  /**
   * Wrap an element with a cross-fade overlay.
   * During theme transitions, clones the old background,
   * cross-fades to the new one.
   *
   * @param {HTMLElement} element
   * @param {number} [duration=200] - ms
   */
  function enhanceElement(element, duration = 200) {
    let previousBg = null;

    ThemeManager.onChange(() => {
      const newBg = getComputedStyle(element).backgroundColor;

      if (previousBg && previousBg !== newBg) {
        // Create cross-fade overlay
        const overlay = document.createElement("div");
        overlay.style.cssText = `
          position: absolute;
          inset: 0;
          background-color: ${previousBg};
          pointer-events: none;
          z-index: 100;
          border-radius: inherit;
          opacity: 1;
          transition: opacity ${duration}ms ease-out;
        `;

        // Ensure parent is positioned
        const pos = getComputedStyle(element).position;
        if (pos === "static") element.style.position = "relative";

        element.appendChild(overlay);

        // Trigger fade-out
        requestAnimationFrame(() => {
          overlay.style.opacity = "0";
        });

        // Clean up
        overlay.addEventListener("transitionend", () => overlay.remove(), { once: true });
        setTimeout(() => overlay.remove(), duration + 100);
      }

      previousBg = newBg;
    });
  }

  return { enhanceElement };
})();
```

## 6. Theme Toggle Component

### 6.1 Web Component (`<theme-toggle>`)

```javascript
/* ═══════════════════════════════════════════════════════════
   theme-toggle.js
   Custom element: <theme-toggle>
   A fully accessible, animated theme toggle button.
   ═══════════════════════════════════════════════════════════ */

class ThemeToggle extends HTMLElement {
  constructor() {
    super();
    this.attachShadow({ mode: "open" });
    this._unsubscribe = null;
  }

  /* ── Observed attributes ────────────────────────────────── */
  static get observedAttributes() {
    return ["label-light", "label-dark", "label-high-contrast", "label-auto", "size"];
  }

  /* ── Lifecycle ──────────────────────────────────────────── */
  connectedCallback() {
    this._render();
    this._attachEvents();
    this._syncState();

    // Subscribe to external theme changes
    if (typeof ThemeManager !== "undefined") {
      this._unsubscribe = ThemeManager.onChange((effective) => {
        this._syncState(effective);
      });
    }
  }

  disconnectedCallback() {
    if (this._unsubscribe) {
      this._unsubscribe();
      this._unsubscribe = null;
    }
  }

  attributeChangedCallback() {
    if (this.shadowRoot.innerHTML) {
      this._render();
      this._attachEvents();
    }
  }

  /* ── Labels ─────────────────────────────────────────────── */
  get _labelLight() {
    return this.getAttribute("label-light") || "Switch to light theme";
  }
  get _labelDark() {
    return this.getAttribute("label-dark") || "Switch to dark theme";
  }
  get _labelHighContrast() {
    return this.getAttribute("label-high-contrast") || "Switch to high contrast theme";
  }
  get _labelAuto() {
    return this.getAttribute("label-auto") || "Use system theme preference";
  }
  get _size() {
    return this.getAttribute("size") || "40";
  }

  /* ── Render ─────────────────────────────────────────────── */
  _render() {
    const size = this._size;
    const half = parseInt(size) / 2;

    this.shadowRoot.innerHTML = `
      <style>
        :host {
          display: inline-block;
          --tt-size: ${size}px;
          --tt-icon-size: ${parseInt(size) * 0.5}px;
        }

  _attachEvents() {
    const button = this.shadowRoot.querySelector(".toggle");
    if (!button) return;

    // Remove old listener by cloning
    const newButton = button.cloneNode(true);
    button.parentNode.replaceChild(newButton, button);

    newButton.addEventListener("click", (e) => {
      e.preventDefault();

      // Cycle through themes
      if (typeof ThemeManager !== "undefined") {
        ThemeManager.cycleTheme();
      }

      // Dispatch custom event
      this.dispatchEvent(
        new CustomEvent("theme-toggle", {
          bubbles: true,
          composed: true,
          detail: {
            theme: typeof ThemeManager !== "undefined"
              ? ThemeManager.getTheme()
              : null,
          },
        })
      );
    });
  }

  /* ── State sync ─────────────────────────────────────────── */
  _syncState(effective) {
    if (!effective && typeof ThemeManager !== "undefined") {
      effective = ThemeManager.getEffectiveTheme();
    }

    const button = this.shadowRoot.querySelector(".toggle");
    if (button) {
      button.setAttribute("data-effective", effective || "light");

      // Update aria-label
      const labels = {
        light: this._labelDark,
        dark: this._labelHighContrast,
        "high-contrast": this._labelAuto,
      };
      const label = labels[effective] || this._labelDark;
      button.setAttribute("aria-label", label);
      button.setAttribute("title", label);
    }
  }

  _getAriaLabel() {
    const effective =
      typeof ThemeManager !== "undefined"
        ? ThemeManager.getEffectiveTheme()
        : "light";

    const labels = {
      light: this._labelDark,
      dark: this._labelHighContrast,
      "high-contrast": this._labelAuto,
    };
    return labels[effective] || this._labelDark;
  }
}

  customElements.define("theme-toggle", ThemeToggle);
```

```javascript
// Show current theme state
window.__themeDebug = () => ({
  explicit: ThemeManager.getTheme(),

  systemDark: window.matchMedia('(prefers-color-scheme: dark)').matches,

    primary: getComputedStyle(document.documentElement).getPropertyValue('--color-primary').trim(),
  }
});

// Force a theme
window.__setTheme = (t) => ThemeManager.setTheme(t);

// List all theme-related CSS custom properties in use
window.__listThemeTokens = () => {
  const styles = getComputedStyle(document.documentElement);
  const tokens = [];
  for (let i = 0; i < styles.length; i++) {
    const prop = styles[i];
    if (prop.startsWith('--color-') || prop.startsWith('--shadow-') || prop.startsWith('--radius-') || prop.startsWith('--border-')) {
      tokens.push(`${prop}: ${styles.getPropertyValue(prop).trim()}`);
    }
  }
  console.table(tokens.map(t => {
    const [key, val] = t.split(': ');
    return { Token: key, Value: val };
  }));
};
```
