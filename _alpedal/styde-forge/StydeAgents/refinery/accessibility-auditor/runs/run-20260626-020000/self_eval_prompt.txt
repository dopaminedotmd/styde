## 1. axe-core Integration Guide

### 1.1 Overview

[axe-core](https://github.com/dequelabs/axe-core) is Deque Labs' open-source accessibility testing engine. It runs in the browser (or via Node.js/JSDOM), analyzes the rendered DOM against WCAG 2.1/2.2 rules, and returns a list of violations, passes, and incomplete results. It is the engine behind the Lighthouse a11y audit, the `@axe-core/playwright` integration, and most modern CI a11y tooling.

### 1.2 Installation

```bash
# Core library (for JSDOM / Node environments)
npm install --save-dev axe-core

# Playwright integration (recommended for E2E)
npm install --save-dev @axe-core/playwright

# React-specific utilities (optional)
npm install --save-dev @axe-core/react

# Puppeteer integration (alternative)
npm install --save-dev @axe-core/puppeteer
```

### 1.3 Basic Usage: JSDOM / Node

```typescript
// axe-core-node.test.ts
import * as axe from 'axe-core';
import { JSDOM } from 'jsdom';

interface AxeViolation {
  id: string;
  impact: 'critical' | 'serious' | 'moderate' | 'minor';
  description: string;
  help: string;
  helpUrl: string;
  nodes: AxeNodeResult[];
}

interface AxeNodeResult {
  html: string;
  target: string[];
  failureSummary: string;
}

async function runAxeOnHtml(html: string): Promise<AxeViolation[]> {
  const dom = new JSDOM(html);
  const { window } = dom;

  // axe-core needs a global document and window
  (globalThis as any).document = window.document;
  (globalThis as any).window = window;

  const results = await axe.run(window.document);

  if (results.violations.length > 0) {
    console.error(`Found ${results.violations.length} accessibility violations:`);
    results.violations.forEach((violation: AxeViolation) => {
      console.error(`\n[${violation.impact.toUpperCase()}] ${violation.id}: ${violation.description}`);
      console.error(`  Help: ${violation.helpUrl}`);
      violation.nodes.forEach((node) => {
        console.error(`  Node: ${node.target.join(' > ')}`);
        console.error(`  Summary: ${node.failureSummary}`);
      });
    });
  }

  // Clean up globals
  delete (globalThis as any).document;
  delete (globalThis as any).window;

  return results.violations;
}

// Usage in test
describe('Accessibility Audit', () => {
  it('homepage should have no critical or serious a11y violations', async () => {
    const html = '<main><h1>Welcome</h1><button>Click me</button></main>';
    const violations = await runAxeOnHtml(html);
    const criticalOrSerious = violations.filter(
      (v) => v.impact === 'critical' || v.impact === 'serious'
    );
    expect(criticalOrSerious).toHaveLength(0);
  });
});
```

### 1.7 Result Reporting & Baselines

```typescript
// axe-reporter.ts
import { AxeResults } from 'axe-core';
import * as fs from 'fs';
import * as path from 'path';

interface ViolationBaseline {
  id: string;
  target: string[];
  reason: string;
  expires: string; // ISO date after which this baseline must be re-reviewed
}

export function loadBaselines(baselinePath: string): ViolationBaseline[] {
  if (!fs.existsSync(baselinePath)) return [];
  return JSON.parse(fs.readFileSync(baselinePath, 'utf-8'));
}

export function filterBaselineViolations(
  violations: AxeResults['violations'],
  baselines: ViolationBaseline[],
  today: Date = new Date()
): { active: AxeResults['violations']; waived: AxeResults['violations'] } {
  const waived: AxeResults['violations'] = [];
  const active: AxeResults['violations'] = [];

  for (const violation of violations) {
    const matched = baselines.find(
      (b) =>
        b.id === violation.id &&
        violation.nodes.some((n) =>
          n.target.some((t) => b.target.includes(t))
        ) &&
        new Date(b.expires) > today
    );

    if (matched) {
      waived.push(violation);
    } else {
      active.push(violation);
    }
  }

  return { active, waived };
}

export function writeJUnitReport(
  violations: AxeResults['violations'],
  outputPath: string,
  suiteName: string
): void {
  const testCases = violations.map((v) => ({
    name: `${v.id}: ${v.help}`,
    failure: v.nodes
      .map(
        (n) =>
          `Target: ${n.target.join(' > ')}\nSummary: ${n.failureSummary}`
      )
      .join('\n\n'),
  }));

  const xml = `<?xml version="1.0" encoding="UTF-8"?>
<testsuite name="${suiteName}" tests="${violations.length}" failures="${violations.length}">
  ${testCases
    .map(
      (tc) =>
        `  <testcase name="${tc.name}">
    <failure message="${tc.name}">${tc.failure.replace(/&/g, '&amp;').replace(/</g, '&lt;')}</failure>
  </testcase>`
    )
    .join('\n')}
</testsuite>`;

  fs.mkdirSync(path.dirname(outputPath), { recursive: true });
  fs.writeFileSync(outputPath, xml);
}
```

```typescript
    'contentinfo',
  ]);

function findNodesByRole(
  node: any,
  roles: string[],
  results: any[] = []
): any[] {
  if (roles.includes(node.role)) results.push(node);
  if (node.children) {
    for (const child of node.children) {
      findNodesByRole(child, roles, results);
    }
  }
  return results;
}
```

### 2.4 Color Contrast Testing Programmatically

```typescript
// color-contrast.test.ts
import { test, expect } from '@playwright/test';

interface ContrastResult {
  ratio: number;
  passes: boolean;
  foreground: string;
  background: string;
  element: string;
}

test.describe('Color Contrast', () => {
  test('text elements meet minimum contrast ratio', async ({ page }) => {
    await page.goto('/');

    const contrastResults: ContrastResult[] = await page.evaluate(() => {
      // Requires axe-core injected into page context
      // This is handled automatically by @axe-core/playwright

      function getLuminance(r: number, g: number, b: number): number {
        const [rs, gs, bs] = [r, g, b].map((c) => {
          c = c / 255;
          return c <= 0.03928 ? c / 12.92 : Math.pow((c + 0.055) / 1.055, 2.4);
        });
        return 0.2126 * rs + 0.7152 * gs + 0.0722 * bs;
      }

      function getContrastRatio(fg: string, bg: string): number {
        const parseRgb = (color: string) => {
          const match = color.match(/rgb\((\d+),\s*(\d+),\s*(\d+)\)/);
          if (!match) return [0, 0, 0];
          return [parseInt(match[1]), parseInt(match[2]), parseInt(match[3])];
        };

        const [r1, g1, b1] = parseRgb(fg);
        const [r2, g2, b2] = parseRgb(bg);

        const l1 = getLuminance(r1, g1, b1);
        const l2 = getLuminance(r2, g2, b2);

        const lighter = Math.max(l1, l2);
        const darker = Math.min(l1, l2);
        return (lighter + 0.05) / (darker + 0.05);
      }

      const results: ContrastResult[] = [];
      const textElements = document.querySelectorAll(
        'p, h1, h2, h3, h4, h5, h6, span, a, button, label, li, td, th'
      );

      textElements.forEach((el) => {
        const style = window.getComputedStyle(el);
        const fg = style.color;
        const bg = style.backgroundColor;
        const fontSize = parseFloat(style.fontSize);
        const isBold = parseInt(style.fontWeight) >= 700;

        if (bg === 'rgba(0, 0, 0, 0)' || bg === 'transparent') return; // Skip transparent backgrounds

        const ratio = getContrastRatio(fg, bg);

        // WCAG AA: 4.5:1 for normal text, 3:1 for large text (18pt+ or 14pt bold)
        const threshold = fontSize >= 18 || (fontSize >= 14 && isBold) ? 3 : 4.5;

        results.push({
          ratio: Math.round(ratio * 100) / 100,
          passes: ratio >= threshold,
          foreground: fg,
          background: bg,
          element: el.tagName + (el.className ? '.' + el.className.split(' ')[0] : ''),
        });
      });

      return results;
    });

    const failures = contrastResults.filter((r) => !r.passes);
    if (failures.length > 0) {
      console.error('Contrast failures:', JSON.stringify(failures, null, 2));
    }
    expect(failures).toHaveLength(0);
  });
});
```

### 2.5 Integrating axe-core + Playwright in a Reusable Helper

```typescript
// a11y-helpers.ts
import { Page, expect } from '@playwright/test';
import AxeBuilder from '@axe-core/playwright';
import { AxeResults } from 'axe-core';

export interface A11yCheckOptions {
  selector?: string;
  includedImpacts?: ('critical' | 'serious' | 'moderate' | 'minor')[];
  excludedRules?: string[];
  tags?: string[];
}

const DEFAULT_TAGS = ['wcag2a', 'wcag2aa', 'wcag21a', 'wcag21aa', 'wcag22aa'];
const DEFAULT_IMPACTS: A11yCheckOptions['includedImpacts'] = ['critical', 'serious'];

export async function checkAccessibility(
  page: Page,
  options: A11yCheckOptions = {}
): Promise<AxeResults> {
  const {
    selector,
    includedImpacts = DEFAULT_IMPACTS,
    excludedRules = [],
    tags = DEFAULT_TAGS,
  } = options;

  let builder = new AxeBuilder({ page }).withTags(tags);

  if (selector) {
    builder = builder.include(selector);
  }

  if (excludedRules.length > 0) {
    builder = builder.disableRules(excludedRules);
  }

  const results = await builder.analyze();

  // Filter to only the impact levels we care about
  const filteredViolations = results.violations.filter((v) =>
    includedImpacts.includes(v.impact as any)
  );

  // Log violations in a readable format
  if (filteredViolations.length > 0) {
    console.error(
      `\n🚨 ${filteredViolations.length} a11y violation(s) found:`
    );
    for (const v of filteredViolations) {
      console.error(`\n  [${v.impact.toUpperCase()}] ${v.id}: ${v.help}`);
      console.error(`    URL: ${v.helpUrl}`);
      for (const node of v.nodes) {
        console.error(`    Element: ${node.target.join(' > ')}`);
        console.error(`    HTML: ${node.html}`);
        console.error(`    Fix: ${node.failureSummary}`);
      }
    }
  }

  // Assert no violations at requested impact levels
  expect(
    filteredViolations,
    `Expected 0 accessibility violations at impact levels: ${includedImpacts.join(', ')}`
  ).toHaveLength(0);

  return results;
}

// Snapshot-based regression: captures the full axe results and compares
// against a stored baseline to catch regressions
export async function checkAccessibilitySnapshot(
  page: Page,
  snapshotName: string,
  options: A11yCheckOptions = {}
): Promise<void> {
  const results = await checkAccessibility(page, options);

  // Strip dynamic data for stable snapshots
  const stableResults = {
    violations: results.violations.map((v) => ({
      id: v.id,
      impact: v.impact,
      description: v.description,
      help: v.help,
      nodeCount: v.nodes.length,
    })),
    passes: results.passes.map((p) => ({ id: p.id })),
    incomplete: results.incomplete.map((i) => ({ id: i.id })),
  };

  expect(stableResults).toMatchSnapshot(`${snapshotName}-a11y.json`);
}
```

### 3.4 Quality Gate Thresholds

```typescript
// quality-gate.ts
export interface QualityGate {
  maxCriticalViolations: number;
  maxSeriousViolations: number;
  maxModerateViolations: number;
  blockOnCritical: boolean;
  blockOnSerious: boolean;
}

export const DEFAULT_QUALITY_GATE: QualityGate = {
  maxCriticalViolations: 0,   // Always block on critical
  maxSeriousViolations: 0,    // Always block on serious
  maxModerateViolations: 5,   // Allow up to 5 moderate
  blockOnCritical: true,
  blockOnSerious: true,
};

export function evaluateQualityGate(
  violations: Array<{ impact: string }>,
  gate: QualityGate = DEFAULT_QUALITY_GATE
): { passed: boolean; summary: string } {
  const criticalCount = violations.filter((v) => v.impact === 'critical').length;
  const seriousCount = violations.filter((v) => v.impact === 'serious').length;
  const moderateCount = violations.filter((v) => v.impact === 'moderate').length;

  const criticalFailed = gate.blockOnCritical && criticalCount > gate.maxCriticalViolations;
  const seriousFailed = gate.blockOnSerious && seriousCount > gate.maxSeriousViolations;
  const moderateFailed = moderateCount > gate.maxModerateViolations;

  const passed = !criticalFailed && !seriousFailed && !moderateFailed;

  const summary = [
    `Critical: ${criticalCount}/${gate.maxCriticalViolations} ${criticalFailed ? '❌' : '✅'}`,
    `Serious: ${seriousCount}/${gate.maxSeriousViolations} ${seriousFailed ? '❌' : '✅'}`,
    `Moderate: ${moderateCount}/${gate.maxModerateViolations} ${moderateFailed ? '❌' : '✅'}`,
  ].join(' | ');

  return { passed, summary };
}
```

### 5.2 Programmatic Announcement Testing Utility

```typescript
// screen-reader-test-utils.ts
import { Page, expect } from '@playwright/test';

/**
 * Collects all text that would be announced by a screen reader
 * for a given element, including ARIA labels, descriptions, and text content.
 */
export async function getAccessibleName(page: Page, selector: string): Promise<string> {
  return page.locator(selector).evaluate((el) => {
    // This runs in the browser context
    // Returns the computed accessible name per the Accessible Name and Description Computation
    const computeAccessibleName = (element: Element): string => {
      // Check aria-labelledby first
      const labelledBy = element.getAttribute('aria-labelledby');
      if (labelledBy) {
        const ids = labelledBy.split(/\s+/);
        const text = ids
          .map((id) => document.getElementById(id)?.textContent?.trim() ?? '')
          .filter(Boolean)
          .join(' ');
        if (text) return text;
      }

      // Check aria-label
      const ariaLabel = element.getAttribute('aria-label');
      if (ariaLabel?.trim()) return ariaLabel.trim();

      // Check associated label element
      if (element instanceof HTMLInputElement) {
        const id = element.id;
        if (id) {
          const label = document.querySelector(`label[for="${id}"]`);
          if (label?.textContent?.trim()) return label.textContent.trim();
        }
      }

      // Check title attribute
      const title = element.getAttribute('title');
      if (title?.trim()) return title.trim();

      // Check text content of the element
      const text = element.textContent?.trim();
      if (text) return text;

      // Check alt text for images
      if (element instanceof HTMLImageElement) {
        return element.alt || '';
      }

      return '';
    };

    return computeAccessibleName(el);
  });
}

/**
 * Captures all announcements that would be made by a screen reader
 * by intercepting live region updates and alert role changes.
 */
export async function captureAnnouncements(page: Page): Promise<{
  getAnnouncements: () => Promise<string[]>;
  clear: () => Promise<void>;
}> {
  // Inject a mutation observer that captures text added to live regions
  await page.evaluate(() => {
    const announcements: string[] = [];
    (window as any).__srAnnouncements = announcements;

    const observer = new MutationObserver((mutations) => {
      for (const mutation of mutations) {
        for (const node of mutation.addedNodes) {
          if (node instanceof HTMLElement) {
            // Check if this is inside a live region or has a live/alert role
            const liveRole = node.getAttribute('role') || '';
            const ariaLive = node.closest('[aria-live]');

            if (
              liveRole === 'alert' ||
              liveRole === 'status' ||
              liveRole === 'log' ||
              liveRole === 'marquee' ||
              ariaLive
            ) {
              const text = node.textContent?.trim();
              if (text) announcements.push(text);
            }
          }
        }
      }
    });

    observer.observe(document.body, {
      childList: true,
      subtree: true,
      characterData: true,
    });
  });

  return {
    getAnnouncements: async () => {
      return page.evaluate(() => {
        return (window as any).__srAnnouncements || [];
      });
    },
    clear: async () => {
      await page.evaluate(() => {
        (window as any).__srAnnouncements = [];
      });
    },
  };
}

/**
 * Assert that an element's accessible description includes specific text.
 */
export async function expectAccessibleDescription(
  page: Page,
  selector: string,
  expectedText: string
): Promise<void> {
  const describedBy = await page.locator(selector).getAttribute('aria-describedby');
  if (describedBy) {
    const descriptionEl = page.locator(`#${describedBy}`);
    await expect(descriptionEl).toContainText(expectedText);
  } else {
    // Some frameworks use a different pattern — check for sibling/adjacent description
    throw new Error(
      `Element ${selector} does not have aria-describedby. ` +
        `Check error message association pattern.`
    );
  }
}
```

```typescript
      expect(
        hasHeading > 0 || hasAriaLabel || hasAriaLabelledBy,
        `Section ${i} is missing a heading or accessible name`
      ).toBeTruthy();
    }
  });

  test('landmarks are unique and labeled', async ({ page }) => {
    await page.goto('/dashboard');

    // Check for non-unique landmarks
    const navLandmarks = page.locator('nav, [role="navigation"]');
    const navCount = await navLandmarks.count();

    if (navCount > 1) {
      // Multiple navigation landmarks must be uniquely labeled
      for (let i = 0; i < navCount; i++) {
        const nav = navLandmarks.nth(i);
        const label =
          (await nav.getAttribute('aria-label')) ||
          (await nav.getAttribute('aria-labelledby')) ||
          (await nav.getAttribute('title'));

        expect(
          label,
          `Navigation landmark ${i} has no unique label. ` +
            `Multiple navigation landmarks require aria-label or aria-labelledby.`
        ).toBeTruthy();
      }
    }
  });

  test('tables have proper headers', async ({ page }) => {
    await page.goto('/reports');

    const tables = page.locator('table');
    const tableCount = await tables.count();

    for (let i = 0; i < tableCount; i++) {
      const table = tables.nth(i);

      // Check for caption
      const caption = await table.locator('caption').count();
      if (caption === 0) {
        // Table might use aria-label or aria-labelledby instead
        const ariaLabel =
          (await table.getAttribute('aria-label')) ||
          (await table.getAttribute('aria-labelledby'));
        expect(
          ariaLabel,
          `Table ${i} has no caption or accessible name`
        ).toBeTruthy();
      }

      // Check for th elements or scope attributes
      const headers = await table.locator('th, [role="columnheader"]').count();
      expect(
        headers,
        `Table ${i} has no header cells`
      ).toBeGreaterThan(0);
    }
  });
});
```

```typescript
// form-accessibility.test.ts
import { test, expect } from '@playwright/test';

test.describe('Form Accessibility', () => {
  test('all inputs have associated labels', async ({ page }) => {
    await page.goto('/contact');

    const inputs = page.locator('input, select, textarea');
    const count = await inputs.count();

    for (let i = 0; i < count; i++) {
      const input = inputs.nth(i);
      const type = await input.getAttribute('type');

      // Skip hidden and submit inputs
      if (type === 'hidden' || type === 'submit' || type === 'button') continue;

      const id = await input.getAttribute('id');
      const ariaLabel = await input.getAttribute('aria-label');
      const ariaLabelledBy = await input.getAttribute('aria-labelledby');

      if (id) {
        // Check for explicit label association
        const labelCount = await page
          .locator(`label[for="${id}"]`)
          .count();
        if (labelCount > 0) continue; // Has explicit label — pass
      }

      if (ariaLabel || ariaLabelledBy) {
        continue; // Has ARIA label — pass
      }

      // Check if wrapped in a label element
      const parentLabel = input.locator('..');
      const parentTag = await parentLabel.evaluate((el) =>
        el.tagName.toLowerCase()
      );
      if (parentTag === 'label') continue; // Wrapped in label — pass

      // If we get here, the input has no accessible name
      const name = await input.getAttribute('name') || '(unnamed)';
      throw new Error(
        `Form input "${name}" at index ${i} has no associated label, ` +
          `aria-label, or aria-labelledby`
      );
    }
  });

  test('required fields have aria-required or required attribute', async ({
    page,
  }) => {
    await page.goto('/checkout');

    const requiredFields = page.locator('[required], [aria-required="true"]');
    const count = await requiredFields.count();
    expect(count).toBeGreaterThan(0);

    // Each required field should also visually indicate it's required
    for (let i = 0; i < count; i++) {
      const field = requiredFields.nth(i);
      const id = await field.getAttribute('id');

      if (id) {
        // Check if the label includes "(required)" or an asterisk
        const label = page.locator(`label[for="${id}"]`);
        const labelText = (await label.textContent()) || '';
        expect(
          labelText.includes('*') ||
            labelText.toLowerCase().includes('required'),
          `Required field ${id} should visually indicate it's required`
        ).toBeTruthy();
      }
    }
  });

  test('error messages are programmatically associated', async ({ page }) => {
    await page.goto('/signup');

    // Submit empty form to trigger validation
    await page.getByRole('button', { name: /sign up|submit/i }).click();

    // Wait for error messages
    await page.waitForSelector('[role="alert"], .error-message', {
      timeout: 3000,
    });

    // Check that each error is linked to its field
    const inputs = page.locator(
      'input:not([type="hidden"]):not([type="submit"])'
    );
    const inputCount = await inputs.count();

    for (let i = 0; i < inputCount; i++) {
      const input = inputs.nth(i);
      const describedBy = await input.getAttribute('aria-describedby');
      const ariaErrormessage = await input.getAttribute('aria-errormessage');

      // At least one association method should be present
      const hasErrorAssociation = describedBy || ariaErrormessage;
      console.log(
        `Input ${i}: aria-describedby=${describedBy}, aria-errormessage=${ariaErrormessage}`
      );

      // If the input is invalid, it MUST have an error association
      const isValid = await input.evaluate(
        (el) => (el as HTMLInputElement).validity.valid
      );
      if (!isValid) {
        expect(
          hasErrorAssociation,
          `Input ${i} is invalid but has no aria-describedby or aria-errormessage`
        ).toBeTruthy();
      }
    }
  });

  test('fieldset and legend group related fields', async ({ page }) => {
    await page.goto('/survey');

    const fieldsets = page.locator('fieldset');
    const fieldsetCount = await fieldsets.count();

    for (let i = 0; i < fieldsetCount; i++) {
      const fieldset = fieldsets.nth(i);

      // Each fieldset must have a legend
      const legendCount = await fieldset.locator('legend').count();
      expect(
        legendCount,
        `Fieldset ${i} has no <legend> element`
      ).toBeGreaterThan(0);
    }
  });
});
```

```typescript
// focus-management.test.ts
import { test, expect } from '@playwright/test';

test.describe('Focus Management', () => {
  test('focus moves to new content after navigation', async ({ page }) => {
    await page.goto('/');

    // Click a link that causes client-side navigation
    await page.click('a[href="/products"]');

    // Focus should move to the main content or a skip link target
    const focused = page.locator(':focus');
    const focusedId = await focused.getAttribute('id');
    const focusedRole = await focused.getAttribute('role');

    expect(
      focusedId === 'main-content' ||
        focusedRole === 'main' ||
        focusedId === 'skip-target'
    ).toBeTruthy();
  });

  test('modal receives focus when opened', async ({ page }) => {
    await page.goto('/settings');

    await page.click('[data-testid="open-delete-dialog"]');
    await page.waitForSelector('[role="dialog"], [role="alertdialog"]');

    // Focus should be inside the dialog
    const dialog = page.locator('[role="dialog"], [role="alertdialog"]');
    const focusedElement = page.locator(':focus');

    // Check that the focused element is a descendant of the dialog
    const isInDialog = await focusedElement.evaluate((el, dialogSelector) => {
      return el.closest(dialogSelector) !== null;
    }, '[role="dialog"], [role="alertdialog"]');

    expect(isInDialog).toBe(true);
  });

  test('focus returns to trigger after modal closes', async ({ page }) => {
    await page.goto('/settings');

    const trigger = page.locator('[data-testid="open-delete-dialog"]');
    await trigger.click();
    await page.waitForSelector('[role="dialog"]');

    // Close the dialog
    await page.keyboard.press('Escape');
    await expect(page.locator('[role="dialog"]')).not.toBeVisible();

    // Focus should return to the trigger
    await expect(trigger).toBeFocused();
  });

  test('skip-to-content link works', async ({ page }) => {
    await page.goto('/');

    // Press Tab to focus the skip link
    await page.keyboard.press('Tab');

    const skipLink = page.locator(':focus');
    const skipLinkText = await skipLink.textContent();
    expect(skipLinkText?.toLowerCase()).toContain('skip');

    // Activate the skip link
    await page.keyboard.press('Enter');

    // Focus should now be on the main content
    const focusedId = await page.locator(':focus').getAttribute('id');
    expect(focusedId).toBe('main-content');
  });

  test('focus indicator is always visible', async ({ page }) => {
    await page.goto('/');

    // Tab through interactive elements
    const interactiveCount = await page
      .locator('a[href], button, input, select, textarea, [tabindex="0"]')
      .count();

    for (let i = 0; i < Math.min(interactiveCount, 10); i++) {
      await page.keyboard.press('Tab');
      const focusedElement = page.locator(':focus');

      // Check for visible focus indicator
      const outline = await focusedElement.evaluate((el) => {
        const style = window.getComputedStyle(el);
        return {
          outline: style.outlineStyle,
          outlineWidth: style.outlineWidth,
          boxShadow: style.boxShadow,
          border: style.border,
        };
      });

      const hasVisibleIndicator =
        (outline.outline !== 'none' && parseFloat(outline.outlineWidth) > 0) ||
        (outline.boxShadow !== 'none' &&
          outline.boxShadow.includes('rgb')) ||
        (outline.border !== 'none' &&
          outline.border !== '0px none rgb(0, 0, 0)');

      expect(
        hasVisibleIndicator,
        `Element ${i} has no visible focus indicator`
      ).toBeTruthy();
    }
  });
});
```

```typescript
// screen-reader-flow.test.ts
import { test, expect } from '@playwright/test';
import { captureAnnouncements } from './screen-reader-test-utils';

test.describe('Screen Reader User Flow — Checkout', () => {
  test('complete checkout flow is navigable and announced properly', async ({
    page,
  }) => {
    const announcer = await captureAnnouncements(page);

    // Step 1: Navigate to checkout
    await page.goto('/checkout');
    let announcements = await announcer.getAnnouncements();

    // At minimum, the page title/live region should announce the page context
    expect(announcements.length).toBeGreaterThanOrEqual(0); // May or may not auto-announce
    await announcer.clear();

    // Step 2: Fill shipping information
    await page.getByLabel('Full Name').fill('Jane Doe');
    await page.getByLabel('Address').fill('123 Main St');
    await page.getByLabel('City').fill('Portland');

    // Tab to next step
    await page.getByRole('button', { name: 'Continue to Payment' }).click();

    // Should get a polite announcement about moving to the next step
    announcements = await announcer.getAnnouncements();
    const stepAnnouncement = announcements.some((a) =>
      a.toLowerCase().includes('payment')
    );
    expect(stepAnnouncement).toBe(true);
    await announcer.clear();

    // Step 3: Fill payment information
    await page.getByLabel('Card Number').fill('4111111111111111');
    await page.getByLabel('Expiration').fill('12/28');
    await page.getByLabel('CVC').fill('123');

    // Submit order
    await page.getByRole('button', { name: 'Place Order' }).click();

    // Should get a status/alert announcement about the order
    await page.waitForTimeout(2000); // Allow async announcements
    announcements = await announcer.getAnnouncements();

    const orderAnnouncement = announcements.some(
      (a) =>
        a.toLowerCase().includes('order') ||
        a.toLowerCase().includes('confirmation') ||
        a.toLowerCase().includes('success')
    );
    expect(orderAnnouncement).toBe(true);
  });
```
