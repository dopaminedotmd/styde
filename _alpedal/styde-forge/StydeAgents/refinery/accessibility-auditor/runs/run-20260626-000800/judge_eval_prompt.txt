```javascript
// Tabs Keyboard & Interaction Logic
class AccessibleTabs {
  constructor(tablist) {
    this.tablist = tablist;
    this.tabs = [...tablist.querySelectorAll('[role="tab"]')];
    this.panels = this.tabs.map(tab =>
      document.getElementById(tab.getAttribute('aria-controls'))
    );
    this.activeIndex = 0;

    this.tabs.forEach((tab, i) => {
      tab.addEventListener('click', () => this.activateTab(i));
      tab.addEventListener('keydown', (e) => this.handleKeydown(e, i));
    });
  }

  activateTab(index) {
    this.tabs.forEach((tab, i) => {
      const selected = i === index;
      tab.setAttribute('aria-selected', String(selected));
      tab.tabIndex = selected ? 0 : -1;
    });
    this.panels.forEach((panel, i) => {
      panel.hidden = i !== index;
    });
    this.activeIndex = index;
  }

  handleKeydown(event, index) {
    let newIndex;
    switch (event.key) {
      case 'ArrowRight':
        event.preventDefault();
        newIndex = (index + 1) % this.tabs.length;
        break;
      case 'ArrowLeft':
        event.preventDefault();
        newIndex = (index - 1 + this.tabs.length) % this.tabs.length;
        break;
      case 'Home':
        event.preventDefault();
        newIndex = 0;
        break;
      case 'End':
        event.preventDefault();
        newIndex = this.tabs.length - 1;
        break;
      default:
        return;
    }
    this.activateTab(newIndex);
    this.tabs[newIndex].focus();
  }
}
```

**Checklist — Tabs:**
- ☐ `role="tablist"` with accessible name (`aria-label` or `aria-labelledby`)
- ☐ Each tab has `role="tab"`, `aria-selected`, `aria-controls` (pointing to panel `id`)
- ☐ Only active tab is in tab order (`tabindex="0"`); inactive tabs `tabindex="-1"`
- ☐ Each panel has `role="tabpanel"`, `aria-labelledby` (pointing to tab `id`), `tabindex="0"`
- ☐ Inactive panels use `hidden` attribute
- ☐ Arrow keys switch tabs (Left/Right for horizontal, Up/Down for vertical)
- ☐ Home/End move to first/last tab

```javascript
class Accordion {
  constructor(container) {
    this.container = container;
    this.buttons = [...container.querySelectorAll('[aria-expanded]')];

    this.buttons.forEach(btn => {
      btn.addEventListener('click', () => this.toggle(btn));
      btn.addEventListener('keydown', (e) => this.handleKeydown(e, btn));
    });
  }

  toggle(button) {
    const expanded = button.getAttribute('aria-expanded') === 'true';
    const panel = document.getElementById(button.getAttribute('aria-controls'));
    button.setAttribute('aria-expanded', String(!expanded));
    panel.hidden = expanded;
  }

  handleKeydown(event, button) {
    const index = this.buttons.indexOf(button);
    switch (event.key) {
      case 'ArrowDown':
        event.preventDefault();
        const next = this.buttons[(index + 1) % this.buttons.length];
        next.focus();
        break;
      case 'ArrowUp':
        event.preventDefault();
        const prev = this.buttons[(index - 1 + this.buttons.length) % this.buttons.length];
        prev.focus();
        break;
      case 'Home':
        event.preventDefault();
        this.buttons[0].focus();
        break;
      case 'End':
        event.preventDefault();
        this.buttons[this.buttons.length - 1].focus();
        break;
    }
  }
}
```

```javascript
class AccessibleDialog {
  constructor(dialogId, triggerId) {
    this.dialog = document.getElementById(dialogId);
    this.trigger = document.getElementById(triggerId);
    this.overlay = this.dialog.previousElementSibling;
    this.closeBtn = this.dialog.querySelector('.dialog-close');
    this.cancelBtn = this.dialog.querySelector('#dialog-cancel');
    this.saveBtn = this.dialog.querySelector('#dialog-save');

    // All focusable elements inside dialog
    this.focusableSelector = 'a[href], button:not([disabled]), textarea, input, select, [tabindex]:not([tabindex="-1"])';
    this.previousActiveElement = null;

    this.trigger.addEventListener('click', () => this.open());
    this.closeBtn.addEventListener('click', () => this.close());
    this.cancelBtn?.addEventListener('click', () => this.close());
    [this.dialog, this.overlay].forEach(el => {
      el?.addEventListener('click', (e) => {
        if (e.target === el) this.close();
      });
    });
    this.dialog.addEventListener('keydown', (e) => this.handleKeydown(e));
  }

  open() {
    this.previousActiveElement = document.activeElement;
    this.dialog.hidden = false;
    this.overlay.hidden = false;
    // Prevent background scroll
    document.body.style.overflow = 'hidden';

    // Focus first focusable element (or close button as fallback)
    const focusTarget = this.dialog.querySelector(this.focusableSelector);
    if (focusTarget) focusTarget.focus();
  }

  close() {
    this.dialog.hidden = true;
    this.overlay.hidden = true;
    document.body.style.overflow = '';
    // Return focus to trigger
    this.previousActiveElement?.focus();
  }

  handleKeydown(event) {
    if (event.key === 'Escape') {
      event.preventDefault();
      this.close();
      return;
    }
    // Trap focus
    if (event.key === 'Tab') {
      const focusable = [...this.dialog.querySelectorAll(this.focusableSelector)];
      const first = focusable[0];
      const last = focusable[focusable.length - 1];

      if (event.shiftKey) {
        if (document.activeElement === first) {
          event.preventDefault();
          last.focus();
        }
      } else {
        if (document.activeElement === last) {
          event.preventDefault();
          first.focus();
        }
      }
    }
  }
}
```

- ☐ Non-modal content marked with `aria-hidden="true"` or dialog is top-layer
- ☐ Background scroll is prevented (`overflow: hidden` on body)

- ☐ Escape closes the dialog
- ☐ Clicking backdrop closes dialog

```javascript
class AccessibleMenu {
  constructor(menubarSelector) {
    this.menubar = document.querySelector(menubarSelector);
    this.menuItems = [...this.menubar.querySelectorAll('[role="menuitem"]')];
    this.submenus = {};

    this.menuItems.forEach(item => {
      const menu = item.nextElementSibling;
      if (menu && menu.getAttribute('role') === 'menu') {
        this.submenus[item.id] = menu;
        item.addEventListener('click', () => this.toggleSubmenu(item));
      }
    });

    this.menubar.addEventListener('keydown', (e) => this.handleKeydown(e));
    this.menubar.addEventListener('mouseover', (e) => {
      const item = e.target.closest('[role="menuitem"]');
      if (item && this.submenus[item.id]) this.openSubmenu(item);
    });
    // Close when clicking outside
    document.addEventListener('click', (e) => {
      if (!this.menubar.contains(e.target)) this.closeAll();
    });
  }

  toggleSubmenu(trigger) {
    const expanded = trigger.getAttribute('aria-expanded') === 'true';
    if (expanded) this.closeSubmenu(trigger);
    else this.openSubmenu(trigger);
  }

  openSubmenu(trigger) {
    this.closeAll();
    trigger.setAttribute('aria-expanded', 'true');
    this.submenus[trigger.id].hidden = false;
    // Focus first item in submenu
    const firstItem = this.submenus[trigger.id].querySelector('[role="menuitem"]');
    firstItem?.focus();
  }

  closeSubmenu(trigger) {
    trigger.setAttribute('aria-expanded', 'false');
    this.submenus[trigger.id].hidden = true;
  }

  closeAll() {
    Object.keys(this.submenus).forEach(id => {
      const trigger = document.getElementById(id);
      if (trigger) this.closeSubmenu(trigger);
    });
  }

  handleKeydown(event) {
    const items = [...this.menubar.querySelectorAll('[role="menuitem"]:not([hidden])')];
    const current = document.activeElement;
    const idx = items.indexOf(current);
    const menu = current.closest('[role="menu"], [role="menubar"]');
    const isVertical = menu?.getAttribute('role') === 'menu';

    switch (event.key) {
      case isVertical ? 'ArrowDown' : 'ArrowRight':
        event.preventDefault();
        items[(idx + 1) % items.length].focus();
        break;
      case isVertical ? 'ArrowUp' : 'ArrowLeft':
        event.preventDefault();
        items[(idx - 1 + items.length) % items.length].focus();
        break;
      case 'Escape':
        event.preventDefault();
        const parentMenu = menu?.closest('[role="menu"]');
        if (parentMenu) {
          this.closeAll();
          // Focus parent trigger
          const trigger = this.menubar.querySelector(`[aria-controls="${parentMenu.id}"]`);
          trigger?.focus();
        }
        break;
      case 'Enter':
      case ' ':
        event.preventDefault();
        if (current.tagName === 'A') {
          current.click();
        } else if (this.submenus[current.id]) {
          this.openSubmenu(current);
        }
        break;
    }
  }
}
```

- ☐ Top-level uses `role="menubar"`; submenu uses `role="menu"`
- ☐ Items use `role="menuitem"`; list items use `role="none"`
- ☐ Parent triggers have `aria-expanded` and `aria-haspopup="true"`
- ☐ Submenus labelled with `aria-labelledby` pointing to trigger
- ☐ Arrow keys navigate within a menu level; Escape closes submenu
- ☐ Enter/Space activates the focused item
- ☐ Hidden submenus use `hidden` attribute
- ☐ `aria-current="page"` on the current page's menu item

```javascript
    // Click on option
    this.listbox.addEventListener('click', (e) => {
      const option = e.target.closest('[role="option"]');
      if (option) this.selectOption(option);
    });
  }

  filter() {
    const query = this.input.value.toLowerCase().trim();
    if (!query) { this.hideListbox(); return; }

    this.options = this.dataSource.filter(item =>
      item.toLowerCase().includes(query)
    ).slice(0, 10);

    this.renderOptions();
    this.announcer.textContent = `${this.options.length} suggestion${this.options.length !== 1 ? 's' : ''} available.`;
  }

  renderOptions() {
    this.listbox.innerHTML = this.options.map((opt, i) => `
      <li role="option" id="city-opt-${i}" aria-selected="false">
        ${this.highlightMatch(opt, this.input.value)}
      </li>
    `).join('');

    this.activeIndex = -1;
    this.input.removeAttribute('aria-activedescendant');

    if (this.options.length > 0) {
      this.showListbox();
    } else {
      this.hideListbox();
    }
  }

  highlightMatch(text, query) {
    const regex = new RegExp(`(${query.replace(/[.*+?^${}()|[\]\\]/g, '\\$&')})`, 'gi');
    return text.replace(regex, '<strong>$1</strong>');
  }

  showListbox() {
    this.listbox.hidden = false;
    this.input.setAttribute('aria-expanded', 'true');
  }

  hideListbox() {
    this.listbox.hidden = true;
    this.input.setAttribute('aria-expanded', 'false');
    this.input.removeAttribute('aria-activedescendant');
  }

  selectOption(option) {
    this.input.value = option.textContent.replace(/<[^>]*>/g, ''); // strip HTML
    this.hideListbox();
    this.input.focus();
  }

  setActiveDescendant(index) {
    const options = [...this.listbox.querySelectorAll('[role="option"]')];
    options.forEach(o => o.setAttribute('aria-selected', 'false'));
    options.forEach(o => o.classList.remove('active'));

    if (index >= 0 && index < options.length) {
      options[index].setAttribute('aria-selected', 'true');
      options[index].classList.add('active');
      this.input.setAttribute('aria-activedescendant', options[index].id);
      options[index].scrollIntoView({ block: 'nearest' });
    } else {
      this.input.removeAttribute('aria-activedescendant');
    }
  }

  handleKeydown(event) {
    const options = [...this.listbox.querySelectorAll('[role="option")'];

    switch (event.key) {
      case 'ArrowDown':
        event.preventDefault();
        if (this.listbox.hidden && this.options.length > 0) {
          this.showListbox();
        }
        this.activeIndex = Math.min(this.activeIndex + 1, options.length - 1);
        this.setActiveDescendant(this.activeIndex);
        break;

      case 'ArrowUp':
        event.preventDefault();
        this.activeIndex = Math.max(this.activeIndex - 1, 0);
        this.setActiveDescendant(this.activeIndex);
        break;

      case 'Enter':
        event.preventDefault();
        if (this.activeIndex >= 0 && options[this.activeIndex]) {
          this.selectOption(options[this.activeIndex]);
        }
        break;

      case 'Escape':
        event.preventDefault();
        this.hideListbox();
        this.input.focus();
        break;

      case 'Tab':
        this.hideListbox();
        break;
    }
  }
}
```

**Checklist — Combobox:**
- ☐ `<input>` has `role="combobox"`, `aria-expanded`, `aria-controls`, `aria-autocomplete="list"`
- ☐ Listbox uses `role="listbox"` with `aria-label`
- ☐ Options use `role="option"` with `aria-selected`
- ☐ Active descendant tracked with `aria-activedescendant` on input
- ☐ Arrow Up/Down navigates options without moving focus from input
- ☐ Enter selects the active option
- ☐ Escape closes the listbox
- ☐ Live region announces number of available suggestions

### 3.3 Roving Tabindex Pattern

Used for grids, listboxes, tablists, menus:

```javascript
// Instead of tabindex="0" on every item, use roving tabindex:
// - Only ONE item has tabindex="0" (the active/focused one)
// - All others have tabindex="-1"
// - Arrow keys move both focus AND tabindex

function moveFocus(container, currentIndex, newIndex, role) {
  const items = [...container.querySelectorAll(`[role="${role}"]`)];
  items[currentIndex].tabIndex = -1;
  items[newIndex].tabIndex = 0;
  items[newIndex].focus();
}
```

### 3.5 Focus Management After Dynamic Updates

```javascript
// Pattern: When content is added to the DOM, manage focus appropriately

// Delete an item from a list → move focus to next item (or previous if last)
function handleDelete(item) {
  const list = item.parentElement;
  const nextItem = item.nextElementSibling || item.previousElementSibling;
  list.removeChild(item);
  if (nextItem?.tabIndex !== undefined) {
    nextItem.tabIndex = 0;
    nextItem.focus();
  } else {
    // Fall back to container
    list.closest('[tabindex]')?.focus();
  }
}

// Add an item → move focus to the new item
function handleAdd(list, newItemHTML) {
  list.insertAdjacentHTML('beforeend', newItemHTML);
  const newItem = list.lastElementChild;
  if (newItem) {
    // Reset roving tabindex
    [...list.children].forEach(c => { c.tabIndex = -1; });
    newItem.tabIndex = 0;
    newItem.focus();
  }
}

// Infinite scroll / load more → focus stays; announce update
function handleLoadMore(container) {
  const announcer = document.getElementById('live-region');
  announcer.textContent = '10 more items loaded. Total: 45 items.';
}
```

### 4.3 Screen Reader Testing Protocol

#### Test 1: Page Load Announcement
1. Load the page with screen reader running.
2. **Expected:** Page `<title>` is announced. Landmarks are discoverable.
3. **Check:** "Skip to content" link is the first focusable element.

#### Test 2: Heading Hierarchy
1. Open headings list (NVDA: Insert+F6, VoiceOver: VO+U → headings).
2. **Expected:** Logical hierarchy (H1 → H2 → H3, no skipped levels).
3. **Check:** Headings are descriptive and unique.

#### Test 3: Link Purpose
1. Navigate by links only (NVDA: K, VoiceOver: VO+U → links).
2. **Expected:** Each link text describes its purpose out of context.
3. **Avoid:** "click here", "read more", "learn more", bare URLs.

#### Test 4: Form Interaction
1. Navigate to form fields (NVDA: F, VoiceOver: VO+U → form controls).
2. **Expected:** Each field announces its label, type, and state.
3. **Check:** Required fields announce "required". Errors are linked to fields.

#### Test 5: Image Alternatives
1. Navigate by graphics (NVDA: G, VoiceOver: VO+U → images).
2. **Expected:** Content images announce meaningful alt text. Decorative images are silent.
3. **Check:** Complex images (charts) have `aria-describedby` linking to full description.

#### Test 6: Dynamic Content
1. Trigger dynamic updates (form submission, filters, modal open).
2. **Expected:** Status messages announced via `aria-live`. Focus moves appropriately for modals.
3. **Check:** Loader/spinner state is communicated.

#### Test 7: Data Tables
1. Navigate into a table (T).
2. **Expected:** Column and row headers are announced as you navigate cells.
3. **Check:** `scope="col"` / `scope="row"` on `<th>`. Caption via `<caption>`.

#### Test 8: Custom Widgets
1. Interact with tabs, accordions, modals, menus, comboboxes.
2. **Expected:** Role, state, and value are announced. Keyboard interactions match WAI-ARIA patterns.
3. **Check:** `aria-expanded`, `aria-selected`, `aria-modal` updates are announced.

### 4.4 Common Screen Reader Issues & Fixes

| Issue | Cause | Fix |
|-------|-------|-----|
| "Blank" read on element | Missing accessible name | Add `aria-label`, `aria-labelledby`, or text content |
| "Clickable" but no role | `<div onclick="...">` | Use `<button>` or add `role="button"` + `tabindex="0"` |
| Repeated content announced | Multiple elements with same label | Add unique context or `aria-describedby` |
| Dynamic update silent | No live region | Add `role="status"` or `aria-live="polite"` |
| Table headers silent | `<td>` used for headers | Use `<th scope="col/row">` |
| Modal content readable in background | Missing `aria-modal="true"` | Add `aria-modal="true"` + `aria-hidden="true"` on main |
| Focus disappears | Managed focus lost | Reapply focus to a logical element |

### 5.3 JavaScript Contrast Calculator

```javascript
/**
 * WCAG 2.2 Color Contrast Calculator
 * Calculates relative luminance and contrast ratio per WCAG definition
 */

function hexToRgb(hex) {
  const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
  if (!result) return null;
  return {
    r: parseInt(result[1], 16),
    g: parseInt(result[2], 16),
    b: parseInt(result[3], 16)
  };
}

function getLuminance(r, g, b) {
  const [rs, gs, bs] = [r, g, b].map(c => {
    c = c / 255;
    return c <= 0.03928 ? c / 12.92 : Math.pow((c + 0.055) / 1.055, 2.4);
  });
  return 0.2126 * rs + 0.7152 * gs + 0.0722 * bs;
}

function contrastRatio(foreground, background) {
  const fgRgb = hexToRgb(foreground);
  const bgRgb = hexToRgb(background);
  if (!fgRgb || !bgRgb) return null;

  const lum1 = getLuminance(fgRgb.r, fgRgb.g, fgRgb.b);
  const lum2 = getLuminance(bgRgb.r, bgRgb.g, bgRgb.b);

  const lighter = Math.max(lum1, lum2);
  const darker = Math.min(lum1, lum2);
  return (lighter + 0.05) / (darker + 0.05);
}

function evaluateContrast(fgHex, bgHex, isLargeText = false) {
  const ratio = contrastRatio(fgHex, bgHex);
  const threshold = isLargeText ? 3.0 : 4.5;

  if (ratio === null) return { ratio: null, pass: false, level: 'Invalid color' };

  const pass = ratio >= threshold;
  return {
    ratio: ratio.toFixed(2),
    pass,
    aa: ratio >= threshold ? '✅ Pass AA' : '❌ Fail AA',
    aaa: ratio >= (isLargeText ? 4.5 : 7.0) ? '✅ Pass AAA' : '❌ Fail AAA',
    isLargeText,
    foreground: fgHex,
    background: bgHex
  };
}

// ─── Examples ───────────────────────────────────────────

// Example 1: Gray text on white — FAILS

// → { ratio: '4.54', pass: true, aa: '✅ Pass AA', ... }
// Borderline — #767676 on white passes at 4.54:1

// Example 2: Pure gray #949494 — FAILS for normal text
console.log(evaluateContrast('#949494', '#FFFFFF'));
// → { ratio: '3.04', pass: false, aa: '❌ Fail AA', ... }

// Example 3: Headline — large text, lower bar
console.log(evaluateContrast('#767676', '#FFFFFF', true));
// → { ratio: '4.54', pass: true, aa: '✅ Pass AA', ... }
// Large text only needs 3:1, so this easily passes

// Example 4: Blue on white (link color)
console.log(evaluateContrast('#2563EB', '#FFFFFF'));
// → { ratio: '4.50', pass: true, aa: '✅ Pass AA', ... }

// Example 5: Dark mode — white text on dark gray

// → { ratio: '15.63', pass: true, aa: '✅ Pass AA', aaa: '✅ Pass AAA' }
```

### 6.2 Form Validation Pattern

```javascript
class AccessibleFormValidator {
  constructor(formSelector) {
    this.form = document.querySelector(formSelector);
    this.fields = [...this.form.querySelectorAll('input, textarea, select')];
    this.errorSummaryId = 'error-summary';

    this.form.addEventListener('submit', (e) => this.handleSubmit(e));
    this.fields.forEach(field => {
      field.addEventListener('blur', () => this.validateField(field));
      field.addEventListener('input', () => {
        // Clear error on input after first validation
        if (field.dataset.touched === 'true') {
          this.clearError(field);
        }
      });
    });
  }

  handleSubmit(event) {
    let hasErrors = false;
    const errors = [];

    this.fields.forEach(field => {
      field.dataset.touched = 'true';
      const error = this.validateField(field);
      if (error) {
        hasErrors = true;
        errors.push({ field, error });
      }
    });

    if (hasErrors) {
      event.preventDefault();
      this.showErrorSummary(errors);
      // Focus first field with an error
      errors[0].field.focus();
    }
  }

  validateField(field) {
    this.clearError(field);

    // Required check
    if (field.required && !field.value.trim()) {
      return this.setError(field, `${this.getLabel(field)} is required.`);
    }

    // Type-specific validation
    if (field.type === 'email' && field.value) {
      const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
      if (!emailRegex.test(field.value)) {
        return this.setError(field, 'Please enter a valid email address.');
      }
    }

    // Min/max length
    if (field.minLength && field.value.length < field.minLength) {
      return this.setError(field,
        `Must be at least ${field.minLength} characters.`);
    }

    if (field.maxLength && field.value.length > field.maxLength) {
      return this.setError(field,
        `Must be no more than ${field.maxLength} characters.`);
    }

    return null;
  }

  setError(field, message) {
    const errorId = `${field.id}-error`;
    let errorEl = document.getElementById(errorId);

    if (!errorEl) {
      errorEl = document.createElement('p');
      errorEl.id = errorId;
      errorEl.className = 'error';
      errorEl.setAttribute('role', 'alert');
      field.parentNode.appendChild(errorEl);
    }

    errorEl.textContent = message;
    errorEl.hidden = false;
    field.setAttribute('aria-invalid', 'true');
    field.setAttribute('aria-describedby', errorId);
    field.classList.add('has-error');

    return message;
  }

  clearError(field) {
    field.setAttribute('aria-invalid', 'false');
    field.classList.remove('has-error');
    const errorEl = document.getElementById(`${field.id}-error`);
    if (errorEl) {
      errorEl.hidden = true;
      errorEl.textContent = '';
    }
  }

  getLabel(field) {
    const label = document.querySelector(`label[for="${field.id}"]`);
    return label ? label.textContent.replace('*', '').trim() : 'This field';
  }

  showErrorSummary(errors) {
    let summary = document.getElementById(this.errorSummaryId);
    if (!summary) {
      summary = document.createElement('div');
      summary.id = this.errorSummaryId;
      summary.setAttribute('role', 'alert');
      summary.className = 'error-summary';
      this.form.prepend(summary);
    }

    summary.innerHTML = `
      <h2>There ${errors.length === 1 ? 'is' : 'are'} ${errors.length}
          problem${errors.length === 1 ? '' : 's'} with your submission</h2>
      <ul>
        ${errors.map(({ field, error }) =>
          `<li><a href="#${field.id}">${error}</a></li>`
        ).join('')}
      </ul>
    `;
    summary.focus();
  }
}
```

### 6.5 Multi-Step Form / Wizard Pattern

```html
<div class="wizard" aria-label="Checkout — 3 steps">
  <!-- Step indicator -->
  <ol class="step-indicator" aria-label="Progress">
    <li aria-current="step" class="active">Shipping</li>
    <li>Payment</li>
    <li>Review</li>
  </ol>

  <!-- Step panels -->
  <div role="tabpanel" id="step-1" aria-labelledby="step-title-1">
    <h2 id="step-title-1">Shipping Address</h2>
    <!-- Step 1 form fields -->
    <div class="wizard-actions">
      <button type="button" id="next-step-1">Continue to Payment</button>
    </div>
  </div>

  <div role="tabpanel" id="step-2" aria-labelledby="step-title-2" hidden>
    <h2 id="step-title-2">Payment Details</h2>
    <!-- Step 2 form fields -->
    <div class="wizard-actions">
      <button type="button" id="prev-step-2">Back to Shipping</button>
      <button type="button" id="next-step-2">Continue to Review</button>
    </div>
  </div>

  <div role="tabpanel" id="step-3" aria-labelledby="step-title-3" hidden>
    <h2 id="step-title-3">Review & Submit</h2>
    <!-- Step 3 summary -->
    <div class="wizard-actions">
      <button type="button" id="prev-step-3">Back to Payment</button>
      <button type="submit">Place Order</button>
    </div>
  </div>
</div>

<!-- Live region for step change announcements -->
<div class="sr-only" aria-live="polite" id="wizard-announcer"></div>
```

```javascript
class Wizard {
  constructor() {
    this.currentStep = 1;
    this.totalSteps = 3;
    this.announcer = document.getElementById('wizard-announcer');

    // Wire up navigation buttons
    document.querySelectorAll('[id^="next-step-"]').forEach(btn => {
      btn.addEventListener('click', () => this.goToStep(this.currentStep + 1));
    });
    document.querySelectorAll('[id^="prev-step-"]').forEach(btn => {
      btn.addEventListener('click', () => this.goToStep(this.currentStep - 1));
    });
  }

  goToStep(step) {
    // Hide all panels
    document.querySelectorAll('[role="tabpanel"]').forEach(panel => {
      panel.hidden = true;
    });

    // Show target
    const target = document.getElementById(`step-${step}`);
    target.hidden = false;
    this.currentStep = step;

    // Update step indicator
    document.querySelectorAll('.step-indicator li').forEach((li, i) => {
      li.removeAttribute('aria-current');
      if (i + 1 === step) li.setAttribute('aria-current', 'step');
    });

    // Focus the heading of the new step
    target.querySelector('h2')?.focus();

    // Announce step change
    this.announcer.textContent = `Step ${step} of ${this.totalSteps}`;

    // Update browser history
    history.pushState({ step }, '', `#step-${step}`);
  }
}
```

**Checklist — Multi-Step Form:**
- ☐ Step indicator is an `<ol>` with `aria-current="step"` on the active step
- ☐ Each step panel has `role="tabpanel"` and `aria-labelledby`
- ☐ Moving between steps announces the change via a live region
- ☐ Focus moves to the new step heading
- ☐ "Back" and "Continue" buttons are clearly labeled
- ☐ Progress is preserved (state is saved between steps)
