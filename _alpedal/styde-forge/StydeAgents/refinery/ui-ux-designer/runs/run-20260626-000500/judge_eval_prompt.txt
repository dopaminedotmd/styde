## 1. Design Overview

### 1.1 Product Vision
A modern, fully-accessible project management dashboard for SaaS teams. ### 1.2 Key Design Decisions

| Decision | Rationale |
|---|---|
| **Single-file architecture** | Zero dependencies, zero build step — open in any browser |
| **CSS Custom Properties** | Seamless theming (light/dark/high-contrast) without JS dependency |
| **State-first component design** | Every component supports 4 states: loading (skeleton), empty, error, data |
| **Native HTML Drag-and-Drop** | Zero-library kanban drag-and-drop using HTML5 Drag and Drop API |
| **Full keyboard navigation** | Arrow keys for kanban columns/cards, Escape for modals, Tab for all interactive elements |
| **Comprehensive ARIA live announcements** | Every state transition and user action announces to screen readers |
| **`prefers-reduced-motion`** | Respects OS settings; skeletons and transitions disabled |
| **Skip-to-content link** | First focusable element for keyboard users |
| **Focus-visible outlines** | 3px ring with 2px offset on `:focus-visible` |

  function switchToView(viewId) {
    navItems.forEach(function(item) {

      item.setAttribute('aria-current', isActive ? 'page' : 'false');
      item.setAttribute('aria-selected', isActive ? 'true' : 'false');
    });

    viewPanels.forEach(function(panel) {
      panel.classList.toggle('active', panel.id === 'view-' + viewId);
    });

    if (pageTitle && viewTitles[viewId]) {
      pageTitle.textContent = viewTitles[viewId];
    }

    if (history.pushState) {
      history.pushState(null, '', '#' + viewId);
    }

    announce('Switched to ' + viewTitles[viewId] + ' view');

    var activePanel = document.getElementById('view-' + viewId);
    if (activePanel) {
      activePanel.setAttribute('tabindex', '-1');
      activePanel.focus({ preventScroll: true });
    }

    if (window.innerWidth <= 768) {
      sidebar.classList.remove('open');
      var overlay = document.querySelector('.mobile-overlay');
      if (overlay) overlay.remove();

      if (e.key === 'Enter' || e.key === ' ') {

        if (vid) switchToView(vid);

  if (initialHash && viewTitles[initialHash]) {
    switchToView(initialHash);

  // THEME TOGGLE
  // ============================================================
  var themeToggle = document.getElementById('themeToggle');

    'light': 'Switch to dark theme',
    'dark': 'Switch to high contrast theme',
    'high-contrast': 'Switch to light theme'

  var currentTheme = savedTheme || 'light';
