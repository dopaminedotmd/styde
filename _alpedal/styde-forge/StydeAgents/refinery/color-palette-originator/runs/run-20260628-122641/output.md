usage-guide:
  accent-role:
    light-mode: Use for interactive components, buttons, toggles, active nav. accent-dark for text-on-accent-bg and hover borders. accent-subtle for filled backgrounds behind accent content.
    dark-mode: accent (brighter variant) maintains the same emotional temperature. accent-dark becomes the sober variant for pressed states and subtle emphasis.
  interactive-states:
    non-destructive: neutral ramp (interactive-hover/pressed/disabled) for secondary and ghost actions. accent for primary actions.
    destructive: reserve critical actions for a future red token; not included in this palette.
  when-to-pair:
    svalt-koppar: data dashboards, professional SaaS, editor interfaces, legal/finance tools
    morgon-i-drivhus: wellness apps, agriculture/outdoor SaaS, creative tools, children's platforms
    midnatts-tag: reading apps, travel interfaces, media players, journaling tools, museum guides
