## 1. Architecture Overview

This architecture models a **multi-tenant dashboard application** with the following domain:

- **Marketing site** (`/`) — public landing pages
- **Authentication** (`/login`, `/signup`, `/logout`) — auth flows
- **Dashboard** (`/app`) — authenticated layout
  - **Projects** (`/app/projects`) — CRUD with nested routes
    - **Project detail** (`/app/projects/:id`) — single project
      - **Settings** (`/app/projects/:id/settings`) — nested settings
      - **Members** (`/app/projects/:id/members`) — team management
  - **Billing** (`/app/billing`) — subscription management

### Key Patterns Demonstrated

| Pattern | Where Used |
|---|---|
| Nested layouts | `__app.tsx` → `__app.projects.tsx` → `__app.projects.$id.tsx` |
| Zod validation (loader) | Every loader that accepts params/query |
| Zod validation (action) | Every form action |
| Progressive enhancement | `<Form>` with `useActionData`, no-JS fallback |
| Error boundaries | Per-route `ErrorBoundary` exports |
| Optimistic UI | `useFetcher` on project star/unstar, member role toggles |
| Deferred loaders | Project detail page (streaming) |

```tsx
  { charset: "utf-8" },
  { title: "Styde Forge — Remix Architecture" },
  { name: "viewport", content: "width=device-width,initial-scale=1" },
];

export async function loader({ request }: LoaderFunctionArgs) {
  const user = await getUser(request);
  return { user, ENV: { API_URL: process.env.API_URL } };

      <html lang="en" className="h-full">

          <Links />
        </head>

          <h1 className="text-3xl font-bold text-red-600">Unexpected Error</h1>

            {message}
          </pre>
          <a href="/" className="mt-6 inline-block text-indigo-600 hover:underline">
            Go back home
          </a>
        </div>
        <Scripts />
```

### 4.2 Zod Error Formatting Utility

```typescript
// app/lib/zod-error.ts
import { ZodError } from "zod";

export type FormErrors<T> = Partial<Record<keyof T, string>> & {
  _form?: string;
};

/**
 * Flattens a ZodError into a key→message map suitable for form display.
 */
export function formatZodError<T>(error: ZodError): FormErrors<T> {
  const fieldErrors: FormErrors<T> = {};

  for (const issue of error.issues) {
    const path = issue.path.join(".") as keyof T;
    if (!fieldErrors[path]) {
      fieldErrors[path] = issue.message;
    }
  }

  return fieldErrors;
}

/**
 * Returns a user-friendly error string from a ZodError.
 */
export function formatZodErrorString(error: ZodError): string {
  return error.issues.map((i) => `${i.path.join(".")}: ${i.message}`).join("; ");
}

/**
 * Parses and returns data, or throws a 400 Response with formatted errors.
 * Use this inside loaders that must have valid params.
 */
export function parseOrThrow<T>(schema: { safeParse: (data: unknown) => { success: true; data: T } | { success: false; error: ZodError } }, data: unknown): T {
  const result = schema.safeParse(data);
  if (!result.success) {
    throw new Response(formatZodErrorString(result.error), {
      status: 400,
      statusText: "Bad Request",
    });
  }
  return result.data;
}
```

### 4.3 Loader Pattern: Project List with Zod Query Validation

```tsx
// app/routes/__app.projects._index.tsx
import type { LoaderFunctionArgs, MetaFunction } from "@remix-run/node";
import { useLoaderData, useSearchParams, useRouteError } from "@remix-run/react";
import { requireUser } from "~/lib/auth.server";
import { listProjects } from "~/lib/db.server";
import { projectListQuerySchema } from "~/lib/validators/project";
import { formatZodError } from "~/lib/zod-error";
import { ProjectCard } from "~/components/project/project-card";

export const meta: MetaFunction = () => [
  { title: "Projects — Styde Forge" },
];

export async function loader({ request }: LoaderFunctionArgs) {
  const user = await requireUser(request);

  // Extract and validate query parameters
  const url = new URL(request.url);
  const rawQuery = Object.fromEntries(url.searchParams);

  const parsed = projectListQuerySchema.safeParse(rawQuery);
  if (!parsed.success) {
    // Return errors to the client instead of throwing
    return {
      user,
      projects: [],
      pagination: { page: 1, perPage: 20, total: 0, totalPages: 0 },
      errors: formatZodError(parsed.error),
    };
  }

  const { page, perPage, status, search, sort, order } = parsed.data;

  const result = await listProjects({
    userId: user.id,
    page,
    perPage,
    status,
    search,
    sort,
    order,
  });

  return {
    user,
    projects: result.projects,
    pagination: {
      page: result.page,
      perPage: result.perPage,
      total: result.total,
      totalPages: Math.ceil(result.total / result.perPage),
    },
    filters: { status, search, sort, order },
    errors: null,
  };
}

export default function ProjectListPage() {
  const { projects, pagination, filters, errors } = useLoaderData<typeof loader>();
  const [searchParams, setSearchParams] = useSearchParams();

  const updateFilter = (key: string, value: string) => {
    const next = new URLSearchParams(searchParams);
    if (value) {
      next.set(key, value);
    } else {
      next.delete(key);
    }
    next.set("page", "1"); // reset page on filter change
    setSearchParams(next, { preventScrollReset: false });
  };

  return (
    <div>
      {errors && (
        <div className="mb-4 p-3 bg-yellow-50 border border-yellow-200 rounded text-sm text-yellow-800">
          Invalid filters applied. Using defaults.
        </div>
      )}

      {/* Filters */}
      <div className="flex gap-4 mb-6">
        <input
          type="search"
          placeholder="Search projects..."
          defaultValue={filters?.search ?? ""}
          onChange={(e) => {
            // Debounce in real app
            updateFilter("search", e.target.value);
          }}
          className="flex-1 rounded-lg border border-gray-300 px-3 py-2 text-sm"
        />
        <select
          defaultValue={filters?.status ?? ""}
          onChange={(e) => updateFilter("status", e.target.value)}
          className="rounded-lg border border-gray-300 px-3 py-2 text-sm"
        >
          <option value="">All Status</option>
          <option value="active">Active</option>
          <option value="draft">Draft</option>
          <option value="archived">Archived</option>
        </select>
        <select
          defaultValue={filters?.sort ?? "updatedAt"}
          onChange={(e) => updateFilter("sort", e.target.value)}
          className="rounded-lg border border-gray-300 px-3 py-2 text-sm"
        >
          <option value="updatedAt">Last Updated</option>
          <option value="createdAt">Created</option>
          <option value="name">Name</option>
        </select>
      </div>

      {/* List */}
      {projects.length === 0 ? (
        <div className="text-center py-12 text-gray-500">
          <p className="text-lg">No projects found</p>
          <a
            href="/app/projects/new"
            className="mt-4 inline-block text-indigo-600 hover:underline"
          >
            Create your first project
          </a>
        </div>
      ) : (
        <div className="grid gap-4 sm:grid-cols-2 lg:grid-cols-3">
          {projects.map((project) => (
            <ProjectCard key={project.id} project={project} />
          ))}
        </div>
      )}

      {/* Pagination */}
      {pagination.totalPages > 1 && (
        <div className="mt-6 flex justify-center gap-2">
          {Array.from({ length: pagination.totalPages }, (_, i) => i + 1).map((p) => (
            <button
              key={p}
              onClick={() => updateFilter("page", String(p))}
              className={`px-3 py-1 rounded text-sm ${
                p === pagination.page
                  ? "bg-indigo-600 text-white"
                  : "bg-white border border-gray-300 hover:bg-gray-50"
              }`}
            >
              {p}
            </button>
          ))}
        </div>
      )}
    </div>
  );
}

export function ErrorBoundary() {
  const error = useRouteError();

  return (
    <div className="text-center py-12">
      <h2 className="text-xl font-semibold text-red-800">Failed to Load Projects</h2>
      <p className="mt-2 text-gray-600">
        {error instanceof Error ? error.message : "An unexpected error occurred."}
```

## 5. Progressive Enhancement Forms

Progressive enhancement in Remix means **forms work without JavaScript**. When JS loads, Remix enhances them with client-side transitions, pending UI, and optimistic updates.

### 5.1 Principles

1. **`<Form>` (not `<form>`)** — Remix enhances `<Form>` with `fetch` + pending states when JS is present. Without JS, it degrades to a standard POST.
2. **`noValidate` on forms** — Disable browser validation so our server-side Zod errors render consistently. Use `aria-invalid` + `aria-describedby` for accessibility.
3. **`useNavigation().state === "submitting"`** — Disable submit buttons and show spinners. Works on both enhanced and native submissions.
4. **`useActionData()`** — Returns server validation errors. Without JS, the page reloads and re-renders with errors. With JS, errors appear without a full page reload.
5. **`useNavigation().formData`** — Access the submitted form data during submission to maintain filled-in values optimistically.

## 6. Error Boundaries Per Route

Every route in this architecture exports its own `ErrorBoundary`. The error boundary closest to the error catches it — errors "bubble up" to parent boundaries if the child doesn't have one.

### 6.1 Error Boundary Strategy

| Route | Catches | Behavior |
|---|---|---|
| `root.tsx` | Catastrophic errors, unmatched 404s | Full-page error with status code |
| `__app.tsx` | Auth errors, app-level failures | In-layout error with "Return to Dashboard" |
| `__app.projects.tsx` | Project listing failures | In-section error with retry |
| `__app.projects.$id.tsx` | Project not found (404), access denied (403) | Specific "not found" UI |
| `__app.projects.new.tsx` | Form rendering failures | Graceful fallback |
| `$.tsx` | Unmatched routes (404) | Friendly 404 page |

## 7. Optimistic UI with useFetcher

`useFetcher` is the key to optimistic UI in Remix. Unlike `<Form>`, it doesn't cause navigation — perfect for inline toggles, star buttons, and small mutations.

### 7.1 Star/Unstar Button (Optimistic Toggle)

```tsx
// app/components/project/star-button.tsx
import { useFetcher } from "@remix-run/react";
import { useEffect, useRef } from "react";

interface StarButtonProps {
  projectId: number;
  isStarred: boolean;
}

export function StarButton({ projectId, isStarred }: StarButtonProps) {
  const fetcher = useFetcher();
  const prevIsStarred = useRef(isStarred);

  // ── Determine the displayed state ────────────────────────────
  // If the fetcher is submitting with form data, optimistically
  // use the target state from the form. Otherwise, use the prop.
  const formData = fetcher.formData;
  const optimisticStarred =
    fetcher.state === "submitting" && formData
      ? formData.get("starred") === "true"
      : isStarred;

  // ── Sync on completion ───────────────────────────────────────
  useEffect(() => {
    if (fetcher.state === "idle" && fetcher.data?.success) {
      prevIsStarred.current = optimisticStarred;
    }
  }, [fetcher.state, fetcher.data, optimisticStarred]);

  // ── Handle error reversion ───────────────────────────────────
  // If the action returned an error, the prop will be stale;
  // we rely on the next re-render from the parent to correct it.
  const hasError = fetcher.data && !fetcher.data.success;

  return (
    <fetcher.Form method="post" action={`/api/projects/${projectId}/star`}>
      <input type="hidden" name="projectId" value={projectId} />
      <input type="hidden" name="starred" value={String(!isStarred)} />

      <button
        type="submit"
        disabled={fetcher.state === "submitting"}
        className={`inline-flex items-center gap-1.5 px-3 py-1.5 rounded-lg text-sm font-medium transition-all ${
          optimisticStarred
            ? "bg-yellow-50 text-yellow-700 border border-yellow-300 hover:bg-yellow-100"
            : "bg-white text-gray-500 border border-gray-300 hover:bg-gray-50 hover:text-gray-700"
        } ${hasError ? "border-red-300" : ""}`}
        title={optimisticStarred ? "Unstar this project" : "Star this project"}
      >
        <StarIcon filled={optimisticStarred} />
        {optimisticStarred ? "Starred" : "Star"}
        {hasError && <span className="text-red-500 text-xs">(failed)</span>}
      </button>
    </fetcher.Form>
  );
}

function StarIcon({ filled }: { filled: boolean }) {
  return (
    <svg
      className={`h-4 w-4 ${filled ? "text-yellow-500" : "text-gray-300"}`}
      fill={filled ? "currentColor" : "none"}
      viewBox="0 0 24 24"
      stroke="currentColor"
      strokeWidth={1.5}
    >
      <path
```

```typescript
export async function listProjects(filters: {
  userId: number;
  page: number;
  perPage: number;
  status?: string;
  search?: string;
  sort: string;
  order: string;
}) {
  const where: any = { ownerId: filters.userId };

  if (filters.status) where.status = filters.status;
  if (filters.search) {
    where.OR = [
      { name: { contains: filters.search, mode: "insensitive" } },
      { description: { contains: filters.search, mode: "insensitive" } },
    ];
  }

  const [projects, total] = await Promise.all([
    db.project.findMany({
      where,
      orderBy: { [filters.sort]: filters.order },
      skip: (filters.page - 1) * filters.perPage,
      take: filters.perPage,
    }),
    db.project.count({ where }),
  ]);

  return { projects, total, page: filters.page, perPage: filters.perPage };
}
```

```typescript
// app/lib/auth.server.ts
import { redirect } from "@remix-run/node";
import { getUser, sessionStorage } from "./session.server";
import bcrypt from "bcryptjs";

export async function requireUser(request: Request) {
  const user = await getUser(request);
  if (!user) {
    const url = new URL(request.url);
    const redirectTo = url.pathname + url.search;
    throw redirect(`/login?redirectTo=${encodeURIComponent(redirectTo)}`);
  }
  return user;
}

export async function verifyLogin(email: string, password: string) {
  const { db } = await import("./db.server");
  const user = await db.user.findUnique({ where: { email } });
  if (!user) return null;

  const valid = await bcrypt.compare(password, user.passwordHash);
  return valid ? user : null;
}

export async function createUserSession(userId: number, redirectTo: string) {
  const session = await sessionStorage.getSession();
  session.set("userId", userId);

  return redirect(redirectTo, {
    headers: { "Set-Cookie": await sessionStorage.commitSession(session) },
  });
}
```

```
┌──────────────┐
```

### 9.2 Form Submission Data Flow (Action — Progressive Enhancement)

```
┌──────────────────────────────────────────────┐
│  WITHOUT JavaScript (HTML form POST)          │
│                                               │
│  <form method="post" action="/app/projects">  │
│    User clicks Submit                         │
│    → Browser POSTs form data                  │
│    → Server processes action                  │
│    → Server returns new page (redirect/errors) │
│    → Full page reload with errors rendered    │
└──────────────────────────────────────────────┘

┌──────────────────────────────────────────────┐
│  WITH JavaScript (Remix-enhanced)             │
│                                               │
│  <Form method="post">                         │
│    User clicks Submit                         │
│    → Remix intercepts, uses fetch()           │
│    → navigation.state = "submitting"          │
│    → Button disabled + spinner shown          │
│    → Server processes action                  │
│    → If errors: useActionData() updates       │
│      → Errors appear without page reload      │
│    → If success: redirect happens via fetch   │
│    → navigation.state = "idle"                │
└──────────────────────────────────────────────┘
```

### 9.3 Optimistic UI Data Flow (useFetcher)

```
┌───────────────────────────────────────────────────────┐
│  StarButton Component                                 │
│                                                       │
│  1. User clicks star button                           │
│  2. fetcher.Form submits POST to /api/projects/:id/star│
│  3. Before response:                                   │
│     fetcher.state = "submitting"                      │
│     fetcher.formData available → read target state     │
│     Button renders as "Starred" immediately ✨         │
│  4. Server processes:                                  │
│     Zod validates, DB updates, returns { success }     │
│  5. Response received:                                 │
│     If success: state = "idle", revalidation triggers  │
│     If error: state = "idle", show error indicator    │
│  6. Parent re-fetches loader → fresh isStarred prop   │
└───────────────────────────────────────────────────────┘
```
