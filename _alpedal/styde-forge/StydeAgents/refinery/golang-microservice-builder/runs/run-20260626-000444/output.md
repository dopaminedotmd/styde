package server
import (
    "context"
    "fmt"
    "log"
    "net"
    "google.golang.org/grpc"
    greetv1 "github.com/user/go-micro/api/proto/v1"
    "github.com/user/go-micro/internal/service"
)
type Server struct {
    greeter *service.Greeter
    grpc    *grpc.Server
    lis     net.Listener
}
func New(greeter *service.Greeter) *Server {
    return &Server{greeter: greeter}
}
func (s *Server) Start(ctx context.Context, addr string) error {
    lis, err := net.Listen("tcp", addr)
    if err != nil {
        return fmt.Errorf("listen: %w", err)
    }
    s.lis = lis
    s.grpc = grpc.NewServer()
    greetv1.RegisterGreeterServiceServer(s.grpc, s.greeter)
    errCh := make(chan error, 1)
    go func() {
        log.Printf("gRPC server listening on %s", addr)
        if err := s.grpc.Serve(lis); err != nil {
            errCh <- fmt.Errorf("serve: %w", err)
        }
    }()
    select {
    case <-ctx.Done():
        s.grpc.GracefulStop()
        return ctx.Err()
    case err := <-errCh:
        return err
    default:
        return nil
    }
}
func (s *Server) Shutdown() {
    if s.grpc != nil {
        log.Println("shutting down gRPC server...")
        s.grpc.GracefulStop()
    }
}

package service
import (
    "context"
    "fmt"
    greetv1 "github.com/user/go-micro/api/proto/v1"
)
type Greeter struct {
    greetv1.UnimplementedGreeterServiceServer
    greetCh chan string
}
func NewGreeter() *Greeter {
    g := &Greeter{
        greetCh: make(chan string, 100),
    }
    go g.processGreetings()
    return g
}
func (g *Greeter) Greet(_ context.Context, req *greetv1.GreetRequest) (*greetv1.GreetResponse, error) {
    msg := fmt.Sprintf("Hello, %s!", req.GetName())
    select {
    case g.greetCh <- msg:
    default:
    }
    return &greetv1.GreetResponse{Message: msg}, nil
}
func (g *Greeter) GreetStream(req *greetv1.GreetRequest, stream greetv1.GreeterService_GreetStreamServer) error {
    for i := 1; i <= 5; i++ {
        msg := fmt.Sprintf("Hello #%d, %s!", i, req.GetName())
        if err := stream.Send(&greetv1.GreetResponse{Message: msg}); err != nil {
            return err
        }
    }
    return nil
}
func (g *Greeter) processGreetings() {
    for msg := range g.greetCh {
        fmt.Printf("[goroutine] processed greeting: %s\n", msg)
    }
}

package main
import (
    "context"
    "log"
    "os"
    "os/signal"
    "syscall"
    "github.com/user/go-micro/internal/wire"
)
func main() {
    ctx, cancel := signal.NotifyContext(context.Background(), syscall.SIGINT, syscall.SIGTERM)
    defer cancel()
    srv, err := wire.InitializeServer()
    if err != nil {
        log.Fatalf("init: %v", err)
    }
    addr := ":50051"
    if v := os.Getenv("GRPC_ADDR"); v != "" {
        addr = v
    }
    if err := srv.Start(ctx, addr); err != nil {
        log.Fatalf("start: %v", err)
    }
    <-ctx.Done()
    srv.Shutdown()
}
