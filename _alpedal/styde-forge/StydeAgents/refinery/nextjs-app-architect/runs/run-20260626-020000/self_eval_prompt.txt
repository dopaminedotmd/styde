## 1. Partial Prerendering (PPR) — Hybrid Static + Dynamic

### Concept

PPR allows a single route to combine a static shell (rendered at build time) with dynamic holes (streamed at request time). The static shell arrives instantly; dynamic islands stream in as they become ready, wrapped in Suspense boundaries.

### Configuration

```typescript
// ============================================================
// next.config.ts — Enable PPR and configure the experimental flag
// ============================================================

import type { NextConfig } from 'next';

const nextConfig: NextConfig = {
  experimental: {
    ppr: 'incremental', // 'incremental' for opt-in per-route; 'true' for all routes
  },
  // Optional: configure dynamicIO for streaming behavior
  // experimental: { dynamicIO: true },
};

export default nextConfig;
```

### Product Detail Page — Static Shell + Dynamic Island

```typescript
// ============================================================
// app/products/[id]/page.tsx
// PPR: Static product shell prerendered at build time.
// The reviews section is a dynamic hole — it streams in at request time.
// ============================================================

import { Suspense } from 'react';
import { unstable_noStore as noStore } from 'next/cache';

// ---------------------------------------------------------------------------
// Static Shell — prerendered at build time for every known product
// ---------------------------------------------------------------------------
export async function generateStaticParams() {
  const products = await fetch('https://api.example.com/products', {
    next: { revalidate: 3600 },
  }).then((r) => r.json());

  return products.map((p: { id: string }) => ({ id: p.id }));
}

interface ProductPageProps {
  params: Promise<{ id: string }>;
}

// The page itself is the static shell — it's prerendered at build time.
// generateStaticParams produces the paths; ISR revalidates stale data.
export default async function ProductPage({ params }: ProductPageProps) {
  const { id } = await params;

  // Static data — fetched at build time, cached
  const product = await fetch(`https://api.example.com/products/${id}`, {
    next: { tags: [`product-${id}`] },
  }).then((r) => r.json());

  return (
    <article className="product-page">
      {/* Static Shell — arrives instantly */}
      <header>
        <h1>{product.name}</h1>
        <p className="price">${product.price.toFixed(2)}</p>
      </header>

      <section className="product-description">
        <p>{product.description}</p>
      </section>

      <section className="product-gallery">
        {product.images.map((img: string, i: number) => (
          <img key={i} src={img} alt={`${product.name} - view ${i + 1}`} />
        ))}
      </section>

      {/* ── Dynamic Island (PPR hole) ── */}
      {/* This Suspense boundary creates the dynamic "hole" in the static shell.
          The shell renders immediately; the reviews stream in when ready. */}
      <Suspense
        fallback={
          <div className="reviews-skeleton" aria-busy="true">
            {Array.from({ length: 3 }).map((_, i) => (
              <div key={i} className="review-card-skeleton">
                <div className="skeleton-line w-32" />
                <div className="skeleton-line w-full" />
                <div className="skeleton-line w-48" />
              </div>
            ))}
          </div>
        }
      >
        <ProductReviews productId={id} />
      </Suspense>

      {/* Multiple dynamic islands can coexist, each streaming independently */}
      <Suspense fallback={<RelatedProductsSkeleton />}>
        <RelatedProducts productId={id} />
      </Suspense>
    </article>
  );
}

// ---------------------------------------------------------------------------
// Dynamic Island — fetched at request time, streamed when ready
// ---------------------------------------------------------------------------
async function ProductReviews({ productId }: { productId: string }) {
  // unstable_noStore opts this subtree out of static prerendering,
  // making it a true dynamic "hole" in the PPR shell.
  noStore();

  const reviews = await fetch(
    `https://api.example.com/products/${productId}/reviews`,
    { next: { revalidate: 60 } } // Stale-while-revalidate for 60s
  ).then((r) => r.json());

  if (reviews.length === 0) {
    return (
      <div className="empty-state">
        <p>No reviews yet. Be the first!</p>
      </div>
    );
  }

  return (
    <div className="reviews-section">
      <h2>Customer Reviews ({reviews.length})</h2>
      <ul className="reviews-list">
        {reviews.map((review: Review) => (
          <li key={review.id} className="review-card">
            <div className="review-header">
              <span className="review-author">{review.author}</span>
              <RatingStars rating={review.rating} />
              <time dateTime={review.createdAt}>
                {new Date(review.createdAt).toLocaleDateString()}
              </time>
            </div>
            <p>{review.body}</p>
          </li>
        ))}
      </ul>
    </div>
  );
}

// ---------------------------------------------------------------------------
// Second dynamic island — Related Products
// ---------------------------------------------------------------------------
async function RelatedProducts({ productId }: { productId: string }) {
  noStore();

  const products = await fetch(
    `https://api.example.com/products/${productId}/related`,
    { next: { revalidate: 300 } }
  ).then((r) => r.json());

  return (
    <aside className="related-products">
      <h2>You Might Also Like</h2>
      <div className="product-grid">
        {products.map((p: ProductCard) => (
          <ProductCard key={p.id} product={p} />

      {'☆'.repeat(5 - Math.round(rating))}

          <div key={i} className="skeleton-card">
            <div className="skeleton-square" />
            <div className="skeleton-line w-32" />

      </div>

  id: string;
  name: string;

  thumbnail: string;
```

### PPR Decision Matrix

| Signal / Pattern | Result |
|---|---|
| `export const experimental_ppr = true` | Route opts into PPR |
| `generateStaticParams` + `Suspense` | Static shell + dynamic islands |
| `unstable_noStore()` in a Suspense boundary | That boundary becomes a dynamic hole |
| No `unstable_noStore()` — data fetched with `next: { revalidate }` | ISR cache controls freshness; still prerendered |
| No Suspense, no `unstable_noStore()` | Fully static at build time |
| `export const dynamic = 'force-dynamic'` | Opts entire route out of PPR |

## 2. Server Actions with Optimistic UI

### Concept

Server Actions provide a type-safe bridge from the client to server-side mutations. Optimistic UI updates the UI instantly before the server confirms, rolling back on failure. The pattern below demonstrates a task management dashboard with optimistic add/update/delete.

### Server Actions Module

```typescript
// ============================================================
// app/dashboard/tasks/actions.ts
// Server Actions for task CRUD. These run on the server but
// are callable from Client Components via `useActionState` or
// direct invocation with `useOptimistic`.
// ============================================================

'use server';

import { revalidatePath } from 'next/cache';
import { redirect } from 'next/navigation';
import { z } from 'zod';
import { auth } from '@/lib/auth';
import { db } from '@/lib/db';
import type { Task } from '@/types';

// ---------------------------------------------------------------------------
// Validation Schema
// ---------------------------------------------------------------------------
const TaskSchema = z.object({
  title: z.string().min(1, 'Title is required').max(200),
  description: z.string().max(2000).optional(),
  priority: z.enum(['low', 'medium', 'high', 'critical']),
  dueDate: z.string().datetime().optional(),
  assigneeId: z.string().uuid().optional(),
});

const UpdateTaskSchema = TaskSchema.partial().extend({
  completed: z.boolean().optional(),
});

// ---------------------------------------------------------------------------
// Create Task
// ---------------------------------------------------------------------------
interface CreateTaskState {
  error: string | null;
  success: boolean;
  taskId?: string;
}

export async function createTask(
  prevState: CreateTaskState,
  formData: FormData
): Promise<CreateTaskState> {
  const session = await auth();
  if (!session?.user) {
    return { error: 'Unauthorized', success: false };
  }

  const raw = {
    title: formData.get('title'),
    description: formData.get('description') || undefined,
    priority: formData.get('priority'),
    dueDate: formData.get('dueDate') || undefined,
    assigneeId: formData.get('assigneeId') || undefined,
  };

  const parsed = TaskSchema.safeParse(raw);
  if (!parsed.success) {
    const messages = parsed.error.issues.map((i) => i.message).join(', ');
    return { error: `Validation failed: ${messages}`, success: false };
  }

  try {
    const task = await db.task.create({
      data: {
        ...parsed.data,
        createdById: session.user.id,
      },
    });

    revalidatePath('/dashboard/tasks');
    // Revalidate the parallel analytics slot too
    revalidatePath('/dashboard', 'layout');

    return { error: null, success: true, taskId: task.id };
  } catch (err) {
    console.error('createTask failed:', err);
    return { error: 'Failed to create task. Please try again.', success: false };
  }
}

// ---------------------------------------------------------------------------
// Update Task (toggle complete, edit fields)
// ---------------------------------------------------------------------------
interface UpdateTaskState {
  error: string | null;
  success: boolean;
}

export async function updateTask(
  taskId: string,
  prevState: UpdateTaskState,
  formData: FormData
): Promise<UpdateTaskState> {
  const session = await auth();
  if (!session?.user) {
    return { error: 'Unauthorized', success: false };
  }

  const raw: Record<string, unknown> = {};
  const completed = formData.get('completed');
  if (completed !== null) raw.completed = completed === 'true';
  const title = formData.get('title');
  if (title) raw.title = title;
  const priority = formData.get('priority');
  if (priority) raw.priority = priority;

  const parsed = UpdateTaskSchema.safeParse(raw);
  if (!parsed.success) {
    return {
      error: parsed.error.issues.map((i) => i.message).join(', '),
      success: false,
    };
  }

  try {
    await db.task.update({
      where: { id: taskId },
      data: parsed.data,
    });

    revalidatePath('/dashboard/tasks');
    return { error: null, success: true };
  } catch (err) {
    console.error('updateTask failed:', err);
    return { error: 'Update failed. Please try again.', success: false };
  }
}

// ---------------------------------------------------------------------------
// Delete Task
// ---------------------------------------------------------------------------
export async function deleteTask(taskId: string): Promise<{ error: string | null }> {
  const session = await auth();
  if (!session?.user) {
    return { error: 'Unauthorized' };
  }

  try {
    await db.task.delete({ where: { id: taskId } });
    revalidatePath('/dashboard/tasks');
    return { error: null };
  } catch (err) {
    console.error('deleteTask failed:', err);
    return { error: 'Failed to delete task.' };
  }
}

// ---------------------------------------------------------------------------
// Optimistic Toggle — a specialized action for instant completion toggle
// This returns the new state so the client doesn't need to re-fetch.
// ---------------------------------------------------------------------------
export async function toggleTaskComplete(
  taskId: string,
  currentState: boolean
): Promise<{ error: string | null; completed: boolean }> {
  const session = await auth();
  if (!session?.user) {
    return { error: 'Unauthorized', completed: currentState };
  }

  const newState = !currentState;

  try {
    await db.task.update({
      where: { id: taskId },
      data: { completed: newState },
    });

    revalidatePath('/dashboard/tasks');
    return { error: null, completed: newState };
  } catch (err) {
    console.error('toggleTaskComplete failed:', err);
    return { error: 'Toggle failed', completed: currentState };
  }
}
```

### Client Component with Optimistic UI

```typescript
// ============================================================
// app/dashboard/tasks/TaskBoard.tsx
// 'use client' component that wires Server Actions to an
// optimistic UI using React's useOptimistic hook.
// ============================================================

'use client';

import {
  startTransition,
  useOptimistic,
  useActionState,
  useRef,
  useState,
} from 'react';
import { createTask, deleteTask, toggleTaskComplete } from './actions';
import type { Task } from '@/types';

// ---------------------------------------------------------------------------
// TaskBoard — the top-level client component
// ---------------------------------------------------------------------------
export function TaskBoard({ tasks: initialTasks }: { tasks: Task[] }) {
  // ── Optimistic state ──
  // useOptimistic keeps a local copy that mirrors the server state.
  // When a mutation fires, we apply an optimistic updater immediately.
  // If the server action fails, React rolls back to the previous state.
  const [optimisticTasks, addOptimisticTask] = useOptimistic<
    Task[],
    OptimisticAction
  >(initialTasks, (state, action) => {
    switch (action.type) {
      case 'create': {
        const optimisticTask: Task = {
          id: `optimistic-${Date.now()}`,
          title: action.title,
          description: null,
          priority: 'medium',
          completed: false,
          dueDate: null,
          createdAt: new Date().toISOString(),
          updatedAt: new Date().toISOString(),
          createdById: '',
          assigneeId: null,
        };
        return [optimisticTask, ...state];
      }

      case 'delete':
        return state.filter((t) => t.id !== action.taskId);

      case 'toggle': {
        return state.map((t) =>
          t.id === action.taskId
            ? { ...t, completed: !t.completed }
            : t
        );
      }

      default:
        return state;
    }
  });

  // ── Server Action + useActionState for form submissions ──
  const [createState, createAction, isCreating] = useActionState(
    async (_prev: { error: string | null; success: boolean }, formData: FormData) => {
      // Apply optimistic update BEFORE the server call
      const title = formData.get('title') as string;
      addOptimisticTask({ type: 'create', title });
      return await createTask({ error: null, success: false }, formData);
    },
    { error: null, success: false }
  );

  const formRef = useRef<HTMLFormElement>(null);
  const [deleteErrors, setDeleteErrors] = useState<Record<string, string>>({});

  // ── Handlers ──
  const handleDelete = async (taskId: string) => {
    startTransition(() => {
      addOptimisticTask({ type: 'delete', taskId });
    });

    const result = await deleteTask(taskId);
    if (result.error) {
      setDeleteErrors((prev) => ({ ...prev, [taskId]: result.error! }));
    }
  };

  const handleToggle = async (taskId: string, completed: boolean) => {
    startTransition(() => {
      addOptimisticTask({ type: 'toggle', taskId });
    });

    const result = await toggleTaskComplete(taskId, completed);
    if (result.error) {
      setDeleteErrors((prev) => ({ ...prev, [taskId]: result.error! }));
    }
  };

  return (
    <div className="task-board">
      {/* ── Create Form ── */}
      <form
        ref={formRef}
        action={createAction}
        className="task-create-form"
      >
        <input
          type="text"
          name="title"
          placeholder="What needs to be done?"
          required
          maxLength={200}
          aria-label="Task title"
        />
        <select name="priority" defaultValue="medium">
          <option value="low">Low</option>
          <option value="medium">Medium</option>
          <option value="high">High</option>
          <option value="critical">Critical</option>
        </select>
        <button type="submit" disabled={isCreating}>
          {isCreating ? 'Adding…' : 'Add Task'}
        </button>

        {createState.error && (
          <p className="error-message" role="alert">
            {createState.error}
          </p>
        )}
      </form>

      {/* ── Task List (renders optimistic state instantly) ── */}
      <ul className="task-list">
        {optimisticTasks.map((task) => (
          <TaskItem
            key={task.id}
            task={task}
            isOptimistic={task.id.startsWith('optimistic-')}
            onDelete={handleDelete}
            onToggle={handleToggle}
            error={deleteErrors[task.id]}
          />
        ))}

        {optimisticTasks.length === 0 && (
          <p className="empty-state">No tasks yet. Add one above!</p>
        )}
      </ul>
    </div>
  );
}

// ---------------------------------------------------------------------------
// TaskItem — individual task row
// ---------------------------------------------------------------------------
function TaskItem({
  task,
  isOptimistic,
  onDelete,
  onToggle,
  error,
}: {
  task: Task;
  isOptimistic: boolean;
  onDelete: (id: string) => void;
  onToggle: (id: string, completed: boolean) => void;
  error?: string;
}) {
  return (
    <li
      className={`task-item ${isOptimistic ? 'optimistic' : ''} ${
        task.completed ? 'completed' : ''
      }`}
      aria-busy={isOptimistic}
    >
      <label className="task-checkbox-label">
        <input
          type="checkbox"
          checked={task.completed}
          onChange={() => onToggle(task.id, task.completed)}
          disabled={isOptimistic}
        />
        <span className="task-title">{task.title}</span>
      </label>

      <span className={`task-priority priority-${task.priority}`}>
        {task.priority}
      </span>

      <button
        className="task-delete-btn"
        onClick={() => onDelete(task.id)}
        disabled={isOptimistic}
        aria-label={`Delete task: ${task.title}`}
      >
        🗑️
      </button>

      {error && (
        <p className="error-message" role="alert">
          {error}
        </p>
      )}
    </li>
  );
}

// ---------------------------------------------------------------------------
// Types
// ---------------------------------------------------------------------------
type OptimisticAction =
  | { type: 'create'; title: string }
  | { type: 'delete'; taskId: string }
  | { type: 'toggle'; taskId: string };
```

### Optimistic UI Best Practices

| Concern | Solution |
|---|---|
| Rollback on failure | `useOptimistic` automatically restores previous state if the transition errors |
| Idempotency keys | Use deterministic temporary IDs (e.g., `optimistic-{timestamp}`) for creates |
| Error surfacing | Track per-item error state; show inline errors, not global toasts alone |
| Concurrent mutations | `startTransition` batches optimistic updates; React handles interleaving |
| Loading indicators | Use the `isOptimistic` flag to show a spinner or muted style |
| Revalidation | Call `revalidatePath` or `revalidateTag` in the Server Action to refresh server data |

## 3. Parallel Routes for Dashboards

### Concept

Parallel Routes let multiple page segments render in the same layout simultaneously, each in its own named slot. This is perfect for dashboards where analytics, tasks, activity feed, and notifications are independent sub-views that load at different speeds.

### Route Structure

```
app/
└── dashboard/
    ├── layout.tsx          # Orchestrates slots: @analytics, @tasks, @activity
    ├── page.tsx            # Fallback when no slot is active (or a welcome page)
    ├── @analytics/
    │   ├── page.tsx        # Default analytics view
    │   ├── revenue/
    │   │   └── page.tsx    # Deep-dive revenue analytics
    │   └── loading.tsx     # Slot-specific loading state
    ├── @tasks/
    │   ├── page.tsx        # Default task board
    │   ├── [taskId]/
    │   │   └── page.tsx    # Task detail (intercepted by modal)
    │   └── loading.tsx
    └── @activity/
        ├── page.tsx        # Default activity feed
        └── loading.tsx
```

```typescript
function formatRelativeTime(iso: string): string {
  const diff = Date.now() - new Date(iso).getTime();

  if (mins < 1) return 'Just now';
  if (mins < 60) return `${mins}m ago`;
```

## 4. The `after()` Hook — Deferred Post-Response Work

### Concept

`after()` schedules a callback to run *after* the HTTP response has been sent to the client. This is ideal for non-blocking side effects — logging, analytics events, cache warming, email notifications — that shouldn't delay the user's response.

### Full Implementation

```typescript
// ============================================================
// lib/after-utils.ts
// Wrappers around the after() hook for common post-response
// patterns: logging, analytics, notifications, cache warming.
// ============================================================

import { after } from 'next/server';
import type { Session } from '@/lib/auth';

// ---------------------------------------------------------------------------
// Audit Logger — write audit trail entries after responding
// ---------------------------------------------------------------------------
export function auditLog(event: AuditEvent) {
  after(async () => {
    try {
      await fetch(`${process.env.INTERNAL_API}/audit`, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'X-Internal-Token': process.env.INTERNAL_API_TOKEN!,
        },
        body: JSON.stringify({
          ...event,
          timestamp: new Date().toISOString(),
        }),
      });
    } catch (err) {
      // Log to a durable fallback — must not throw inside after()
      console.error('[after.auditLog] Failed to persist audit event:', err);
    }
  });
}

// ---------------------------------------------------------------------------

export function trackAnalytics(event: AnalyticsEvent) {

// ---------------------------------------------------------------------------
// Cache Warmer — pre-warm related pages after a mutation
// ---------------------------------------------------------------------------
export function warmCache(urls: string[]) {
  after(async () => {
    const base = process.env.NEXT_PUBLIC_APP_URL || 'http://localhost:3000';
    const results = await Promise.allSettled(
      urls.map((path) =>
        fetch(`${base}${path}`, {
          headers: { 'X-Cache-Warm': '1' },
        }).catch(() => null) // Silently ignore failures
      )
    );

    const failures = results.filter((r) => r.status === 'rejected').length;
    if (failures > 0) {
      console.warn(`[after.warmCache] ${failures}/${urls.length} cache warm requests failed`);
    }
  });
}

// ---------------------------------------------------------------------------
// Webhook Dispatcher — notify external systems after mutations
// ---------------------------------------------------------------------------
export function dispatchWebhook(webhook: WebhookPayload) {
  after(async () => {
    try {
      const controller = new AbortController();
      const timeout = setTimeout(() => controller.abort(), 10_000); // 10s timeout

      await fetch(webhook.url, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'X-Webhook-Signature': computeHmac(webhook.payload, process.env.WEBHOOK_SECRET!),
        },
        body: JSON.stringify(webhook.payload),
        signal: controller.signal,
      });

      clearTimeout(timeout);
    } catch (err) {
      console.error(`[after.dispatchWebhook] Failed for ${webhook.url}:`, err);

      // Queue for retry in a background job system
      await fetch(`${process.env.INTERNAL_API}/webhook-retries`, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'X-Internal-Token': process.env.INTERNAL_API_TOKEN!,
        },
        body: JSON.stringify({
          url: webhook.url,
          payload: webhook.payload,
          failedAt: new Date().toISOString(),
        }),
      }).catch(() => {});
    }
  });
}
```

### `after()` Constraints & Best Practices

| Rule | Detail |
|---|---|
| **No response modification** | `after()` callbacks run *after* the response is sent; you cannot set cookies, redirect, or modify headers |
| **No throwing** | Errors inside `after()` will be silently swallowed by Next.js — always wrap in try/catch and log |
| **No long-running work** | The callback runs within the request lifecycle's extended scope; use a queue for heavy work |
| **Request context available** | The callback inherits the request's `AsyncLocalStorage` context (cookies, headers, etc.) |
| **Multiple calls coalesce** | Multiple `after()` calls in the same request are batched and run sequentially |
| **Server-only** | `after()` cannot be used in Client Components or `'use client'` modules |

## 5. `taintObjectReference` — Security Hardening

### Concept

`taintObjectReference` marks a JavaScript object or its properties as "tainted," preventing it from being accidentally passed to the client. If tainted data reaches a Client Component boundary, Next.js throws a build-time or runtime error. This enforces data-security invariants at the framework level.

### Implementation

```typescript
// ============================================================
// lib/security/taint.ts
// Centralized taint utilities for protecting sensitive server-side
// data from leaking to the client.
// ============================================================

import { experimental_taintObjectReference as taintObjectReference } from 'react';

// ---------------------------------------------------------------------------
// Taint entire sensitive objects
// ---------------------------------------------------------------------------

/** Taint a user session object so it can never be serialized to the client. */
export function protectSession(session: Session): Session {
  taintObjectReference(
    'Do not pass the full session object to the client. ' +
      'Extract only the fields the client needs (id, name, email, role).',
    session
  );
  return session;
}

/** Taint a database connection handle. */
export function protectDbHandle(db: DatabaseConnection): DatabaseConnection {
  taintObjectReference(
    'Database connection handles must never leave the server. ' +
      'Use data-fetching functions to return serializable results.',
    db
  );
  return db;
}

/** Taint API keys and secrets. */
export function protectSecret(value: string, name: string): string {
  taintObjectReference(
    `Secret "${name}" must not be exposed to the client. ` +
      'Use environment variables and server-only code.',
    value
  );
  return value;
}

// ---------------------------------------------------------------------------
// Taint specific properties on an object (leaving the rest safe to pass)
// ---------------------------------------------------------------------------

/** Return a user object safe for the client — only public fields, rest tainted. */
export function sanitizeUserForClient(user: FullUser): PublicUser {
  // Taint sensitive properties in-place so they cannot leak
  taintObjectReference(
    'User password hash must never be sent to the client.',
    user.passwordHash
  );
  taintObjectReference(
    'User 2FA secret must never be sent to the client.',
    user.twoFactorSecret
  );
  taintObjectReference(
    'User security question answers must never be sent to the client.',
    user.securityAnswers
  );
  taintObjectReference(
    'Internal user flags must never be sent to the client.',
    user.internalFlags
  );

  // Return only the safe subset
  return {
    id: user.id,
    name: user.name,
    email: user.email,
    avatarUrl: user.avatarUrl,
    role: user.role,
  };
}

/** Taint payment details but allow order metadata to pass through. */
export function sanitizeOrderForClient(order: FullOrder): PublicOrder {
  taintObjectReference(
    'Payment method details must never be sent to the client.',
    order.paymentMethod
  );
  taintObjectReference(
    'Customer billing address must never be sent to the client.',
    order.billingAddress
  );
  taintObjectReference(
    'Internal cost/revenue data must never be sent to the client.',
    order.internalCost
  );

  return {
    id: order.id,
    status: order.status,
    items: order.items, // Already sanitized product data
    total: order.total,
    createdAt: order.createdAt,
    estimatedDelivery: order.estimatedDelivery,
  };
}
```

### Taint Checklist

| Data | Taint? | Rationale |
|---|---|---|
| Password hashes | ✅ Always | Never, ever serialize |
| Session tokens / JWTs | ✅ Always | Auth boundary violation |
| API keys / secrets | ✅ Always | Could be extracted from bundle |
| Database handles | ✅ Always | Not serializable, dangerous |
| User emails | ⚠️ Conditional | Taint if privacy setting disallows sharing |
| Internal IDs (DB PKs) | ⚠️ Conditional | Prefer opaque slugs in URLs; taint raw IDs |
| Payment info (CC, billing) | ✅ Always | PCI compliance |
| Server-side feature flags | ✅ Always | Prevents client-side enumeration |
| Public user profile | ❌ Safe | Name, avatar, public bio |
