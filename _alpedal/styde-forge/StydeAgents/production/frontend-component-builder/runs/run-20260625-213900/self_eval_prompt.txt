```tsx
import React, {
  useCallback,
  useMemo,
  useState,
  type ChangeEvent,
  type KeyboardEvent,
} from 'react';
import type {
  ColumnFilter,
  DataTableColumn,
  DataTableProps,
  SortConfig,
} from './DataTable.types';
import type { SortDirection } from '../types/common';
import styles from './DataTable.module.css';

// ─── Helper: sort data ──────────────────────────────────────────────

function sortData<T extends Record<string, unknown>>(
  data: T[],
  sort: SortConfig<T> | undefined,
): T[] {
  if (!sort) return data;
  return [...data].sort((a, b) => {
    const aVal = a[sort.key];
    const bVal = b[sort.key];
    if (aVal == null) return 1;
    if (bVal == null) return -1;

    let cmp = 0;
    if (typeof aVal === 'string' && typeof bVal === 'string') {
      cmp = aVal.localeCompare(bVal);
    } else if (typeof aVal === 'number' && typeof bVal === 'number') {
      cmp = aVal - bVal;
    } else {
      cmp = String(aVal).localeCompare(String(bVal));
    }
    return sort.direction === 'asc' ? cmp : -cmp;
  });
}

// ─── Helper: filter data ────────────────────────────────────────────

function filterData<T extends Record<string, unknown>>(
  data: T[],
  filters: ColumnFilter[],
  globalSearch: string,
): T[] {
  let result = data;

  // Apply column filters
  for (const f of filters) {
    if (!f.value) continue;
    const lowerVal = f.value.toLowerCase();
    result = result.filter((row) => {
      const cell = String(row[f.key] ?? '').toLowerCase();
      switch (f.operator) {
        case 'contains': return cell.includes(lowerVal);
        case 'equals': return cell === lowerVal;
        case 'startsWith': return cell.startsWith(lowerVal);
        case 'endsWith': return cell.endsWith(lowerVal);
        case 'gt': return Number(cell) > Number(lowerVal);
        case 'lt': return Number(cell) < Number(lowerVal);
        case 'gte': return Number(cell) >= Number(lowerVal);
        case 'lte': return Number(cell) <= Number(lowerVal);
        case 'isEmpty': return cell.trim() === '';
        case 'isNotEmpty': return cell.trim() !== '';
        default: return true;
      }
    });
  }

  // Apply global search
  if (globalSearch) {
    const lowerGS = globalSearch.toLowerCase();
    result = result.filter((row) =>
      Object.values(row).some((val) =>
        String(val ?? '').toLowerCase().includes(lowerGS),
      ),
    );
  }

  return result;
}

// ─── Helper: paginate ───────────────────────────────────────────────

function paginate<T>(data: T[], page: number, pageSize: number): T[] {
  const start = (page - 1) * pageSize;
  return data.slice(start, start + pageSize);
}

// ─── Sort Arrow Icon ────────────────────────────────────────────────

function SortArrow({ direction }: { direction: SortDirection | null }) {
  return (
    <span className={styles.sortArrow} aria-hidden="true">
      {direction === 'asc' ? ' ▲' : direction === 'desc' ? ' ▼' : ' ⇅'}
    </span>
  );
}

// ─── Skeleton Row ───────────────────────────────────────────────────

function SkeletonRow({ columns }: { columns: number }) {
  return (
    <tr className={styles.skeletonRow}>
      {Array.from({ length: columns }).map((_, i) => (
        <td key={i} className={styles.skeletonCell}>
          <span className={styles.skeletonBar} />
        </td>
      ))}
    </tr>
  );
}

// ─── Main Component ─────────────────────────────────────────────────

export function DataTable<TData extends Record<string, unknown>>({
  // Data
  data,
  columns,
  rowKey = 'id' as keyof TData & string,

  // Loading / Empty
  loading = false,
  loadingRows = 10,
  emptyState,
  errorState,
  error,

  // Sorting
  sortable = true,
  sortConfig: controlledSort,
  defaultSortConfig,
  onSortChange,

  // Filtering
  filterable = true,
  filters: controlledFilters,
  onFiltersChange,

  // Global search
  globalSearch = false,
  globalSearchPlaceholder = 'Search...',
  globalSearchValue: controlledGS,
  onGlobalSearchChange,

  // Pagination
  pagination: enablePagination = true,
  pageSizeOptions = [10, 25, 50, 100],
  pageSize: controlledPageSize,
  defaultPageSize = 25,
  page: controlledPage,
  totalItems,
  onPaginationChange,

  // Server-side
  serverSide = false,
  onServerStateChange,

  // Selection
  selectionMode = 'none',
  selectedKeys: controlledSelected,
  onSelectionChange,

  // Styling
  rowClassName,
  onRowClick,
  striped = false,
  hoverable = true,
  density = 'default',
  stickyHeader = false,
  maxHeight,

  // Base
  className,
  style,
  'data-testid': dataTestId,
  'aria-label': ariaLabel,
}: DataTableProps<TData>) {
  // ── Uncontrolled / controlled state ───────────────────────────────
  const [internalSort, setInternalSort] = useState<SortConfig<TData> | undefined>(defaultSortConfig);
  const [internalFilters, setInternalFilters] = useState<ColumnFilter[]>([]);
  const [internalGS, setInternalGS] = useState('');
  const [internalPage, setInternalPage] = useState(1);
  const [internalPageSize, setInternalPageSize] = useState(defaultPageSize);
  const [internalSelected, setInternalSelected] = useState<string[]>([]);

  const sort = controlledSort ?? internalSort;
  const filters = controlledFilters ?? internalFilters;
  const gs = controlledGS ?? internalGS;
  const page = controlledPage ?? internalPage;
  const pageSize = controlledPageSize ?? internalPageSize;
  const selected = controlledSelected ?? internalSelected;

  const visibleColumns = useMemo(() => columns.filter((c) => !c.hidden), [columns]);

  // ── Data processing (client-side) ─────────────────────────────────
  const processed = useMemo(() => {
    if (serverSide) return data; // server handles everything
    let result = filterData(data, filters, gs);
    result = sortData(result, sort);
    return result;
  }, [data, filters, gs, sort, serverSide]);

  const total = totalItems ?? processed.length;
  const totalPages = Math.max(1, Math.ceil(total / pageSize));
  const paginatedData = useMemo(
    () => (enablePagination ? paginate(processed, page, pageSize) : processed),
    [processed, page, pageSize, enablePagination],
  );

  // ── Helpers for state updates ─────────────────────────────────────
  const updateSort = useCallback(
    (key: keyof TData & string) => {
      const newSort: SortConfig<TData> | undefined =
        sort?.key === key
          ? sort.direction === 'asc'
            ? { key, direction: 'desc' }
            : undefined
          : { key, direction: 'asc' };
      setInternalSort(newSort);
      onSortChange?.(newSort ?? null);
      serverProxy();
    },
    [sort],
  );

  const updateFilters = useCallback(
    (newFilters: ColumnFilter[]) => {
      setInternalFilters(newFilters);
      onFiltersChange?.(newFilters);
      setInternalPage(1);
      serverProxy();
    },
    [],
  );

  const updateGS = useCallback(
    (val: string) => {
      setInternalGS(val);
      onGlobalSearchChange?.(val);
      setInternalPage(1);
      serverProxy();
    },
    [],
  );

  const updatePage = useCallback(
    (p: number) => {
      setInternalPage(p);
      onPaginationChange?.({ page: p, pageSize });
      serverProxy();
    },
    [pageSize],
  );

  const updatePageSize = useCallback(
    (ps: number) => {
      setInternalPageSize(ps);
      setInternalPage(1);
      onPaginationChange?.({ page: 1, pageSize: ps });
      serverProxy();
    },
    [],
  );

  const toggleSelectAll = useCallback(() => {
    const allKeys = paginatedData.map((r) => String(r[rowKey]));
    const newSel = selected.length === allKeys.length ? [] : allKeys;
    setInternalSelected(newSel);
    onSelectionChange?.(newSel);
  }, [paginatedData, rowKey, selected]);

  const toggleSelectRow = useCallback(
    (key: string) => {
      let newSel: string[];
      if (selectionMode === 'single') {
        newSel = selected.includes(key) ? [] : [key];
      } else {
        newSel = selected.includes(key)
          ? selected.filter((k) => k !== key)
          : [...selected, key];
      }
      setInternalSelected(newSel);
      onSelectionChange?.(newSel);
    },
    [selectionMode, selected],
  );

  const serverProxy = useCallback(() => {
    if (serverSide && onServerStateChange) {
      onServerStateChange({
        sort,
        filters,
        pagination: { page, pageSize, totalItems: total, totalPages },
        globalSearch: gs,
      });
    }
  }, [serverSide, sort, filters, page, pageSize, total, totalPages, gs]);

  // ── Render ────────────────────────────────────────────────────────

  if (error) {
    return (
      <div className={styles.errorContainer} data-testid={dataTestId}>
        {errorState ?? (
          <div className={styles.errorDefault}>
            <span>⚠️</span>
            <p>{error.message || 'An error occurred while loading data.'}</p>
          </div>
        )}
      </div>
    );
  }

  const containerClasses = [styles.container, className].filter(Boolean).join(' ');
  const tableClasses = [
    styles.table,
    striped && styles.striped,
    hoverable && styles.hoverable,
    density !== 'default' && styles[`density_${density}`],
    stickyHeader && styles.stickyHeader,
  ]
    .filter(Boolean)
    .join(' ');

  return (
    <div
      className={containerClasses}
      style={{ ...style, maxHeight: maxHeight ?? undefined }}
      data-testid={dataTestId}
      aria-label={ariaLabel ?? 'Data table'}
    >
      {/* ── Global Search ──────────────────────────────────────── */}
      {globalSearch && (
        <div className={styles.globalSearch}>
          <input
            type="text"
            className={styles.searchInput}
            placeholder={globalSearchPlaceholder}
            value={gs}
            onChange={(e) => updateGS(e.target.value)}
            aria-label="Search table"
          />
          {gs && (
            <button
              className={styles.clearSearch}
              onClick={() => updateGS('')}
              aria-label="Clear search"
            >
              ✕
            </button>
          )}
        </div>
      )}

      {/* ── Table ──────────────────────────────────────────────── */}
      <div className={styles.tableWrapper}>
        <table className={tableClasses}>
          <thead>
            <tr>
              {/* Selection checkbox column */}
              {selectionMode === 'multi' && (
                <th className={styles.checkboxCell}>
                  <input
                    type="checkbox"
                    checked={
                      paginatedData.length > 0 &&
                      paginatedData.every((r) => selected.includes(String(r[rowKey])))
                    }
                    onChange={toggleSelectAll}
                    aria-label="Select all rows"
                  />
                </th>
              )}
              {selectionMode === 'single' && <th className={styles.checkboxCell} />}

              {visibleColumns.map((col) => (
                <th
                  key={col.key}
                  className={[
                    styles.th,
                    col.align && styles[`align_${col.align}`],
                  ]
                    .filter(Boolean)
                    .join(' ')}
                  style={{ width: col.width }}
                  aria-sort={
                    sort?.key === col.key
                      ? sort.direction === 'asc'
                        ? 'ascending'
                        : 'descending'
                      : undefined
                  }
                >
                  <div
                    className={[
                      styles.thContent,
                      col.sortable && sortable && styles.sortableHeader,
                    ]
                      .filter(Boolean)
                      .join(' ')}
                    onClick={() => {
                      if (col.sortable !== false && sortable) updateSort(col.key);
                    }}
                    onKeyDown={(e: KeyboardEvent) => {
                      if ((e.key === 'Enter' || e.key === ' ') && col.sortable !== false && sortable) {
                        e.preventDefault();
                        updateSort(col.key);
                      }
                    }}
                    role={col.sortable !== false && sortable ? 'button' : undefined}
                    tabIndex={col.sortable !== false && sortable ? 0 : undefined}
                  >
                    {col.renderHeader ? col.renderHeader(col) : col.header}
                    {col.sortable !== false && sortable && (
                      <SortArrow
                        direction={sort?.key === col.key ? sort.direction : null}
                      />
                    )}
                  </div>

                  {/* Column filter row */}
                  {filterable && col.filterable !== false && (
                    <div className={styles.filterRow}>
                      {col.renderFilter ? (
                        col.renderFilter({
                          column: col,
                          value: filters.find((f) => f.key === col.key)?.value ?? '',
                          onChange: (v) => {
                            const rest = filters.filter((f) => f.key !== col.key);
                            const newF: ColumnFilter[] = v
                              ? [...rest, { key: col.key, value: v, operator: 'contains' }]
                              : rest;
                            updateFilters(newF);
                          },
                          onClear: () => {
                            updateFilters(filters.filter((f) => f.key !== col.key));
                          },
                        })
                      ) : (
                        <input
                          type="text"
                          className={styles.filterInput}
                          placeholder={`Filter ${col.header}...`}
                          value={filters.find((f) => f.key === col.key)?.value ?? ''}
                          onChange={(e: ChangeEvent<HTMLInputElement>) => {
                            const rest = filters.filter((f) => f.key !== col.key);
                            const newF: ColumnFilter[] = e.target.value
                              ? [...rest, { key: col.key, value: e.target.value, operator: 'contains' }]
                              : rest;
                            updateFilters(newF);
                          }}
                          aria-label={`Filter by ${col.header}`}
                        />
                      )}
                    </div>
                  )}
                </th>
              ))}
            </tr>
          </thead>

          <tbody>
            {loading ? (
              Array.from({ length: loadingRows }).map((_, i) => (
                <SkeletonRow key={i} columns={visibleColumns.length + (selectionMode !== 'none' ? 1 : 0)} />
              ))
            ) : paginatedData.length === 0 ? (
              <tr>
                <td
                  colSpan={visibleColumns.length + (selectionMode !== 'none' ? 1 : 0)}
                  className={styles.emptyCell}
                >
                  {emptyState ?? (
                    <div className={styles.emptyDefault}>
                      <span>📭</span>
                      <p>No data to display.</p>
                    </div>
                  )}
                </td>
              </tr>
            ) : (
              paginatedData.map((row, idx) => {
                const key = String(row[rowKey]);
                const isSelected = selected.includes(key);
                return (
                  <tr
                    key={key}
                    className={[
                      styles.row,
                      rowClassName?.(row, idx),
                      onRowClick && styles.clickableRow,
                      isSelected && styles.selectedRow,
                    ]
                      .filter(Boolean)
                      .join(' ')}
                    onClick={() => onRowClick?.(row, idx)}
                    aria-selected={selectionMode !== 'none' ? isSelected : undefined}
                  >
                    {selectionMode === 'multi' && (
                      <td className={styles.checkboxCell}>
                        <input
                          type="checkbox"
                          checked={isSelected}
                          onChange={() => toggleSelectRow(key)}
                          aria-label={`Select row ${key}`}
                        />
                      </td>
                    )}
                    {selectionMode === 'single' && (
                      <td className={styles.checkboxCell}>
                        <input
                          type="radio"
                          name="datatable-select"
                          checked={isSelected}
                          onChange={() => toggleSelectRow(key)}
                          aria-label={`Select row ${key}`}
                        />
                      </td>
                    )}
                    {visibleColumns.map((col) => (
                      <td
                        key={col.key}
                        className={[
                          styles.td,
                          col.align && styles[`align_${col.align}`],
                        ]
                          .filter(Boolean)
                          .join(' ')}
                      >

              : 'No results'}
          </div>

          <div className={styles.paginationControls}>
            <select
              className={styles.pageSizeSelect}
              value={pageSize}
              onChange={(e) => updatePageSize(Number(e.target.value))}
              aria-label="Rows per page"
            >
              {pageSizeOptions.map((opt) => (
                <option key={opt} value={opt}>
                  {opt} per page
                </option>
              ))}
            </select>

            <button
              className={styles.pageBtn}
              disabled={page <= 1}
              onClick={() => updatePage(1)}
              aria-label="First page"
            >
              ««
            </button>
            <button
              className={styles.pageBtn}
              disabled={page <= 1}
              onClick={() => updatePage(page - 1)}
              aria-label="Previous page"
            >
              «
            </button>

            <span className={styles.pageIndicator}>
              Page {page} of {totalPages}
            </span>

            <button
              className={styles.pageBtn}
              disabled={page >= totalPages}
              onClick={() => updatePage(page + 1)}
              aria-label="Next page"
            >
              »
            </button>
            <button
              className={styles.pageBtn}
              disabled={page >= totalPages}
              onClick={() => updatePage(totalPages)}
              aria-label="Last page"
            >
              »»
            </button>
          </div>
        </div>
      )}
    </div>
  );
}
```

## 3. ChartWidget

A flexible chart wrapper that abstracts away chart library specifics. Supports **Recharts** (default), and can be adapted for **Chart.js**, **ECharts**, or **D3**.

### `ChartWidget.types.ts`

```typescript
import type { BaseComponentProps, ColorVariant } from '../types/common';

// ─── Chart Engine ──────────────────────────────────────────────────

export type ChartEngine = 'recharts' | 'chartjs' | 'echarts' | 'custom';

// ─── Data Shape ────────────────────────────────────────────────────

export interface ChartDataPoint {
  label?: string;
  [key: string]: string | number | undefined;
}

// ─── Chart Types ───────────────────────────────────────────────────

export type ChartType =
  | 'line'
  | 'bar'
  | 'area'
  | 'pie'
  | 'donut'
  | 'scatter'
  | 'radar'
  | 'funnel';

export interface ChartAxisConfig {

  /** Axis label */
  label?: string;

export interface ChartSeries {
```

### `ChartWidget.tsx`

```tsx
import React, { useMemo } from 'react';
import type {
  ChartWidgetProps,
  ChartElementClickPayload,
  ChartWidgetInternalProps,
} from './ChartWidget.types';
import {
  LineChart,
  Line,
  BarChart,
  Bar,
  AreaChart,
  Area,
  PieChart,
  Pie,
  Cell,
  ScatterChart,
  Scatter,
  XAxis,
  YAxis,
  CartesianGrid,
  Tooltip,
  Legend,
  ResponsiveContainer,
} from 'recharts';
import styles from './ChartWidget.module.css';

// ─── Default color palette ─────────────────────────────────────────

const DEFAULT_COLORS = [
  '#3b82f6', // blue
  '#ef4444', // red
  '#10b981', // green
  '#f59e0b', // amber
  '#8b5cf6', // violet
  '#ec4899', // pink
  '#06b6d4', // cyan
  '#f97316', // orange
  '#6366f1', // indigo
  '#14b8a6', // teal
];

// ─── Empty state ───────────────────────────────────────────────────

function ChartEmptyState({ message = 'No data available' }: { message?: string }) {
  return (
    <div className={styles.empty}>
      <span>📊</span>
      <p>{message}</p>
    </div>
  );
}

// ─── Loading skeleton ──────────────────────────────────────────────

function ChartSkeleton({ height }: { height: number | string }) {
  return (
    <div className={styles.skeleton} style={{ height }}>
      <div className={styles.skeletonInner} />
    </div>
  );
}

// ─── Recharts Renderer ─────────────────────────────────────────────

function RechartsRenderer({
  data,
  type,
  series,
  xAxis,
  yAxis,
  yAxisSecondary,
  colors,
  showGrid,
  showTooltip,
  showLegend,
  legendPosition,
  animated,
  onElementClick,
  renderTooltip,
  height,
  width,
}: ChartWidgetInternalProps) {
  const dataKey = xAxis?.dataKey ?? 'label';

  const handleClick = (entry: Record<string, unknown>) => {
    if (!onElementClick) return;
    onElementClick({
      label: String(entry[dataKey] ?? ''),
      dataKey: series[0]?.dataKey ?? '',
      value: (entry[series[0]?.dataKey] as number | string) ?? '',
      dataPoint: entry as never,
    });
  };

  const chartProps = {
    data,
    onClick: handleClick as never,
    style: { cursor: onElementClick ? 'pointer' : undefined },
  };

  // Pie / Donut charts
  if (type === 'pie' || type === 'donut') {
    const pieData = data.map((d) => ({
      name: String(d[dataKey] ?? d.label ?? ''),
      value: Number(d[series[0]?.dataKey] ?? 0),
    }));

    return (
      <ResponsiveContainer width="100%" height={height}>
        <PieChart>
          {showTooltip && <Tooltip />}
          {showLegend && (
            <Legend
              verticalAlign={legendPosition === 'top' ? 'top' : 'bottom'}
            />
          )}
          <Pie
            data={pieData}
            cx="50%"
            cy="50%"
            innerRadius={type === 'donut' ? '60%' : 0}
            outerRadius="80%"
            dataKey="value"
            nameKey="name"
            label={type === 'pie'}
            isAnimationActive={animated}
          >
            {pieData.map((_, i) => (
              <Cell key={i} fill={colors[i % colors.length]} />
            ))}
          </Pie>
        </PieChart>
      </ResponsiveContainer>
    );
  }

  // Line, Bar, Area, Scatter
  const renderSeries = () =>
    series
      .filter((s) => !s.hidden)
      .map((s, i) => {
        const color = s.color ?? colors[i % colors.length];
        const props = {
          key: s.dataKey,
          dataKey: s.dataKey,
          name: s.name ?? s.dataKey,
          stroke: color,
          fill: color,
          isAnimationActive: animated,
          stackId: s.stackId,
        };

        switch (type) {
          case 'bar':
            return <Bar {...props} />;
          case 'area':
            return <Area {...props} strokeWidth={2} fillOpacity={0.15} />;
          case 'scatter':
            return <Scatter {...props} />;
          case 'line':
          default:
            return <Line {...props} strokeWidth={2} dot={{ r: 3 }} />;
        }
      });

  const ChartComponent = {
    line: LineChart,
    bar: BarChart,
    area: AreaChart,
    scatter: ScatterChart,
  }[type];

  if (!ChartComponent) return null;

  return (
    <ResponsiveContainer width="100%" height={height}>
      <ChartComponent {...chartProps}>
        {showGrid && <CartesianGrid strokeDasharray="3 3" stroke="#e2e8f0" />}
        <XAxis
          dataKey={dataKey}
          hide={xAxis?.hide}
          tickFormatter={xAxis?.tickFormatter}
          domain={xAxis?.domain as [number, number]}
          label={xAxis?.label ? { value: xAxis.label, position: 'bottom' } : undefined}
          tick={{ fontSize: 12, fill: '#64748b' }}
          axisLine={{ stroke: '#e2e8f0' }}
          tickLine={false}
        />
        <YAxis
          hide={yAxis?.hide}
          tickFormatter={yAxis?.tickFormatter}
          domain={yAxis?.domain as [number, number]}
          label={yAxis?.label ? { value: yAxis.label, angle: -90, position: 'insideLeft' } : undefined}
          tick={{ fontSize: 12, fill: '#64748b' }}
          axisLine={{ stroke: '#e2e8f0' }}
          tickLine={false}
        />
        {yAxisSecondary && (
          <YAxis
            yAxisId="secondary"
            orientation="right"
            hide={yAxisSecondary.hide}
            tickFormatter={yAxisSecondary.tickFormatter}
            tick={{ fontSize: 12, fill: '#64748b' }}
            axisLine={{ stroke: '#e2e8f0' }}
            tickLine={false}
          />
        )}
        {showTooltip && (renderTooltip ? <Tooltip content={renderTooltip as never} /> : <Tooltip />)}
        {showLegend && <Legend verticalAlign={legendPosition === 'top' ? 'top' : 'bottom'} />}
        {renderSeries()}
      </ChartComponent>
    </ResponsiveContainer>
  );
}

// ─── Main Component ────────────────────────────────────────────────

export function ChartWidget({
  data,
  type,
  series,
  xAxis,
  yAxis,
  yAxisSecondary,
  height = 300,
  width = '100%',
  colors = DEFAULT_COLORS,
  background,
  showGrid = true,
  borderRadius = 8,
  showLegend = true,
  legendPosition = 'bottom',
  showTooltip = true,
  title,
  subtitle,
  loading = false,
  emptyState,
  error,
  animated = true,
  responsive = true,
  aspectRatio,
  onElementClick,
  engine = 'recharts',
  renderChart,
  renderTooltip,
  className,
  style,
  'data-testid': dataTestId,
  'aria-label': ariaLabel,
}: ChartWidgetProps) {
  const containerStyle: React.CSSProperties = {
    ...style,
    background: background ?? 'transparent',
    borderRadius: typeof borderRadius === 'number' ? `${borderRadius}px` : borderRadius,
    height: responsive ? undefined : height,
    width: responsive ? '100%' : width,
    aspectRatio: aspectRatio ? String(aspectRatio) : undefined,
  };

  const internalProps: ChartWidgetInternalProps = {
    data,
    type,
    series,
    xAxis,
    yAxis,
    yAxisSecondary,
    colors,
    showGrid,
    showTooltip,
    showLegend,
    legendPosition,
    animated,
    onElementClick,
    renderTooltip,
    height,
    width,
  };

  const renderContent = () => {
    if (error) {
      return (
        <div className={styles.error}>
          <span>⚠️</span>
          <p>{error.message}</p>
        </div>
      );
    }

    if (loading) {
      return <ChartSkeleton height={height} />;
    }

    if (!data || data.length === 0) {
      return emptyState ?? <ChartEmptyState />;
    }

    if (engine === 'custom' && renderChart) {
      return renderChart(internalProps);
    }

    return <RechartsRenderer {...internalProps} />;
  };

  return (
    <div
      className={[styles.container, className].filter(Boolean).join(' ')}
      style={containerStyle}
      data-testid={dataTestId}
      aria-label={ariaLabel ?? title ?? 'Chart widget'}
    >
      {/* Header */}
      {(title || subtitle) && (
        <div className={styles.header}>
          {title && <h3 className={styles.title}>{title}</h3>}
          {subtitle && <p className={styles.subtitle}>{subtitle}</p>}
        </div>
      )}

      {/* Chart area */}
      <div className={styles.chartArea} style={{ height: responsive ? undefined : height }}>
        {renderContent()}
      </div>
    </div>
  );
}

export default ChartWidget;
```

### `SearchBar.tsx`

```tsx
import React, {
  useCallback,
  useEffect,
  useMemo,
  useRef,
  useState,
  type ChangeEvent,
  type KeyboardEvent,
} from 'react';
import type { SearchBarProps, SuggestionItem } from './SearchBar.types';
import styles from './SearchBar.module.css';

// ─── Hook: useDebounce ─────────────────────────────────────────────

function useDebounce<T>(value: T, delay: number): T {
  const [debounced, setDebounced] = useState(value);
  useEffect(() => {
    const timer = setTimeout(() => setDebounced(value), delay);
    return () => clearTimeout(timer);
  }, [value, delay]);
  return debounced;
}

// ─── Search Icon ───────────────────────────────────────────────────

function SearchIcon() {
  return (
    <svg
      width="16"
      height="16"
      viewBox="0 0 24 24"
      fill="none"
      stroke="currentColor"
      strokeWidth="2"
      strokeLinecap="round"
      strokeLinejoin="round"
    >
      <circle cx="11" cy="11" r="8" />
      <path d="M21 21l-4.35-4.35" />
    </svg>
  );
}

// ─── Highlight matching text ───────────────────────────────────────

function highlightMatch(text: string, query: string): React.ReactNode {
  if (!query) return text;
  const idx = text.toLowerCase().indexOf(query.toLowerCase());
  if (idx === -1) return text;
  return (
    <>
      {text.slice(0, idx)}
      <mark className={styles.highlight}>{text.slice(idx, idx + query.length)}</mark>
      {text.slice(idx + query.length)}
    </>
  );
}

// ─── Default Suggestion Renderer ───────────────────────────────────

function DefaultSuggestion({
  item,
  isHighlighted,
  query,
}: {
  item: SuggestionItem;
  isHighlighted: boolean;
  query: string;
}) {
  return (
    <div
      className={[
        styles.suggestionItem,
        isHighlighted && styles.suggestionHighlighted,
      ]
        .filter(Boolean)
        .join(' ')}
    >
      {item.icon && <span className={styles.suggestionIcon}>{item.icon}</span>}
      <div className={styles.suggestionContent}>
        <span className={styles.suggestionLabel}>
          {highlightMatch(item.label, query)}
        </span>
        {item.description && (
          <span className={styles.suggestionDescription}>
            {highlightMatch(item.description, query)}
          </span>
        )}
      </div>
      {item.category && (
        <span className={styles.suggestionCategory}>{item.category}</span>
      )}
    </div>
  );
}

// ─── Main Component ────────────────────────────────────────────────

export function SearchBar({
  // Value
  value: controlledValue,
  defaultValue = '',
  onChange,

  // Suggestions
  suggestions: staticSuggestions,
  fetchSuggestions,

  // Behavior
  minQueryLength = 2,
  debounceMs = 300,
  maxSuggestions = 10,
  clearOnSelect = false,
  showIcon = true,
  showClearButton = true,

  // Styling
  placeholder = 'Search...',
  size = 'md',
  fullWidth = false,
  rounded = 'md',

  // Callbacks
  onSelect,
  onSubmit,
  onFocus,
  onBlur,

  // Accessibility
  id,
  label = 'Search',
  disabled = false,
  helpText,
  error,

  // Custom rendering
  renderSuggestion,
  renderNoResults,
  renderHeader,
  renderFooter,

  // Base
  className,
  style,
  'data-testid': dataTestId,
  'aria-describedby': ariaDescribedby,
}: SearchBarProps) {
  // ── State ──────────────────────────────────────────────────────
  const [internalValue, setInternalValue] = useState(defaultValue);
  const [isOpen, setIsOpen] = useState(false);
  const [highlightIndex, setHighlightIndex] = useState(-1);
  const [asyncSuggestions, setAsyncSuggestions] = useState<SuggestionItem[]>([]);
  const [isLoading, setIsLoading] = useState(false);
  const [isFocused, setIsFocused] = useState(false);

  const inputRef = useRef<HTMLInputElement>(null);
  const listRef = useRef<HTMLUListElement>(null);
  const containerRef = useRef<HTMLDivElement>(null);

  const value = controlledValue ?? internalValue;
  const debouncedValue = useDebounce(value, debounceMs);

  // ── Fetch suggestions ──────────────────────────────────────────
  useEffect(() => {
    if (!fetchSuggestions || debouncedValue.length < minQueryLength) {
      setAsyncSuggestions([]);
      return;
    }

    let cancelled = false;
    setIsLoading(true);

    fetchSuggestions(debouncedValue)
      .then((results) => {
        if (!cancelled) {
          setAsyncSuggestions(results.slice(0, maxSuggestions));
          setIsLoading(false);
        }
      })
      .catch(() => {
        if (!cancelled) setIsLoading(false);
      });

    return () => {
      cancelled = true;
    };
  }, [debouncedValue, fetchSuggestions, minQueryLength, maxSuggestions]);

  // ── Compute suggestions ────────────────────────────────────────
  const suggestions = useMemo(() => {
    if (staticSuggestions) {
      const query = debouncedValue.toLowerCase();
      return staticSuggestions
        .filter(
          (s) =>
            s.label.toLowerCase().includes(query) ||
            (s.description?.toLowerCase().includes(query) ?? false),
        )
        .slice(0, maxSuggestions);
    }
    if (fetchSuggestions) {
      return asyncSuggestions;
    }
    return [];
  }, [staticSuggestions, asyncSuggestions, debouncedValue, fetchSuggestions, maxSuggestions]);

  const showDropdown =
    isOpen &&
    value.length >= minQueryLength &&
    (suggestions.length > 0 || isLoading || !!renderNoResults);

  // ── Helpers ───────────────────────────────────────────────────
  const setValue = useCallback(
    (val: string) => {
      setInternalValue(val);
      onChange?.(val);
    },
    [onChange],
  );

  const selectSuggestion = useCallback(
    (item: SuggestionItem) => {
      setValue(clearOnSelect ? '' : item.label);
      onSelect?.(item);
      setIsOpen(false);
      setHighlightIndex(-1);
      inputRef.current?.blur();
    },
    [setValue, clearOnSelect, onSelect],
  );

  const closeDropdown = useCallback(() => {
    // Delay to allow click events on suggestions to fire
    setTimeout(() => {
      setIsOpen(false);
      setIsFocused(false);
      setHighlightIndex(-1);
      onBlur?.();
    }, 150);
  }, [onBlur]);

  const handleInputChange = useCallback(
    (e: ChangeEvent<HTMLInputElement>) => {
      const val = e.target.value;
      setValue(val);
      setIsOpen(true);
      setHighlightIndex(-1);
    },
    [setValue],
  );

  const handleKeyDown = useCallback(
    (e: KeyboardEvent<HTMLInputElement>) => {
      if (!showDropdown) {
        if (e.key === 'Enter') {
          e.preventDefault();
          onSubmit?.(value);
          setIsOpen(false);
        }
        return;
      }

      switch (e.key) {
        case 'ArrowDown': {
          e.preventDefault();
          setHighlightIndex((prev) =>
            prev < suggestions.length - 1 ? prev + 1 : 0,
          );
          break;
        }
        case 'ArrowUp': {
          e.preventDefault();
          setHighlightIndex((prev) =>
            prev > 0 ? prev - 1 : suggestions.length - 1,
          );
          break;
        }
        case 'Enter': {
          e.preventDefault();
          if (highlightIndex >= 0 && highlightIndex < suggestions.length) {
            selectSuggestion(suggestions[highlightIndex]);
          } else {
            onSubmit?.(value);
            setIsOpen(false);
          }
          break;
        }
        case 'Escape': {
          e.preventDefault();
          setIsOpen(false);
          setHighlightIndex(-1);
          inputRef.current?.blur();
          break;
        }
        case 'Tab': {
          setIsOpen(false);
          setHighlightIndex(-1);
          break;
        }
      }
    },
    [showDropdown, suggestions, highlightIndex, value, selectSuggestion, onSubmit],
  );

  // ── Scroll highlighted suggestion into view ──────────────────
  useEffect(() => {
    if (highlightIndex >= 0 && listRef.current) {
      const el = listRef.current.children[highlightIndex] as HTMLElement | undefined;
      el?.scrollIntoView({ block: 'nearest' });
    }
  }, [highlightIndex]);

  // ── Close on outside click ────────────────────────────────────
  useEffect(() => {
    const handleClickOutside = (e: MouseEvent) => {
      if (containerRef.current && !containerRef.current.contains(e.target as Node)) {
        closeDropdown();
      }
    };
    document.addEventListener('mousedown', handleClickOutside);
    return () => document.removeEventListener('mousedown', handleClickOutside);
  }, [closeDropdown]);

  // ── Render ────────────────────────────────────────────────────
  const wrapperClasses = [
    styles.wrapper,
    fullWidth && styles.fullWidth,
    className,
  ]
    .filter(Boolean)
    .join(' ');

  const inputClasses = [
    styles.input,
    styles[`size_${size}`],
    styles[`rounded_${rounded}`],
    showIcon && styles.inputWithIcon,
    showClearButton && value && styles.inputWithClear,
    error && styles.inputError,
    isFocused && styles.inputFocused,
  ]
    .filter(Boolean)
    .join(' ');

  return (
    <div
      ref={containerRef}
      className={wrapperClasses}
      style={style}
      data-testid={dataTestId}
    >
      {/* Visual label (screen-reader only) */}
      <label htmlFor={id} className={styles.srOnly}>
        {label}
      </label>

      <div className={styles.inputWrapper}>
        {showIcon && (
          <span className={styles.iconLeft}>
            <SearchIcon />
          </span>
        )}

        <input
          ref={inputRef}
          id={id}
          type="text"
          className={inputClasses}
          value={value}
          onChange={handleInputChange}
          onKeyDown={handleKeyDown}
          onFocus={() => {
            setIsFocused(true);
            setIsOpen(true);
            onFocus?.();
          }}
          placeholder={placeholder}
          disabled={disabled}
          role="combobox"
          aria-expanded={showDropdown}
          aria-autocomplete="list"
          aria-controls={showDropdown ? `${id}-listbox` : undefined}
          aria-activedescendant={
            highlightIndex >= 0 ? `${id}-option-${highlightIndex}` : undefined
          }
          aria-label={label}
          aria-describedby={ariaDescribedby}
          autoComplete="off"
        />

        {showClearButton && value && !disabled && (
          <button
            type="button"
            className={styles.clearBtn}
            onClick={() => {
              setValue('');
              inputRef.current?.focus();
            }}
            aria-label="Clear search"
          >
            ✕
          </button>
        )}

        {isLoading && (
          <span className={styles.spinner} aria-label="Loading suggestions" />
        )}
      </div>

      {/* Help / Error text */}
      {helpText && !error && <p className={styles.helpText}>{helpText}</p>}
      {error && <p className={styles.errorText}>{error}</p>}

      {/* Dropdown */}
      {showDropdown && (
        <div className={styles.dropdown}>
          {renderHeader?.()}

          {isLoading && suggestions.length === 0 && (
            <div className={styles.dropdownMessage}>Searching...</div>
          )}

          {!isLoading && suggestions.length === 0 && (
            <div className={styles.dropdownMessage}>
              {renderNoResults?.(value) ?? `No results for "${value}"`}
            </div>
          )}

          {suggestions.length > 0 && (
            <ul
              ref={listRef}
              id={`${id}-listbox`}
              className={styles.suggestionsList}
              role="listbox"
            >
              {suggestions.map((item, idx) => (
                <li
                  key={item.id}
                  id={`${id}-option-${idx}`}
                  role="option"
                  aria-selected={idx === highlightIndex}
                  onMouseDown={(e) => {
                    e.preventDefault();
                    selectSuggestion(item);
                  }}
                  onMouseEnter={() => setHighlightIndex(idx)}
                >
                  {renderSuggestion ? (
                    renderSuggestion(item, idx === highlightIndex)
                  ) : (
                    <DefaultSuggestion
                      item={item}
                      isHighlighted={idx === highlightIndex}
                      query={value}
                    />
                  )}
                </li>
              ))}
            </ul>
          )}

          {renderFooter?.(value)}
        </div>
      )}
    </div>
  );
}

export default SearchBar;
```

```tsx
            e.stopPropagation();

// ─── Time ago helper ──────────────────────────────────────────────

function getTimeAgo(timestamp: string): string {
  const now = Date.now();
  const then = new Date(timestamp).getTime();
  const seconds = Math.floor((now - then) / 1000);

  if (seconds < 60) return 'just now';
  const minutes = Math.floor(seconds / 60);
  if (minutes < 60) return `${minutes}m ago`;
  const hours = Math.floor(minutes / 60);
  if (hours < 24) return `${hours}h ago`;
  const days = Math.floor(hours / 24);
  if (days < 7) return `${days}d ago`;
  return new Date(timestamp).toLocaleDateString();
}

// ─── Main Component ───────────────────────────────────────────────

export function NotificationBell({
  notifications,
  unreadCount,
  loading = false,
  error,
  onNotificationClick,
  onMarkAsRead,
  onMarkAllRead,
  onOpen,
  onClose,
  onLoadMore,
  hasMore = false,
  size = 'md',
  bellIcon,
  dropdownMaxHeight = 400,
  dropdownWidth = 380,
  emptyState,
  renderNotification,
  renderHeader,
  renderFooter,
  className,
  style,
  'data-testid': dataTestId,
  'aria-label': ariaLabel,
}: NotificationBellProps) {
  const [isOpen, setIsOpen] = useState(false);
  const containerRef = useRef<HTMLDivElement>(null);
  const dropdownRef = useRef<HTMLDivElement>(null);

  const unread = unreadCount ?? notifications.filter((n) => !n.read).length;

  // ── Toggle ────────────────────────────────────────────────────
  const toggle = useCallback(() => {
    setIsOpen((prev) => {
      const next = !prev;
      if (next) onOpen?.();
      else onClose?.();
      return next;
    });
  }, [onOpen, onClose]);

  // ── Close on outside click ─────────────────────────────────────
  useEffect(() => {
    const handler = (e: MouseEvent) => {
      if (containerRef.current && !containerRef.current.contains(e.target as Node)) {
        setIsOpen(false);
        onClose?.();
      }
    };
    if (isOpen) {
      document.addEventListener('mousedown', handler);
    }
    return () => document.removeEventListener('mousedown', handler);
  }, [isOpen, onClose]);

  // ── Close on Escape ────────────────────────────────────────────
  useEffect(() => {
    const handler = (e: globalThis.KeyboardEvent) => {
      if (e.key === 'Escape' && isOpen) {
        setIsOpen(false);
        onClose?.();
      }
    };
    document.addEventListener('keydown', handler);
    return () => document.removeEventListener('keydown', handler);
  }, [isOpen, onClose]);

  // ── Infinite scroll ────────────────────────────────────────────
  const handleScroll = useCallback(
    (e: React.UIEvent<HTMLDivElement>) => {
      if (!onLoadMore || !hasMore || loading) return;
      const el = e.currentTarget;
      if (el.scrollHeight - el.scrollTop - el.clientHeight < 40) {
        onLoadMore();
      }
    },
    [onLoadMore, hasMore, loading],
  );

  // ── Bell Button Classes ────────────────────────────────────────
  const bellClasses = [
    styles.bell,
    styles[`size_${size}`],
    isOpen && styles.bellActive,
  ]
    .filter(Boolean)
    .join(' ');

  return (
    <div
      ref={containerRef}
      className={[styles.container, className].filter(Boolean).join(' ')}
      style={style}
      data-testid={dataTestId}
    >
      {/* Bell Button */}
      <button
        className={bellClasses}
        onClick={toggle}
        aria-label={ariaLabel ?? `Notifications (${unread} unread)`}
        aria-expanded={isOpen}
        aria-haspopup="true"
      >
        {bellIcon ?? <BellIcon />}
        {unread > 0 && (
          <span className={styles.badge} aria-hidden="true">
            {unread > 99 ? '99+' : unread}
          </span>
        )}
      </button>

      {/* Dropdown */}
      {isOpen && (
        <div
          ref={dropdownRef}
          className={styles.dropdown}
          style={{
            maxHeight: dropdownMaxHeight,
            width: dropdownWidth,
          }}
          onScroll={handleScroll}
        >
          {/* Header */}
          <div className={styles.dropdownHeader}>
            {renderHeader?.() ?? (
              <>
                <span className={styles.headerTitle}>Notifications</span>
                {unread > 0 && onMarkAllRead && (
                  <button
                    className={styles.markAllRead}
                    onClick={() => {
                      onMarkAllRead();
                    }}
                  >
                    Mark all read
                  </button>
                )}
              </>
            )}
          </div>

          {/* Body */}
          <div className={styles.dropdownBody}>
            {error ? (
              <div className={styles.dropdownError}>
                <span>⚠️</span>
                <p>{error.message || 'Failed to load notifications'}</p>
              </div>
            ) : loading && notifications.length === 0 ? (
              <div className={styles.dropdownLoading}>
                {Array.from({ length: 4 }).map((_, i) => (
                  <div key={i} className={styles.skeletonItem}>
                    <div className={styles.skeletonCircle} />
                    <div className={styles.skeletonLines}>
                      <div className={styles.skeletonLine} />
                      <div className={styles.skeletonLineShort} />
                    </div>
                  </div>
                ))}
              </div>
            ) : notifications.length === 0 ? (
              <div className={styles.dropdownEmpty}>
                {emptyState ?? (
                  <>
                    <span>🔔</span>
                    <p>No notifications yet</p>
                    <span className={styles.emptySubtext}>
                      We'll let you know when something arrives
                    </span>
                  </>
                )}
              </div>
            ) : (
              notifications.map((notification) => (
                <div key={notification.id}>
                  {renderNotification ? (
                    renderNotification(notification)
                  ) : (
                    <DefaultNotificationItem
                      notification={notification}
                      onClick={onNotificationClick}
                      onMarkAsRead={onMarkAsRead}
                    />
                  )}
                </div>
              ))
            )}
          </div>

          {/* Footer */}
          {renderFooter?.({
            hasMore,
            loading,
            onLoadMore: onLoadMore ?? (() => {}),
          })}

          {/* Loading more indicator */}
          {loading && notifications.length > 0 && (
            <div className={styles.loadingMore}>Loading more...</div>
          )}
        </div>
      )}
    </div>
  );
}

export default NotificationBell;
```

```tsx
// ─── Generate initials from name ──────────────────────────────────

function getInitials(name?: string): string {
  if (!name) return '?';
  const parts = name.trim().split(/\s+/);
  if (parts.length === 1) return parts[0].charAt(0).toUpperCase();
  return (parts[0].charAt(0) + parts[parts.length - 1].charAt(0)).toUpperCase();
}

// ─── Deterministic color from string ──────────────────────────────

function stringToColor(str: string): string {
  let hash = 0;
  for (let i = 0; i < str.length; i++) {
    hash = str.charCodeAt(i) + ((hash << 5) - hash);
  }
  const h = Math.abs(hash) % 360;
  return `hsl(${h}, 55%, 55%)`;
}

export function UserAvatar({
  src,
  alt = '',
  name,
  initials: explicitInitials,
  size = 'md',
  shape = 'circle',
  status,
  showStatus = false,
  statusPosition = 'bottom-right',
  pulseOnline = true,
  background,
  foreground = '#fff',
  borderColor,
  borderWidth,
  onClick,
  showRing = false,
  ringColor,
  stacked = false,
  overflowCount,
  role = 'img',
  className,
  style,
  'data-testid': dataTestId,
  'aria-label': ariaLabel,
}: UserAvatarProps) {
  const [imgError, setImgError] = useState(false);

  const initials = explicitInitials ?? getInitials(name);
  const bgColor = background ?? (name ? stringToColor(name) : '#cbd5e1');
  const dimension = { xs: 24, sm: 32, md: 40, lg: 48, xl: 64, '2xl': 96 }[size];

    circle: '50%',
    rounded: size === 'xs' || size === 'sm' ? 6 : size === 'md' ? 8 : 12,
    square: 0,
  }[shape];

  const containerClasses = [
    styles.avatar,
    styles[`size_${size}`],
    styles[`shape_${shape}`],
    onClick && styles.clickable,
    stacked && styles.stacked,
    showRing && styles.ring,
  ]
    .filter(Boolean)
    .join(' ');

  const containerStyle: React.CSSProperties = {
    ...style,
    width: dimension,
    height: dimension,
    borderRadius: typeof shapeRadius === 'number' ? `${shapeRadius}px` : shapeRadius,
    borderColor: borderColor ?? (showRing ? ringColor ?? STATUS_COLORS[status ?? 'offline'] : undefined),
    borderWidth: borderWidth ?? (showRing ? 2 : undefined),
  };

  // Overflow count
  if (overflowCount !== undefined) {
    return (
      <div
        className={containerClasses}
        style={{
          ...containerStyle,
          backgroundColor: '#e2e8f0',
          color: '#475569',
          display: 'flex',
          alignItems: 'center',
          justifyContent: 'center',

          fontWeight: 600,
        }}
        data-testid={dataTestId}
        aria-label={ariaLabel ?? `${overflowCount} more users`}
        role="status"
      >
        +{overflowCount > 99 ? '99' : overflowCount}
        {showStatus && status && (
          <StatusDot status={status} position={statusPosition} pulse={pulseOnline} size={size} />
        )}
      </div>
    );
  }

  return (
    <div
      className={containerClasses}
      style={containerStyle}
      onClick={onClick}
      onKeyDown={
        onClick
          ? (e) => {
              if (e.key === 'Enter' || e.key === ' ') {
                e.preventDefault();
                onClick();
              }
            }
          : undefined
      }
      role={onClick ? 'button' : role}
      tabIndex={onClick ? 0 : undefined}
      data-testid={dataTestId}
      aria-label={ariaLabel ?? alt ?? name ?? 'User avatar'}
      title={name}
    >
      {/* Image */}
      {src && !imgError ? (
        <img
          className={styles.image}
          src={src}
          alt={alt ?? name ?? ''}
          onError={() => setImgError(true)}
          style={{ borderRadius: containerStyle.borderRadius }}
        />
      ) : (
        /* Initials fallback */
        <span
          className={styles.initials}
          style={{
            backgroundColor: name ? bgColor : '#cbd5e1',
            color: foreground,
            fontSize: dimension * 0.4,
          }}
        >
          {initials}
        </span>
      )}

      {/* Status indicator */}
      {showStatus && status && (
        <StatusDot status={status} position={statusPosition} pulse={pulseOnline} size={size} />
      )}
    </div>
  );
}
```

### `FormBuilder.tsx`

```tsx
import React, { useCallback, useEffect, useMemo, useState, type FormEvent } from 'react';
import type {
  FormBuilderProps,
  FormField,
  FormFieldType,
  FormSection,
  FieldRenderProps,
  ValidationRule,
} from './FormBuilder.types';
import styles from './FormBuilder.module.css';

// ─── Validation Engine ────────────────────────────────────────────

function validateField(
  field: FormField,
  value: unknown,
  formValues: Record<string, unknown>,
): string | null {
  if (!field.rules) return null;

  for (const rule of field.rules) {
    const strVal = typeof value === 'string' ? value : String(value ?? '');

    switch (rule.type) {
      case 'required': {
        if (value === undefined || value === null || value === '' || (Array.isArray(value) && value.length === 0)) {
          return rule.message;
        }
        break;
      }
      case 'minLength': {
        if (typeof rule.value === 'number' && strVal.length < rule.value) {
          return rule.message;
        }
        break;
      }
      case 'maxLength': {
        if (typeof rule.value === 'number' && strVal.length > rule.value) {
          return rule.message;
        }
        break;
      }
      case 'min': {
        if (typeof rule.value === 'number' && Number(value) < rule.value) {
          return rule.message;
        }
        break;
      }
      case 'max': {
        if (typeof rule.value === 'number' && Number(value) > rule.value) {
          return rule.message;
        }
        break;
      }
      case 'pattern': {
        const regex = rule.value instanceof RegExp ? rule.value : new RegExp(String(rule.value));
        if (!regex.test(strVal)) {
          return rule.message;
        }
        break;
      }
      case 'email': {
        const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
        if (!emailRegex.test(strVal)) {
          return rule.message;
        }
        break;
      }
      case 'custom': {
        if (rule.validator) {
          const result = rule.validator(value, formValues);
          if (typeof result === 'string') return result;
          if (result === false) return rule.message;
        }
        break;
      }
    }
  }

  return null;
}

function validateForm(
  fields: FormField[],
  values: Record<string, unknown>,
): Record<string, string> {
  const errors: Record<string, string> = {};
  for (const field of fields) {
    if (field.hidden) continue;
    const error = validateField(field, values[field.name], values);
    if (error) errors[field.name] = error;
  }
  return errors;
}

      disabled={field.disabled}
      readOnly={field.readOnly}
      onChange={(e) => onChange(e.target.value)}
      onBlur={onBlur}
      aria-invalid={!!(error && touched)}
    />
  );
}

function RangeInput({ field, value, error, touched, onChange, onBlur }: FieldRenderProps) {
  return (
    <div className={styles.rangeWrapper}>
      <input
        type="range"
        className={styles.range}
        min={field.min ?? 0}
        max={field.max ?? 100}
        step={field.step ?? 1}
        value={(value as number) ?? field.min ?? 0}
        disabled={field.disabled}
        onChange={(e) => onChange(Number(e.target.value))}
        onBlur={onBlur}
        aria-invalid={!!(error && touched)}
      />
      <span className={styles.rangeValue}>{String(value ?? field.min ?? 0)}</span>
    </div>
  );
}

// ─── Default renderer map ─────────────────────────────────────────

const DEFAULT_RENDERERS: Record<FormFieldType, React.FC<FieldRenderProps>> = {
  text: TextInput,
  number: NumberInput,
  email: TextInput,
  password: TextInput,
  textarea: TextareaInput,
  select: SelectInput,
  multiselect: SelectInput,
  checkbox: CheckboxInput,
  radio: RadioInput,
  switch: SwitchInput,
  date: DateInput,
  datetime: DateInput,
  time: DateInput,
  file: () => <input type="file" />, // simplified
  color: ({ field, value, onChange, onBlur }: FieldRenderProps) => (
    <input
      type="color"
      value={(value as string) ?? '#000000'}
      disabled={field.disabled}
      onChange={(e) => onChange(e.target.value)}
      onBlur={onBlur}
    />
  ),
  range: RangeInput,
  richtext: TextareaInput, // fallback
  custom: () => <div>Custom renderer not provided</div>,
};

// ─── Main Component ───────────────────────────────────────────────

export function FormBuilder({
  fields: flatFields,
  sections,
  values: controlledValues,
  defaultValues = {},
  onValuesChange,
  onSubmit,
  submitLabel = 'Submit',
  showSubmit = true,
  isSubmitting: externalSubmitting = false,
  validateOnBlur = true,
  validateOnChange = false,
  resetOnSubmit = false,
  columns = 12,
  gap = '16px',
  labelPosition = 'top',
  fieldSize = 'md',
  onSubmitError,
  onFieldFocus,
  onFieldBlur,
  fieldRenderers,
  renderFieldWrapper,
  renderSubmit,
  className,
  style,
  'data-testid': dataTestId,
}: FormBuilderProps) {
  // ── Compute all fields ────────────────────────────────────────
  const allFields = useMemo(() => {
    if (sections) {
      return sections.flatMap((s) => s.fields);
    }
    return (flatFields ?? []).sort((a, b) => (a.order ?? 0) - (b.order ?? 0));
  }, [flatFields, sections]);

  // ── Values state ─────────────────────────────────────────────
  const [internalValues, setInternalValues] = useState<Record<string, unknown>>(() => {
    const init: Record<string, unknown> = { ...defaultValues };
    for (const f of allFields) {
      if (!(f.name in init) && f.defaultValue !== undefined) {
        init[f.name] = f.defaultValue;
      }
    }
    return init;
  });

  const values = controlledValues ?? internalValues;

  const setValues = useCallback(
    (updater: Record<string, unknown> | ((prev: Record<string, unknown>) => Record<string, unknown>)) => {
      setInternalValues((prev) => {
        const next = typeof updater === 'function' ? updater(prev) : { ...prev, ...updater };
        onValuesChange?.(next);
        return next;
      });
    },
    [onValuesChange],
  );

  // ── Errors & touched ─────────────────────────────────────────
  const [errors, setErrors] = useState<Record<string, string>>({});
  const [touched, setTouched] = useState<Record<string, boolean>>({});
  const [isSubmitting, setIsSubmitting] = useState(false);

  // ── Helpers ──────────────────────────────────────────────────
  const resolveRenderer = useCallback(
    (field: FormField): React.FC<FieldRenderProps> => {
      if (field.render) return field.render as unknown as React.FC<FieldRenderProps>;
      if (fieldRenderers?.[field.type]) return fieldRenderers[field.type]!;
      return DEFAULT_RENDERERS[field.type] ?? DEFAULT_RENDERERS.text;
    },
    [fieldRenderers],
  );

  const handleFieldChange = useCallback(
    (field: FormField, val: unknown) => {
      setValues({ [field.name]: val });
      if (validateOnChange) {
        const err = validateField(field, val, values);
        setErrors((prev) => {
          const next = { ...prev };
          if (err) next[field.name] = err;
          else delete next[field.name];
          return next;
        });
      }
    },
    [setValues, validateOnChange, values],
  );

  const handleFieldBlur = useCallback(
    (field: FormField) => {
      setTouched((prev) => ({ ...prev, [field.name]: true }));
      onFieldBlur?.(field.name);
      if (validateOnBlur) {
        const err = validateField(field, values[field.name], values);
        setErrors((prev) => {
          const next = { ...prev };
          if (err) next[field.name] = err;
          else delete next[field.name];
          return next;
        });
      }
    },
    [validateOnBlur, values, onFieldBlur],
  );

  const handleSubmit = useCallback(
    async (e: FormEvent) => {
      e.preventDefault();

      // Mark all touched
      const allTouched: Record<string, boolean> = {};
      allFields.forEach((f) => (allTouched[f.name] = true));
      setTouched(allTouched);

      // Validate
      const formErrors = validateForm(allFields, values);
      setErrors(formErrors);

      if (Object.keys(formErrors).length > 0) {
        onSubmitError?.(formErrors);
        return;
      }

      setIsSubmitting(true);
      try {
        await onSubmit?.(values);
        if (resetOnSubmit) {
          setTouched({});
          setErrors({});
        }
      } catch (err) {
        console.error('Form submission error:', err);
      } finally {
        setIsSubmitting(false);
      }
    },
    [allFields, values, onSubmit, resetOnSubmit, onSubmitError],
  );

  // ── Render single field ──────────────────────────────────────
  const renderField = useCallback(
    (field: FormField) => {
      if (field.hidden) return null;

      // Conditional visibility
      if (field.showWhen && !field.showWhen(values)) return null;

      const isDisabled = field.disabled || (field.enableWhen ? !field.enableWhen(values) : false);
      const fieldWithDisabled = { ...field, disabled: isDisabled };

      const error = errors[field.name] ?? null;
      const isTouched = touched[field.name] ?? false;

      const fp: FieldRenderProps = {
        field: fieldWithDisabled,
        value: values[field.name],
        error,
        touched: isTouched,
        onChange: (val) => handleFieldChange(field, val),
        onBlur: () => handleFieldBlur(field),
        formValues: values,
        allFields,
      };

      const Renderer = resolveRenderer(field);

      const showLabel =
        labelPosition !== 'floating' &&
        field.type !== 'checkbox' &&
        field.type !== 'switch';

      const fieldContent = (
        <div
          key={field.name}
          className={[
            styles.field,
            styles[`label_${labelPosition}`],
            styles[`size_${fieldSize}`],
            field.colSpan && styles[`colSpan_${field.colSpan}`],
          ]
            .filter(Boolean)
            .join(' ')}
          style={{ gridColumn: field.colSpan ? `span ${field.colSpan}` : undefined }}
        >
          {showLabel && field.label && (
            <label className={styles.label} htmlFor={`form-${field.name}`}>
              {field.label}
              {field.required && <span className={styles.required}>*</span>}
            </label>
          )}
          <Renderer {...fp} />
          {field.helpText && !error && (
            <p className={styles.helpText}>{field.helpText}</p>
          )}
          {error && isTouched && (
            <p className={styles.errorText} id={`${field.name}-error`} role="alert">
              {error}
            </p>
          )}
        </div>
      );

      if (renderFieldWrapper) {
        return (
          <React.Fragment key={field.name}>
            {renderFieldWrapper(fp, fieldContent)}
          </React.Fragment>
        );
      }

      return fieldContent;
    },
    [
      values, errors, touched, allFields,
      handleFieldChange, handleFieldBlur, resolveRenderer,
      labelPosition, fieldSize, renderFieldWrapper,
    ],
  );

  // ── Compute is-valid / dirty ─────────────────────────────────
  const isValid = Object.keys(errors).length === 0;
  const dirty = Object.keys(touched).length > 0;

  // ── Render ───────────────────────────────────────────────────
  return (
    <form
      className={[styles.form, className].filter(Boolean).join(' ')}
      style={{ ...style, gap }}
      onSubmit={handleSubmit}
      noValidate
      data-testid={dataTestId}
    >
      {/* Sections or flat layout */}
      {sections ? (
        sections.map((section, si) => (
          <SectionRenderer
            key={si}
            section={section}
            renderField={renderField}
          />
        ))
      ) : (
        <div className={styles.fieldsGrid} style={{ gridTemplateColumns: `repeat(${columns}, 1fr)` }}>
          {allFields.map(renderField)}
        </div>
      )}

      {/* Submit */}
      {showSubmit && (
        <div className={styles.submitRow}>
          {renderSubmit ? (
            renderSubmit({
              isSubmitting: externalSubmitting || isSubmitting,
              isValid,
              dirty,
            })
          ) : (
            <button
              type="submit"
              className={styles.submitBtn}
              disabled={externalSubmitting || isSubmitting}
            >
              {externalSubmitting || isSubmitting ? 'Submitting...' : submitLabel}
            </button>
          )}
        </div>
      )}
    </form>
  );
}

// ─── Section Renderer ─────────────────────────────────────────────

function SectionRenderer({
  section,
  renderField,
}: {
  section: FormSection;
  renderField: (field: FormField) => React.ReactNode;
}) {
  const [collapsed, setCollapsed] = useState(section.defaultCollapsed ?? false);

  return (
    <div className={styles.section}>
      {(section.title || section.collapsible) && (
        <div
          className={[
            styles.sectionHeader,
            section.collapsible && styles.sectionHeaderClickable,
          ]
            .filter(Boolean)
            .join(' ')}
          onClick={() => section.collapsible && setCollapsed(!collapsed)}
          role={section.collapsible ? 'button' : undefined}
          tabIndex={section.collapsible ? 0 : undefined}
          aria-expanded={section.collapsible ? !collapsed : undefined}
          onKeyDown={(e) => {
            if (section.collapsible && (e.key === 'Enter' || e.key === ' ')) {
              e.preventDefault();
              setCollapsed(!collapsed);
            }
          }}
        >
          {section.title && <h3 className={styles.sectionTitle}>{section.title}</h3>}
          {section.description && <p className={styles.sectionDesc}>{section.description}</p>}
          {section.collapsible && (
            <span className={styles.collapseIcon}>{collapsed ? '▶' : '▼'}</span>
          )}
        </div>
      )}
      {!collapsed && (
        <div className={styles.sectionFields}>
          {section.fields.map(renderField)}
        </div>
      )}
    </div>
  );
}

// ─── Named exports ────────────────────────────────────────────────

export { validateField, validateForm };
export default FormBuilder;
```
