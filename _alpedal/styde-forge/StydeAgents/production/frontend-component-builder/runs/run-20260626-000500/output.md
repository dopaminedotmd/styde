## 1. VirtualList

A high-performance virtual-scrolling list that renders only visible items. Handles 100K+ rows with smooth 60fps scrolling. Features dynamic row heights, scroll-to-index, and overscan tuning.

### `VirtualList.tsx`

```tsx
import React, {
  useCallback,
  useEffect,
  useLayoutEffect,
  useMemo,
  useRef,
  useState,
} from 'react';
import type { VirtualListProps, ScrollAlignment } from './VirtualList.types';
import styles from './VirtualList.module.css';

// ─── Types for internal tracking ────────────────────────────────────

interface MeasuredItem {
  index: number;
  size: number;
  offset: number;
}

interface Range {
  start: number;
  end: number;
}

// ─── Binary search for start index from scrollTop ───────────────────

function findStartIndex(
  offsets: number[],
  scrollTop: number,
  itemCount: number,
): number {
  if (itemCount === 0) return 0;
  if (scrollTop <= 0) return 0;

  let low = 0;
  let high = Math.min(itemCount, offsets.length) - 1;

  while (low <= high) {
    const mid = (low + high) >>> 1;
    const offset = offsets[mid] ?? 0;
    if (offset < scrollTop) {
      low = mid + 1;
    } else if (offset > scrollTop) {
      high = mid - 1;
    } else {
      return mid;
    }
  }
  return Math.max(0, high);
}

// ─── Main Component ─────────────────────────────────────────────────

function VirtualListInner<TItem>(
  {
    items,
    getKey = (_item: TItem, index: number) => index,
    children: renderRow,
    rowHeight: fixedRowHeight,
    estimatedRowHeight = 40,
    height = '100%',
    width = '100%',
    overscanCount = 3,
    scrollToIndex,
    scrollAlignment = 'auto',
    onVisibleRangeChange,
    onScroll,
    containerClassName,
    containerStyle,
    gap = 0,
    role = 'list',
    totalLabel,
    className,
    style,
    'data-testid': dataTestId,
    'aria-label': ariaLabel,
  }: VirtualListProps<TItem>,
) {
  // ── Refs ──────────────────────────────────────────────────────────
  const containerRef = useRef<HTMLDivElement>(null);
  const resizeObserverRef = useRef<ResizeObserver | null>(null);
  const rafRef = useRef<number | null>(null);

  // ── State ─────────────────────────────────────────────────────────
  const [scrollTop, setScrollTop] = useState(0);
  const [containerHeight, setContainerHeight] = useState(0);

  // Dynamic height tracking
  const sizesRef = useRef<Map<number, number>>(new Map());
  const [, forceRender] = useState(0);

  // ── Derived values ────────────────────────────────────────────────

  const isFixedHeight = fixedRowHeight !== undefined;
  const rowHeight = fixedRowHeight ?? estimatedRowHeight;

  // Build offset array
  const { offsets, totalHeight } = useMemo(() => {
    const offsetsArr: number[] = [];
    let offset = 0;

    for (let i = 0; i < items.length; i++) {
      offsetsArr[i] = offset;
      const h = isFixedHeight
        ? rowHeight
        : (sizesRef.current.get(i) ?? estimatedRowHeight);
      offset += h + gap;
    }

    return {
      offsets: offsetsArr,
      totalHeight: offset > 0 ? offset - gap : 0,
    };
  }, [items.length, rowHeight, estimatedRowHeight, gap, isFixedHeight, sizesRef.current.size]);

  // Visible range
  const range: Range = useMemo(() => {
    const startIdx = findStartIndex(offsets, scrollTop, items.length);
    const start = Math.max(0, startIdx - overscanCount);

    let currentOffset = offsets[start] ?? 0;
    let end = start;

    for (let i = start; i < items.length; i++) {
      const itemH = isFixedHeight
        ? rowHeight
        : (sizesRef.current.get(i) ?? estimatedRowHeight);
      currentOffset += itemH + gap;

      if (currentOffset >= scrollTop + containerHeight + (overscanCount * rowHeight)) {
        end = i + 1 + overscanCount;
        break;
      }
      end = i + 1;
    }

    return {
      start,
      end: Math.min(items.length, end),
    };
  }, [offsets, scrollTop, containerHeight, items.length, rowHeight, estimatedRowHeight, gap, overscanCount, isFixedHeight]);

  // Fire visible-range callback
  useLayoutEffect(() => {
    onVisibleRangeChange?.({ start: range.start, end: range.end });
  }, [range.start, range.end]);

  // ── Container height measurement ──────────────────────────────────

  useLayoutEffect(() => {
    const container = containerRef.current;
    if (!container) return;

    const updateHeight = () => {
      const rect = container.getBoundingClientRect();
      if (rect.height > 0) {
        setContainerHeight(rect.height);
      }
    };

    updateHeight();

    resizeObserverRef.current = new ResizeObserver(updateHeight);
    resizeObserverRef.current.observe(container);

    return () => {
      resizeObserverRef.current?.disconnect();
    };
  }, []);

  // ── Scroll handler (rAF throttled) ────────────────────────────────

  const handleScroll = useCallback(() => {
    if (rafRef.current) return;
    rafRef.current = requestAnimationFrame(() => {
      if (containerRef.current) {
        const newScrollTop = containerRef.current.scrollTop;
        setScrollTop(newScrollTop);
        onScroll?.(newScrollTop);
      }
      rafRef.current = null;
    });
  }, [onScroll]);

  // ── Scroll-to-index ───────────────────────────────────────────────

  const scrollToIdx = useCallback(
    (index: number, alignment: ScrollAlignment) => {
      const el = containerRef.current;
      if (!el || index < 0 || index >= items.length) return;

      const itemOffset = offsets[index] ?? 0;
      const itemH = isFixedHeight
        ? rowHeight
        : (sizesRef.current.get(index) ?? estimatedRowHeight);

      let targetScrollTop: number;

      switch (alignment) {
        case 'start':
          targetScrollTop = itemOffset;
          break;
        case 'center':
          targetScrollTop = itemOffset - (containerHeight - itemH) / 2;
          break;
        case 'end':
          targetScrollTop = itemOffset - containerHeight + itemH;
          break;
        case 'auto':
        default:
          // Only scroll if not already visible
          if (itemOffset >= scrollTop && itemOffset + itemH <= scrollTop + containerHeight) {
            return; // Already visible
          }
          targetScrollTop =
            itemOffset < scrollTop
              ? itemOffset // Scroll up to show it at top
              : itemOffset - containerHeight + itemH; // Scroll down to show at bottom
          break;
      }

      el.scrollTo({
        top: Math.max(0, targetScrollTop),
        behavior: 'smooth',
      });
    },
    [offsets, rowHeight, estimatedRowHeight, containerHeight, scrollTop, items.length, isFixedHeight],
  );

  useEffect(() => {
    if (scrollToIndex !== undefined) {
      scrollToIdx(scrollToIndex, scrollAlignment);
    }
  }, [scrollToIndex, scrollAlignment, scrollToIdx]);

  // ── Dynamic height measurement callback ───────────────────────────

  const createMeasureRef = useCallback(
    (index: number) => (node: HTMLElement | null) => {
      if (!node || isFixedHeight) return;

      const current = sizesRef.current.get(index) ?? 0;
      const measured = node.getBoundingClientRect().height;

      if (Math.abs(measured - current) > 1) {
        sizesRef.current.set(index, measured);
        forceRender((n) => n + 1);
      }
    },
    [isFixedHeight],
  );

  // ── Build visible rows ────────────────────────────────────────────

  const visibleRows = useMemo(() => {
    const rows: React.ReactNode[] = [];
    let currentOffset = offsets[range.start] ?? 0;

    for (let i = range.start; i < range.end; i++) {
      const item = items[i];
      const itemH = isFixedHeight
        ? rowHeight
        : (sizesRef.current.get(i) ?? estimatedRowHeight);

      const rowStyle: React.CSSProperties = {
        position: 'absolute',
        top: currentOffset,
        left: 0,
        right: 0,
        height: isFixedHeight ? rowHeight : undefined,
        minHeight: isFixedHeight ? undefined : itemH,
      };

      rows.push(
        <div
          key={getKey(item, i)}
          role={role === 'list' ? 'listitem' : undefined}
          style={rowStyle}
        >
          {renderRow({
            item,
            index: i,
            isVisible: true,
            style: { width: '100%', height: '100%' },
            measureRef: isFixedHeight ? undefined : createMeasureRef(i),
          })}
        </div>,
      );

      currentOffset += itemH + gap;
    }

    return rows;
  }, [
    range.start,
    range.end,
    items,
    offsets,
    rowHeight,
    estimatedRowHeight,
    gap,
    isFixedHeight,
    getKey,
    renderRow,
    createMeasureRef,
    role,
  ]);

  // ── Cleanup ───────────────────────────────────────────────────────

  useEffect(() => {
    return () => {
      if (rafRef.current) {
        cancelAnimationFrame(rafRef.current);
      }
    };
  }, []);

  // ── Render ────────────────────────────────────────────────────────

  const containerClasses = [styles.container, containerClassName, className]
    .filter(Boolean)
    .join(' ');

  const resolvedHeight =
    typeof height === 'number' ? `${height}px` : height;
  const resolvedWidth =
    typeof width === 'number' ? `${width}px` : width;

  const mergedStyle: React.CSSProperties = {
    ...style,
    ...containerStyle,
    height: resolvedHeight,
    width: resolvedWidth,
  };

  return (
    <div
      ref={containerRef}
      className={containerClasses}
      style={mergedStyle}
      onScroll={handleScroll}
      role={role}
      aria-label={ariaLabel ?? `List with ${items.length} items`}
      aria-rowcount={items.length}
      data-testid={dataTestId}
    >
      <div
        className={styles.inner}
        style={{ height: totalHeight }}
      >
        {visibleRows}
      </div>

      {items.length === 0 && (
        <div className={styles.placeholder} aria-live="polite">
          {totalLabel ?? 'No items to display'}
        </div>
      )}
    </div>
  );
}

// Memoize to prevent unnecessary re-renders from parent
export const VirtualList = React.memo(VirtualListInner) as typeof VirtualListInner;
export type { VirtualListProps, VirtualRowRenderProps, VirtualRowRenderer, ScrollAlignment };
```

### `DragAndDrop.tsx`

```tsx
import React, {
  useCallback,
  useEffect,
  useMemo,
  useRef,
  useState,
} from 'react';
import type { DragAndDropProps, ReorderResult } from './DragAndDrop.types';
import styles from './DragAndDrop.module.css';

// ─── Helper: clamp ──────────────────────────────────────────────────

function clamp(value: number, min: number, max: number): number {
  return Math.min(Math.max(value, min), max);
}

// ─── Helper: move item in array ─────────────────────────────────────

function moveInArray<T>(arr: T[], from: number, to: number): T[] {
  const result = [...arr];
  const [removed] = result.splice(from, 1);
  result.splice(to, 0, removed);
  return result;
}

// ─── Touch / Mouse utils ────────────────────────────────────────────

function getEventPosition(
  e: MouseEvent | TouchEvent | React.MouseEvent | React.TouchEvent,
): { x: number; y: number } {
  if ('touches' in e) {
    const touch = e.touches[0] || (e as TouchEvent).changedTouches[0];
    return { x: touch.clientX, y: touch.clientY };
  }
  return { x: (e as MouseEvent).clientX, y: (e as MouseEvent).clientY };
}

// ─── Main Component ─────────────────────────────────────────────────

function DragAndDropInner<TItem>(
  props: DragAndDropProps<TItem>,
) {
  const {
    items,
    getKey = (_item: TItem, idx: number) => idx,
    children: renderItem,
    onReorder,
    handle: dragHandle = 'auto',
    enabled = true,
    axis = 'vertical',
    lockAxis = true,
    dragThreshold = 5,
    listClassName,
    draggingClassName,
    overClassName,
    transitionDuration = 200,
    getAnnouncement,
    className,
    style,
    'data-testid': dataTestId,
    'aria-label': ariaLabel,
  } = props;

  // ── Refs ──────────────────────────────────────────────────────────
  const listRef = useRef<HTMLUListElement>(null);
  const itemRefs = useRef<Map<number, HTMLLIElement>>(new Map());

  // ── Drag state ────────────────────────────────────────────────────
  const [dragState, setDragState] = useState<{
    activeIndex: number;
    overIndex: number;
    startIndex: number;
    startPosition: { x: number; y: number };
    currentPosition: { x: number; y: number };
    offsets: number[];
    hasMoved: boolean;
  } | null>(null);

  // Keyboard reorder state
  const [keyboardActiveIndex, setKeyboardActiveIndex] = useState<number | null>(null);
  const [announcement, setAnnouncement] = useState('');

  // ── Calculate item offsets ────────────────────────────────────────

  const calculateOffsets = useCallback((): number[] => {
    const offsets: number[] = [];
    let offset = 0;
    const el = listRef.current;
    if (!el) return offsets;

    const children = Array.from(el.children) as HTMLElement[];
    for (let i = 0; i < children.length; i++) {
      offsets[i] = offset;
      const rect = children[i].getBoundingClientRect();
      offset += axis === 'vertical' ? rect.height : rect.width;
    }
    return offsets;
  }, [axis]);

  // ── Determine over index from position ────────────────────────────

  const getOverIndex = useCallback(
    (pos: { x: number; y: number }, offsets: number[]): number => {
      if (!listRef.current) return 0;

      const listRect = listRef.current.getBoundingClientRect();
      const point = axis === 'vertical'
        ? pos.y - listRect.top + listRef.current.scrollTop
        : pos.x - listRect.left + listRef.current.scrollLeft;

      // Find the index
      for (let i = offsets.length - 1; i >= 0; i--) {
        if (point >= (offsets[i] ?? 0)) {
          return i;
        }
      }
      return 0;
    },
    [axis],
  );

  // ── Mouse / Touch drag handlers ───────────────────────────────────

  const onPointerStart = useCallback(
    (index: number, e: React.MouseEvent | React.TouchEvent) => {
      if (!enabled) return;
      e.preventDefault();

      const initPos = getEventPosition(e);
      const offsets = calculateOffsets();

      setDragState({
        activeIndex: index,
        overIndex: index,
        startIndex: index,
        startPosition: { x: initPos.x, y: initPos.y },
        currentPosition: initPos,
        offsets,
        hasMoved: false,
      });
    },
    [enabled, calculateOffsets],
  );

  useEffect(() => {
    if (!dragState) return;

    const { activeIndex, startPosition } = dragState;

    const onPointerMove = (e: MouseEvent | TouchEvent) => {
      const pos = getEventPosition(e);
      const dx = pos.x - startPosition.x;
      const dy = pos.y - startPosition.y;
      const hasMoved = Math.abs(axis === 'vertical' ? dy : dx) >= dragThreshold;

      if (!dragState.hasMoved && !hasMoved) return;

      setDragState((prev) => {
        if (!prev) return null;
        const offsets = calculateOffsets();
        let newPos = pos;

        if (lockAxis && axis === 'vertical') {
          newPos = { ...pos, x: prev.startPosition.x };
        } else if (lockAxis && axis === 'horizontal') {
          newPos = { ...pos, y: prev.startPosition.y };
        }

        let overIndex = getOverIndex(newPos, offsets);
        overIndex = clamp(overIndex, 0, items.length - 1);

        return {
          ...prev,
          currentPosition: newPos,
          overIndex,
          offsets,
          hasMoved: true,
        };
      });
    };

    const onPointerUp = () => {
      setDragState((prev) => {
        if (!prev || !prev.hasMoved) return null;

        const { startIndex, overIndex } = prev;
        if (startIndex !== overIndex) {
          const newItems = moveInArray(items, startIndex, overIndex);
          const result: ReorderResult<TItem> = {
            items: newItems,
            fromIndex: startIndex,
            toIndex: overIndex,
            item: items[startIndex],
          };
          onReorder(result);
          setAnnouncement(
            getAnnouncement?.(result) ??
              `Moved item from position ${startIndex + 1} to ${overIndex + 1}`,
          );
        }
        return null;
      });
    };

    const onKeyDown = (e: KeyboardEvent) => {
      if (e.key === 'Escape') {
        setDragState(null);
      }
    };

    document.addEventListener('mousemove', onPointerMove);
    document.addEventListener('mouseup', onPointerUp);
    document.addEventListener('touchmove', onPointerMove, { passive: false });
    document.addEventListener('touchend', onPointerUp);
    document.addEventListener('keydown', onKeyDown);

    return () => {
      document.removeAllListeners?.('mousemove', onPointerMove);
      document.removeEventListener('mousemove', onPointerMove);
      document.removeEventListener('mouseup', onPointerUp);
      document.removeEventListener('touchmove', onPointerMove);
      document.removeEventListener('touchend', onPointerUp);
      document.removeEventListener('keydown', onKeyDown);
    };
  }, [dragState, axis, dragThreshold, lockAxis, items, onReorder, getAnnouncement, calculateOffsets, getOverIndex]);

  // ── Keyboard reorder ──────────────────────────────────────────────

  const handleKeyboard = useCallback(
    (index: number, e: React.KeyboardEvent) => {
      if (!enabled) return;

      const targetIndex =
        e.key === 'ArrowDown' || e.key === 'ArrowRight'
          ? Math.min(index + 1, items.length - 1)
          : e.key === 'ArrowUp' || e.key === 'ArrowLeft'
            ? Math.max(index - 1, 0)
            : null;

      if (e.key === ' ' || e.key === 'Enter') {
        // Toggle keyboard reorder mode
        setKeyboardActiveIndex((prev) => (prev === index ? null : index));
        e.preventDefault();
        return;
      }

      if (targetIndex !== null && keyboardActiveIndex === index) {
        e.preventDefault();
        const newItems = moveInArray(items, index, targetIndex);
        const result: ReorderResult<TItem> = {
          items: newItems,
          fromIndex: index,
          toIndex: targetIndex,
          item: items[index],
        };
        onReorder(result);
        setKeyboardActiveIndex(targetIndex);
        setAnnouncement(
          getAnnouncement?.(result) ??
            `Moved item from position ${index + 1} to ${targetIndex + 1}`,
        );

        // Focus the moved element
        setTimeout(() => {
          itemRefs.current.get(targetIndex)?.focus();
        }, 0);
      }
    },
    [enabled, items, keyboardActiveIndex, onReorder, getAnnouncement],
  );

  // ── Item positions during drag ────────────────────────────────────

  const itemStyles = useMemo(() => {
    const stylesMap = new Map<number, React.CSSProperties>();

    if (!dragState || !dragState.hasMoved) return stylesMap;

    const { activeIndex, overIndex, offsets } = dragState;

    for (let i = 0; i < items.length; i++) {
      if (i === activeIndex) continue;

      // Items between activeIndex and overIndex shift
      let translatePx: number | undefined;

      if (activeIndex < overIndex) {
        // Dragging down: items from activeIndex+1 to overIndex shift up
        if (i > activeIndex && i <= overIndex) {
          const itemHeight = (offsets[i + 1] ?? offsets[i]) - (offsets[i] ?? 0);
          translatePx = -(itemHeight || 40);
        }
      } else if (activeIndex > overIndex) {
        // Dragging up: items from overIndex to activeIndex-1 shift down
        if (i >= overIndex && i < activeIndex) {
          const activeHeight = (offsets[activeIndex + 1] ?? offsets[activeIndex]) - (offsets[activeIndex] ?? 0);
          translatePx = activeHeight || 40;
        }
      }

      if (translatePx !== undefined) {
        stylesMap.set(i, {
          transform: axis === 'vertical'
            ? `translateY(${translatePx}px)`
            : `translateX(${translatePx}px)`,
        });
      }
    }

    return stylesMap;
  }, [dragState, items.length, axis]);

  // ── Active item drag style ────────────────────────────────────────

  const activeStyle: React.CSSProperties | undefined = useMemo(() => {
    if (!dragState || !dragState.hasMoved) return undefined;

    const { activeIndex, currentPosition, startPosition, offsets } = dragState;
    const baseOffset = offsets[activeIndex] ?? 0;

    const dx = currentPosition.x - startPosition.x;
    const dy = currentPosition.y - startPosition.y;

    return {
      position: 'relative',
      zIndex: 1000,
      transform: axis === 'vertical'
        ? `translateY(${dy}px)`
        : `translateX(${dx}px)`,
      pointerEvents: 'none',
      opacity: 0.85,
    };
  }, [dragState, axis]);

  // ── Render ────────────────────────────────────────────────────────

  const listClasses = [styles.list, listClassName, className]
    .filter(Boolean)
    .join(' ');

  return (
    <>
      {/* Screen-reader announcements */}
      <div
        aria-live="assertive"
        aria-atomic="true"
        style={{
          position: 'absolute',
          width: 1,
          height: 1,
          margin: -1,
          padding: 0,
          overflow: 'hidden',
          clip: 'rect(0, 0, 0, 0)',
          whiteSpace: 'nowrap',
          border: 0,
        }}
      >
        {announcement}
      </div>

      <ul
        ref={listRef}
        className={listClasses}
        style={{
          '--dd-transition': `${transitionDuration}ms`,
          ...style,
        } as React.CSSProperties}
        role="listbox"
        aria-label={ariaLabel ?? 'Sortable list'}
        data-testid={dataTestId}
      >
        {items.map((item, index) => {
          const isDragging = dragState?.activeIndex === index && dragState.hasMoved;
          const isOver = dragState?.overIndex === index && !isDragging;

          const itemClass = [
            styles.item,
            isDragging && (draggingClassName ?? styles.dragging),
            isOver && (overClassName ?? ''),
          ]
            .filter(Boolean)
            .join(' ');

          const dragStyle: React.CSSProperties = {
            ...(isDragging ? activeStyle : itemStyles.get(index) ?? {}),
          };

          const handleProps = {
            onMouseDown: (e: React.MouseEvent) => onPointerStart(index, e),
            onTouchStart: (e: React.TouchEvent) => onPointerStart(index, e),
            onKeyDown: (e: React.KeyboardEvent) => handleKeyboard(index, e),
            tabIndex: 0,
            role: 'option' as const,
            'aria-label': `Item ${index + 1} of ${items.length}. Press Space to reorder with arrow keys.`,
            'aria-grabbed': isDragging,
          };

          const dragRenderProps = {
            isDragging,
            isOver,
            isFocusVisible: keyboardActiveIndex === index,
            handleProps,
          };

          return (
            <li
              key={getKey(item, index)}
              ref={(el) => {
                if (el) itemRefs.current.set(index, el);
                else itemRefs.current.delete(index);
              }}
              className={itemClass}
              style={dragStyle}
              aria-selected={isDragging}
              aria-setsize={items.length}
              aria-posinset={index + 1}
              {...(dragHandle === 'auto' ? handleProps : {})}
            >
              {dragHandle === 'manual' && (
                <span className={styles.handle} {...handleProps} aria-hidden={false}>
                  ⠿
                </span>
              )}
              {renderItem({ item, index, drag: dragRenderProps })}
            </li>
          );
        })}
      </ul>
    </>
  );
}

export const DragAndDrop = React.memo(DragAndDropInner) as typeof DragAndDropInner;
export type { DragAndDropProps, DragDropRenderProps, DragItemRenderProps, ReorderResult, DragHandle };
```

```tsx
  underline: styles.iconUnderline,

// ─── Get word count ─────────────────────────────────────────────────

function countWords(text: string): number {

  return trimmed ? trimmed.split(/\s+/).length : 0;
}

// ─── Main Component ─────────────────────────────────────────────────

function RichTextEditorInner(
  {
    value: controlledValue,
    defaultValue = '',
    onChange,
    placeholder = 'Start writing...',
    readOnly = false,
    minHeight = 120,
    maxHeight,
    toolbar: showToolbar = true,
    toolbarGroups,
    renderToolbar,
    spellCheck = true,
    autoFocus = false,
    debounceMs = 300,
    label,
    error,
    hint,
    className,
    style,
    'data-testid': dataTestId,
    'aria-label': ariaLabel,
  }: RichTextEditorProps,
) {
  const editorRef = useRef<HTMLDivElement>(null);
  const debounceRef = useRef<ReturnType<typeof setTimeout> | null>(null);
  const lastHtmlRef = useRef(defaultValue);

  const [html, setHtml] = useState(controlledValue ?? defaultValue);
  const [isFocused, setIsFocused] = useState(false);

  const isControlled = controlledValue !== undefined;

  // Sync controlled value
  useEffect(() => {
    if (isControlled && controlledValue !== html && editorRef.current) {
      setHtml(controlledValue);
      if (editorRef.current.innerHTML !== controlledValue) {
        editorRef.current.innerHTML = controlledValue;
      }
      lastHtmlRef.current = controlledValue;
    }
  }, [controlledValue, isControlled]);

  // ── Derived state ─────────────────────────────────────────────────

  const text = useMemo(() => {
    const div = document.createElement('div');
    div.innerHTML = html;
    return div.textContent || '';
  }, [html]);

  const wordCount = useMemo(() => countWords(text), [text]);
  const charCount = text.length;
  const isDirty = html !== (controlledValue ?? defaultValue);
  const isEmpty = text.trim().length === 0;

  // ── Query active formats ──────────────────────────────────────────

  const getActiveFormats = useCallback((): Set<EditorAction> => {
    const active = new Set<EditorAction>();
    if (typeof document === 'undefined') return active;

    for (const [action, cmd] of Object.entries(QUERY_COMMANDS)) {
      try {
        if (document.queryCommandState(cmd)) {
          active.add(action as EditorAction);
        }
      } catch {
        // Ignore unsupported commands
      }
    }

    // Check for heading/block tags
    const sel = document.getSelection();
    if (sel && sel.rangeCount > 0) {
      let node: Node | null = sel.anchorNode;
      // Walk up to find block-level element
      while (node && node !== editorRef.current) {
        if (node.nodeType === 1) {
          const el = node as HTMLElement;
          const tag = el.tagName?.toLowerCase();
          if (tag === 'h1') active.add('heading1');
          if (tag === 'h2') active.add('heading2');
          if (tag === 'h3') active.add('heading3');
          if (tag === 'blockquote') active.add('blockquote');
          if (tag === 'code' && el.parentElement?.closest('pre') === null) active.add('code');
        }
        node = node.parentNode;
      }
    }

    return active;
  }, []);

  const [activeFormats, setActiveFormats] = useState<Set<EditorAction>>(new Set());

  // ── Update formats on selection change ────────────────────────────

  useEffect(() => {
    const updateFormats = () => setActiveFormats(getActiveFormats());
    document.addEventListener('selectionchange', updateFormats);
    return () => document.removeEventListener('selectionchange', updateFormats);
  }, [getActiveFormats]);

  // ── Execute an action ─────────────────────────────────────────────

  const exec = useCallback(
    (action: EditorAction, value?: string) => {
      const ed = editorRef.current;
      if (!ed) return;

      ed.focus();

      if (action === 'link') {
        const url = value ?? window.prompt('Enter URL:');
        if (url) {
          document.execCommand('createLink', false, url);
        }
        flushChange();
        return;
      }

      if (action === 'code') {
        // Toggle inline code: wrap selection in <code> or unwrap
        const sel = document.getSelection();
        if (sel && sel.rangeCount > 0 && !sel.isCollapsed) {
          const range = sel.getRangeAt(0);
          const codeEl = document.createElement('code');
          try {
            range.surroundContents(codeEl);
          } catch {
            // If surround fails (partial selection of block), toggle via execCommand
          }
        }
        flushChange();
        return;
      }

      const cmd = ACTION_COMMANDS[action];
      if (cmd) {
        const tagValue = ACTION_TAG[action];
        if (tagValue) {
          // For block-level formats, check if already applied and toggle
          const currentBlock = getActiveFormats();
          if (currentBlock.has(action)) {
            document.execCommand('formatBlock', false, '<p>');
          } else {
            document.execCommand(cmd, false, tagValue);
          }
        } else {
          document.execCommand(cmd);
        }
        flushChange();
      }
    },
    [getActiveFormats],
  );

  // ── Flush changes ─────────────────────────────────────────────────

  const flushChange = useCallback(() => {
    if (debounceRef.current) {
      clearTimeout(debounceRef.current);
    }

    debounceRef.current = setTimeout(() => {
      const ed = editorRef.current;
      if (!ed) return;

      const currentHtml = ed.innerHTML;

      // Skip if unchanged
      if (currentHtml === lastHtmlRef.current) return;
      lastHtmlRef.current = currentHtml;

      if (!isControlled) {
        setHtml(currentHtml);
      }

      const div = document.createElement('div');
      div.innerHTML = currentHtml;
      const currentText = div.textContent || '';

      const state: EditorState = {
        html: currentHtml,
        text: currentText,
        isFocused,
        activeFormats: getActiveFormats(),
        isDirty: currentHtml !== (controlledValue ?? defaultValue),
        wordCount: countWords(currentText),
        charCount: currentText.length,
      };

      onChange?.(state);
    }, debounceMs);
  }, [isControlled, isFocused, controlledValue, defaultValue, onChange, debounceMs, getActiveFormats]);

  // ── Input handler ─────────────────────────────────────────────────

  const handleInput = useCallback(() => {
    flushChange();
  }, [flushChange]);

  // ── Paste handler (sanitize) ──────────────────────────────────────

  const handlePaste = useCallback(
    (e: React.ClipboardEvent) => {
      e.preventDefault();
      const text = e.clipboardData.getData('text/plain');
      // Insert as plain text
      document.execCommand('insertText', false, text);
      flushChange();
    },
    [flushChange],
  );

  // ── Focus / Blur ──────────────────────────────────────────────────

  const handleFocus = useCallback(() => setIsFocused(true), []);
  const handleBlur = useCallback(() => {
    setIsFocused(false);
    flushChange();
  }, [flushChange]);

  // ── Editor state ──────────────────────────────────────────────────

  const editorState: EditorState = useMemo(
    () => ({
      html,
      text,
      isFocused,
      activeFormats,
      isDirty,
      wordCount,
      charCount,
    }),
    [html, text, isFocused, activeFormats, isDirty, wordCount, charCount],
  );

  // ── Toolbar render props ──────────────────────────────────────────

  const toolbarProps: ToolbarRenderProps = useMemo(
    () => ({
      exec,
      isActive: (action: EditorAction) => activeFormats.has(action),
      editorState,
      undoDepth: 0, // Browser manages history
      redoDepth: 0,
    }),
    [exec, activeFormats, editorState],
  );

  // ── Auto-focus ────────────────────────────────────────────────────

  useEffect(() => {
    if (autoFocus && editorRef.current) {
      editorRef.current.focus();
    }
  }, [autoFocus]);

  // Cleanup
  useEffect(() => {
    return () => {
      if (debounceRef.current) clearTimeout(debounceRef.current);
    };
  }, []);

  // ── Render ────────────────────────────────────────────────────────

  const wrapperClasses = [
    styles.wrapper,
    error && styles.wrapperError,
    readOnly && styles.readOnly,
    className,
  ]
    .filter(Boolean)
    .join(' ');

  const editorWrapperStyle: React.CSSProperties = {
    '--rte-min-height':
      typeof minHeight === 'number' ? `${minHeight}px` : minHeight,
    '--rte-max-height':
      maxHeight !== undefined
        ? typeof maxHeight === 'number'
          ? `${maxHeight}px`
          : maxHeight
        : 'none',
  } as React.CSSProperties;

  const groups = toolbarGroups ?? DEFAULT_TOOLBAR;

  return (
    <div data-testid={dataTestId}>
      {label && (
        <label className={styles.label} htmlFor={ariaLabel ? undefined : undefined}>
          {label}
        </label>
      )}

      <div className={wrapperClasses} style={style}>
        {/* Toolbar */}
        {showToolbar && groups.length > 0 && (
          <div className={styles.toolbar} role="toolbar" aria-label="Formatting toolbar">
            {renderToolbar
              ? renderToolbar(toolbarProps)
              : groups.map((group: ToolbarGroup, gi: number) => (
                  <React.Fragment key={group.key}>
                    {gi > 0 && <div className={styles.toolbarDivider} />}
                    <div className={styles.toolbarGroup} role="group" aria-label={group.label}>
                      {group.actions.map((action: EditorAction) => {
                        const isActive = activeFormats.has(action);
                        return (
                          <button
                            key={action}
                            type="button"
                            className={[
                              styles.toolbarButton,
                              isActive && styles.toolbarButtonActive,
                              ACTION_ICONS[action] ?? '',
                            ]
                              .filter(Boolean)
                              .join(' ')}
                            onClick={() => exec(action)}
                            aria-label={ACTION_LABELS[action]}
                            aria-pressed={isActive}
                            title={ACTION_LABELS[action]}
                            tabIndex={0}
                          />
                        );
                      })}
                    </div>
                  </React.Fragment>
                ))}
          </div>
        )}

        {/* Editor area */}
        <div
          ref={editorRef}
          className={`${styles.editor} ${isEmpty ? 'empty' : ''}`}
          contentEditable={!readOnly}
          suppressContentEditableWarning
          dangerouslySetInnerHTML={{ __html: html }}
          onInput={handleInput}
          onPaste={handlePaste}
          onFocus={handleFocus}
          onBlur={handleBlur}
          spellCheck={spellCheck}
          data-placeholder={placeholder}
          role="textbox"
          aria-multiline="true"
          aria-label={ariaLabel ?? label ?? 'Rich text editor'}
          aria-readonly={readOnly}
          style={editorWrapperStyle}
        />

        {/* Footer */}
        <div className={styles.footer}>
          <span>{charCount} chars</span>
          <span>{wordCount} words</span>
        </div>
      </div>

      {error && <p className={styles.error} role="alert">{error}</p>}
      {hint && !error && <p className={styles.hint}>{hint}</p>}
    </div>
  );
}
```

```tsx
// ─── Date utilities ─────────────────────────────────────────────────

function startOfDay(d: Date): Date {
  return new Date(d.getFullYear(), d.getMonth(), d.getDate());
}

function addDays(d: Date, days: number): Date {
  const result = new Date(d);
  result.setDate(result.getDate() + days);
  return result;
}

function addMonths(d: Date, months: number): Date {
  const result = new Date(d);
  result.setMonth(result.getMonth() + months);
  return result;
}

function isSameDay(a: Date | null, b: Date | null): boolean {
  if (!a || !b) return false;
  return (
    a.getFullYear() === b.getFullYear() &&
    a.getMonth() === b.getMonth() &&
    a.getDate() === b.getDate()
  );
}

function isBetween(day: Date, start: Date | null, end: Date | null): boolean {
  if (!start || !end) return false;
  const d = day.getTime();
  return d > start.getTime() && d < end.getTime();
}

function toISODate(d: Date): string {
  const y = d.getFullYear();
  const m = String(d.getMonth() + 1).padStart(2, '0');
  const day = String(d.getDate()).padStart(2, '0');
  return `${y}-${m}-${day}`;
}

function parseISODate(s: string | null): Date | null {
  if (!s) return null;
  const d = new Date(s + 'T00:00:00');
  return isNaN(d.getTime()) ? null : d;
}

// ─── Build calendar grid ────────────────────────────────────────────

function buildMonthGrid(
  year: number,
  month: number,
  range: DateRange | null,
  hoverDate: Date | null,
  today: Date,
  firstDayOfWeek: number,
  minDate: Date | null,
  maxDate: Date | null,
  disableWeekends: boolean,
  isDateDisabled?: (date: Date) => boolean,
): DayCell[] {
  const cells: DayCell[] = [];
  const firstOfMonth = new Date(year, month, 1);
  const lastOfMonth = new Date(year, month + 1, 0);

  // Start from the configured first day of week
  let current = new Date(firstOfMonth);
  const dayOfWeek = current.getDay();
  const offset = (dayOfWeek - firstDayOfWeek + 7) % 7;
  current.setDate(current.getDate() - offset);

  const rangeStart = parseISODate(range?.start ?? null);
  const rangeEnd = parseISODate(range?.end ?? null);
  const hoverEnd = hoverDate;

  for (let i = 0; i < 42; i++) {
    const isCurrentMonth =
      current.getMonth() === month && current.getFullYear() === year;
    const isWeekend = current.getDay() === 0 || current.getDay() === 6;

    // Compute range states
    const isSelected =
      isSameDay(current, rangeStart) || isSameDay(current, rangeEnd);
    const isRangeStart = isSameDay(current, rangeStart);
    const isRangeEnd = isSameDay(current, rangeEnd);

    // If we have a start but no end, and user is hovering, show preview range
    const effectiveEnd = hoverEnd && rangeStart && !rangeEnd ? hoverEnd : rangeEnd;
    const isInRange = isBetween(current, rangeStart, effectiveEnd);

    // Disabled logic
    let isDisabled = false;
    if (minDate && current < startOfDay(minDate)) isDisabled = true;
    if (maxDate && current > startOfDay(maxDate)) isDisabled = true;
    if (disableWeekends && isWeekend) isDisabled = true;
    if (isDateDisabled?.(current)) isDisabled = true;

    cells.push({
      date: new Date(current),
      dayOfMonth: current.getDate(),
      isToday: isSameDay(current, today),
      isCurrentMonth,
      isWeekend,
      isDisabled,
      isSelected,
      isInRange,
      isRangeStart,
      isRangeEnd,
    });

    current.setDate(current.getDate() + 1);
  }

  return cells;
}

function defaultFormatDisplay(range: DateRange): string {
  if (range.start && range.end) {
    if (range.start === range.end) return range.start;
    return `${range.start} — ${range.end}`;
  }
  if (range.start) return `${range.start} — ...`;
  return '';
}

// ─── Main Component ─────────────────────────────────────────────────

function DateRangePickerInner({
  value: controlledValue,
  defaultValue,
  onChange,
  minDate,
  maxDate,
  disableWeekends = false,
  isDateDisabled,
  presets: showPresets = true,
  customPresets,
  placeholder = 'Select date range...',
  clearable = true,
  formatDisplay = defaultFormatDisplay,
  monthsToShow = 2,
  disabled = false,
  readOnly = false,
  closeOnSelect = true,
  locale = 'en-US',
  firstDayOfWeek = 0,
  weekdayLabels,
  monthLabels,
  label,
  error,
  className,
  style,
  'data-testid': dataTestId,
  'aria-label': ariaLabel,
}: DateRangePickerProps) {
  const containerRef = useRef<HTMLDivElement>(null);

  // ── State ─────────────────────────────────────────────────────────
  const isControlled = controlledValue !== undefined;
  const [internalValue, setInternalValue] = useState<DateRange>(
    defaultValue ?? { start: null, end: null },
  );
  const range = isControlled ? (controlledValue ?? { start: null, end: null }) : internalValue;

  const [isOpen, setIsOpen] = useState(false);
  const [hoverDate, setHoverDate] = useState<Date | null>(null);
  const [selectionPhase, setSelectionPhase] = useState<'start' | 'end'>(
    range.start && range.end ? 'start' : range.start ? 'end' : 'start',
  );

  // Current viewing months
  const [viewMonths, setViewMonths] = useState<CalendarMonth[]>(() => {
    const today = new Date();
    const months: CalendarMonth[] = [];
    for (let i = 0; i < monthsToShow; i++) {
      months.push({
        year: today.getFullYear(),
        month: today.getMonth() + i,
      });
    }
    return months;
  });

  // ── Derived values ────────────────────────────────────────────────

  const today = useMemo(() => startOfDay(new Date()), []);
  const presets = customPresets ?? (showPresets ? DEFAULT_PRESETS : []);
  const weekdays = weekdayLabels ?? getDefaultWeekdays(firstDayOfWeek, locale);
  const monthNames = monthLabels ?? getDefaultMonths(locale);

  // ── Actions ───────────────────────────────────────────────────────

  const updateRange = useCallback(
    (newRange: DateRange) => {
      if (!isControlled) {
        setInternalValue(newRange);
      }
      onChange?.(newRange);
    },
    [isControlled, onChange],
  );

  const selectDay = useCallback(
    (day: Date) => {
      const isoDay = toISODate(day);

      if (selectionPhase === 'start') {
        setSelectionPhase('end');
        updateRange({ start: isoDay, end: null });
      } else {
        const rangeStart = parseISODate(range.start);
        const selectedDay = parseISODate(isoDay);

        // If selected end is before start, swap
        if (rangeStart && selectedDay && selectedDay < rangeStart) {
          updateRange({ start: isoDay, end: range.start });
        } else {
          updateRange({ start: range.start, end: isoDay });
        }

        setSelectionPhase('start');

        if (closeOnSelect) {
          setIsOpen(false);
        }
      }
    },
    [selectionPhase, range.start, closeOnSelect, updateRange],
  );

  const selectPreset = useCallback(
    (preset: DateRangePreset) => {
      const newRange = preset.value();
      updateRange(newRange);
      if (closeOnSelect) setIsOpen(false);
    },
    [closeOnSelect, updateRange],
  );

  const clearRange = useCallback(() => {
    setSelectionPhase('start');
    updateRange({ start: null, end: null });
  }, [updateRange]);

  const navigateMonth = useCallback(
    (calendarIndex: number, direction: -1 | 1) => {
      setViewMonths((prev) =>
        prev.map((m, i) =>
          i === calendarIndex
            ? {
                year: m.month + direction > 11
                  ? m.year + 1
                  : m.month + direction < 0
                    ? m.year - 1
                    : m.year,
                month: (m.month + direction + 12) % 12,
              }
            : m,
        ),
      );
    },
    [],
  );

  // ── Close on outside click ────────────────────────────────────────

  useEffect(() => {
    if (!isOpen) return;

    const handleClickOutside = (e: MouseEvent) => {
      if (containerRef.current && !containerRef.current.contains(e.target as Node)) {
        setIsOpen(false);
      }
    };

    const handleEscape = (e: KeyboardEvent) => {
      if (e.key === 'Escape') setIsOpen(false);
    };

    document.addEventListener('mousedown', handleClickOutside);
    document.addEventListener('keydown', handleEscape);
    return () => {
      document.removeEventListener('mousedown', handleClickOutside);
      document.removeEventListener('keydown', handleEscape);
    };
  }, [isOpen]);

  // ── Keyboard navigation inside popover ────────────────────────────

  const handlePopoverKeyDown = useCallback(
    (e: React.KeyboardEvent) => {
      if (e.key === 'Escape') {
        setIsOpen(false);
        e.stopPropagation();
      }
    },
    [],
  );

  // ── Render ────────────────────────────────────────────────────────

  const displayText = formatDisplay(range);
  const isEmpty = !range.start && !range.end;

  const inputClasses = [
    styles.inputWrapper,
    isOpen && styles.inputWrapperOpen,
    disabled && styles.inputWrapperDisabled,
    error && styles.inputWrapperError,
  ]
    .filter(Boolean)
    .join(' ');

  return (
    <div
      ref={containerRef}
      className={`${styles.container} ${className ?? ''}`}
      style={style}
      data-testid={dataTestId}
    >
      {label && <label className={styles.label}>{label}</label>}

      {/* Trigger */}
      <div
        className={inputClasses}
        onClick={() => !disabled && !readOnly && setIsOpen(!isOpen)}
        role="combobox"
        aria-expanded={isOpen}
        aria-haspopup="dialog"
        aria-label={ariaLabel ?? label ?? 'Date range picker'}
        tabIndex={disabled ? -1 : 0}
        onKeyDown={(e) => {
          if (e.key === 'Enter' || e.key === ' ') {
            e.preventDefault();
            if (!disabled && !readOnly) setIsOpen(!isOpen);
          }
        }}
      >
        <span className={styles.calendarIcon} aria-hidden="true">📅</span>
        <span className={`${styles.inputValue} ${isEmpty ? styles.inputPlaceholder : ''}`}>
          {isEmpty ? placeholder : displayText}
        </span>
        {clearable && !isEmpty && (
          <button
            type="button"
            className={styles.clearButton}
            onClick={(e) => {
              e.stopPropagation();
              clearRange();
            }}
            aria-label="Clear date range"
          >
            ✕
          </button>
        )}
      </div>

      {/* Popover */}
      {isOpen && (
        <div
          className={styles.popover}
          role="dialog"
          aria-label="Date range picker dialog"
          onKeyDown={handlePopoverKeyDown}
        >
          {/* Presets */}
          {presets.length > 0 && (
            <div className={styles.presets}>
              {presets.map((preset) => {
                const presetVal = preset.value();
                const isActive =
                  range.start === presetVal.start && range.end === presetVal.end;
                return (
                  <button
                    key={preset.label}
                    type="button"
                    className={`${styles.presetButton} ${isActive ? styles.presetButtonActive : ''}`}
                    onClick={() => selectPreset(preset)}
                  >
                    {preset.label}
                  </button>
                );
              })}
            </div>
          )}

          {/* Calendars */}
          <div className={styles.calendarsRow}>
            {viewMonths.map((cm, ci) => {
              const grid = buildMonthGrid(
                cm.year,
                cm.month,
                range,
                hoverDate,
                today,
                firstDayOfWeek,
                minDate ?? null,
                maxDate ?? null,
                disableWeekends,
                isDateDisabled,
              );

              const isFirst = ci === 0;
              const isLast = ci === viewMonths.length - 1;

              return (
                <div key={`${cm.year}-${cm.month}`} className={styles.calendar}>
                  {/* Month header */}
                  <div className={styles.monthHeader}>
                    {isFirst && (
                      <button
                        type="button"
                        className={styles.navButton}
                        onClick={() => navigateMonth(ci, -1)}
                        aria-label="Previous month"
                      >
                        ‹
                      </button>
                    )}
                    <span className={styles.monthLabel}>
                      {monthNames[cm.month]} {cm.year}
                    </span>
                    {isLast && (
                      <button
                        type="button"
                        className={styles.navButton}
                        onClick={() => navigateMonth(ci, 1)}
                        aria-label="Next month"
                      >
                        ›
                      </button>
                    )}
                  </div>

                  {/* Weekday headers */}
                  <div className={styles.weekdayRow}>
                    {weekdays.map((wd) => (
                      <div key={wd} className={styles.weekday}>
                        {wd}
                      </div>
                    ))}
                  </div>

                  {/* Day grid */}
                  <div className={styles.dayGrid}>
                    {grid.map((cell, idx) => {
                      const cellClasses = [
                        styles.day,
                        !cell.isCurrentMonth && styles.dayOutsideMonth,
                        cell.isToday && styles.dayToday,
                        cell.isDisabled && styles.dayDisabled,
                        cell.isSelected && styles.daySelected,
                        cell.isInRange && !cell.isSelected && styles.dayInRange,
                        cell.isRangeStart && !cell.isRangeEnd && styles.dayRangeStart,
                        cell.isRangeEnd && !cell.isRangeStart && styles.dayRangeEnd,
                        cell.isRangeStart && cell.isRangeEnd && cell.isSelected && styles.daySelected,
                      ]
                        .filter(Boolean)
                        .join(' ');

                      return (
                        <button
                          key={idx}
                          type="button"
                          className={cellClasses}
                          disabled={cell.isDisabled || !cell.isCurrentMonth}
                          onClick={() => {
                            if (cell.isCurrentMonth && !cell.isDisabled) {
                              selectDay(cell.date);
                            }
                          }}
                          onMouseEnter={() => {
                            if (selectionPhase === 'end' && cell.isCurrentMonth && !cell.isDisabled) {
                              setHoverDate(cell.date);
                            }
                          }}
                          onMouseLeave={() => setHoverDate(null)}
                          aria-label={`${monthNames[cell.date.getMonth()]} ${cell.dayOfMonth}, ${cell.date.getFullYear()}`}
                          aria-selected={cell.isSelected}
                          aria-disabled={cell.isDisabled}
                          tabIndex={cell.isCurrentMonth && !cell.isDisabled ? 0 : -1}
                        >
                          {cell.dayOfMonth}
                        </button>
                      );
                    })}
                  </div>
                </div>
              );
            })}
          </div>
        </div>
      )}

      {error && (
        <p className={styles.error} role="alert">
          {error}
        </p>
      )}
    </div>
  );
}

export const DateRangePicker = React.memo(DateRangePickerInner) as typeof DateRangePickerInner;
export type { DateRangePickerProps, DateRange, DateRangePreset, DayCell, CalendarMonth };
```

### `BreadcrumbNav.tsx`

```tsx
import React, { useCallback, useEffect, useRef, useState } from 'react';
import type { BreadcrumbNavProps, BreadcrumbItem } from './BreadcrumbNav.types';
import styles from './BreadcrumbNav.module.css';

// ─── Default separator ──────────────────────────────────────────────

function DefaultSeparator() {
  return <span className={styles.separator} aria-hidden="true">/</span>;
}

// ─── Main Component ─────────────────────────────────────────────────

function BreadcrumbNavInner({
  items,
  maxVisibleItems = 4,
  collapseMode = 'auto',
  separator,
  renderCollapsedTrigger,
  renderItem,
  currentClickable = false,
  onItemClick,
  navLabel = 'Breadcrumb',
  className,
  style,
  'data-testid': dataTestId,
  'aria-label': ariaLabel,
}: BreadcrumbNavProps) {
  const containerRef = useRef<HTMLDivElement>(null);

  // ── Collapsed dropdown state ─────────────────────────────────────
  const [dropdownOpen, setDropdownOpen] = useState(false);
  const dropdownRef = useRef<HTMLDivElement>(null);

  // ── Responsive collapse state ─────────────────────────────────────
  const [overflowCount, setOverflowCount] = useState(0);

  const needsCollapse = collapseMode === 'auto' && items.length > maxVisibleItems;

  // Detect overflow via ResizeObserver
  useEffect(() => {
    if (!needsCollapse || !containerRef.current) return;

    const checkOverflow = () => {
      const container = containerRef.current;
      if (!container) return;

      // Simple approach: if items exceed maxVisibleItems, collapse middle ones
      if (items.length > maxVisibleItems) {
        // Keep first item, last item, and collapse middle
        setOverflowCount(items.length - maxVisibleItems);
      } else {
        setOverflowCount(0);
      }
    };

    const observer = new ResizeObserver(checkOverflow);
    observer.observe(containerRef.current);
    checkOverflow();

    return () => observer.disconnect();
  }, [needsCollapse, items.length, maxVisibleItems]);

  // ── Close dropdown on outside click ───────────────────────────────

  useEffect(() => {
    if (!dropdownOpen) return;

    const closeDropdown = (e: MouseEvent) => {
      if (dropdownRef.current && !dropdownRef.current.contains(e.target as Node)) {
        setDropdownOpen(false);
      }
    };

    const handleEscape = (e: KeyboardEvent) => {
      if (e.key === 'Escape') setDropdownOpen(false);
    };

    document.addEventListener('mousedown', closeDropdown);
    document.addEventListener('keydown', handleEscape);
    return () => {
      document.removeEventListener('mousedown', closeDropdown);
      document.removeEventListener('keydown', handleEscape);
    };
  }, [dropdownOpen]);

  // ── Build visible items ───────────────────────────────────────────

  const { visibleItems, collapsedItems } = React.useMemo(() => {
    if (!needsCollapse || overflowCount === 0) {
      return { visibleItems: items, collapsedItems: [] as BreadcrumbItem[] };
    }

    // Strategy: first item + last (maxVisibleItems - 1) items visible
    const first = items[0];
    const keepEnd = maxVisibleItems - 1; // -1 for the ellipsis slot
    const endItems = items.slice(items.length - keepEnd);

    const collapsedStart = 1;
    const collapsedEnd = items.length - keepEnd;
    const collapsed = items.slice(collapsedStart, collapsedEnd);

    return {
      visibleItems: [first, ...endItems],
      collapsedItems: collapsed,
    };
  }, [items, needsCollapse, overflowCount, maxVisibleItems]);

  const Sep = separator ? (
    <span className={styles.separator} aria-hidden="true">
      {separator}
    </span>
  ) : (
    <DefaultSeparator />
  );

  const handleItemClick = useCallback(
    (item: BreadcrumbItem) => {
      if (item.disabled) return;
      onItemClick?.(item);
      item.onClick?.();
    },
    [onItemClick],
  );

  // ── Render single item ────────────────────────────────────────────

  const defaultRenderItem = useCallback(
    (item: BreadcrumbItem, idx: number) => {
      const isLast = idx === items.indexOf(item) && item === items[items.length - 1];
      const isFirst = idx === 0;

      if (renderItem) {
        return renderItem({
          item,
          isFirst,
          isLast,
          isCollapsed: false,
        });
      }

      const isCurrent = item.isCurrent ?? isLast;
      const clickable = isCurrent ? currentClickable : true;
      const disabled = item.disabled ?? false;

      if (isCurrent && !clickable) {
        return (
          <span
            className={`${styles.current} ${item.className ?? ''}`}
            aria-current="page"
          >
            {item.icon && <span className={styles.icon}>{item.icon}</span>}
            {item.label}
          </span>
        );
      }

      if (item.href && !disabled) {
        return (
          <a
            href={item.href}
            className={`${styles.link} ${item.className ?? ''}`}
            onClick={(e) => {
              if (item.onClick) {
                e.preventDefault();
                handleItemClick(item);
              }
            }}
            aria-current={isCurrent ? 'page' : undefined}
          >
            {item.icon && <span className={styles.icon}>{item.icon}</span>}
            {item.label}
          </a>
        );
      }

      return (
        <button
          type="button"
          className={`${styles.link} ${disabled ? styles.linkDisabled : ''} ${item.className ?? ''}`}
          onClick={() => handleItemClick(item)}
          disabled={disabled}
          aria-current={isCurrent ? 'page' : undefined}
        >
          {item.icon && <span className={styles.icon}>{item.icon}</span>}
          {item.label}
        </button>
      );
    },
    [renderItem, currentClickable, handleItemClick],
  );

  // ── Render ────────────────────────────────────────────────────────

  const navClasses = [
    styles.nav,
    collapseMode === 'wrap' && styles.wrap,
    className,
  ]
    .filter(Boolean)
    .join(' ');

  const listClasses = [
    styles.list,
    collapseMode === 'wrap' && styles.listWrap,
  ]
    .filter(Boolean)
    .join(' ');

  return (
    <nav
      ref={containerRef}
      className={navClasses}
      style={style}
      aria-label={ariaLabel ?? navLabel}
      data-testid={dataTestId}
    >
      <ol className={listClasses}>
        {visibleItems.map((item, idx) => {
          const isLastInVisible = idx === visibleItems.length - 1;

          // If this is index 1 and we have collapsed items, render ellipsis first
          if (idx === 1 && collapsedItems.length > 0) {
            return (
              <React.Fragment key="collapsed">
                {/* Separator before ellipsis */}
                <li className={styles.item} aria-hidden="true">
                  {Sep}
                </li>

                {/* Collapsed trigger */}
                <li className={styles.item}>
                  <div className={styles.collapsedWrapper} ref={dropdownRef}>
                    {renderCollapsedTrigger ? (
                      renderCollapsedTrigger(collapsedItems)
                    ) : (
                      <button
                        type="button"
                        className={styles.collapsedTrigger}
                        onClick={() => setDropdownOpen(!dropdownOpen)}
                        aria-label={`Show ${collapsedItems.length} more breadcrumb items`}
                        aria-expanded={dropdownOpen}
                        aria-haspopup="listbox"
                      >
                        …
                      </button>
                    )}

                    {dropdownOpen && (
                      <div className={styles.dropdown} role="listbox">
                        {collapsedItems.map((ci) => (
                          <button
                            key={ci.key}
                            type="button"
                            className={styles.dropdownItem}
                            onClick={() => {
                              handleItemClick(ci);
                              setDropdownOpen(false);
                            }}
                            role="option"
                          >
                            {ci.icon && <span className={styles.icon}>{ci.icon}</span>}
                            {ci.label}
                          </button>
                        ))}
                      </div>
                    )}
                  </div>
                </li>

                {/* Separator after ellipsis */}
                <li className={styles.item} aria-hidden="true">
                  {Sep}
                </li>

                {/* Continue with current item */}
                <li key={item.key} className={styles.item}>
                  {defaultRenderItem(item, idx)}
                </li>

                {!isLastInVisible && (
                  <li className={styles.item} aria-hidden="true">
                    {Sep}
                  </li>
                )}
              </React.Fragment>
            );
          }

          return (
            <React.Fragment key={item.key}>
              <li className={styles.item}>
                {defaultRenderItem(item, idx)}
              </li>
              {!isLastInVisible && (
                <li className={styles.item} aria-hidden="true">
                  {Sep}
                </li>
              )}
            </React.Fragment>
          );
        })}
      </ol>
    </nav>
  );
}

export const BreadcrumbNav = React.memo(BreadcrumbNavInner) as typeof BreadcrumbNavInner;
export type { BreadcrumbNavProps, BreadcrumbItem, BreadcrumbCollapseMode, BreadcrumbItemRenderProps };
```
